45impl StoredExcerptRequest {
46    fn resolve(&self, file_end_line: Option<usize>) -> Option<ResolvedStoredExcerptRequest> {
47        let file_end_line = file_end_line?;
48        let required_start = self.required_start_line.max(1);
49        if required_start > file_end_line {
50            return None;
51        }
52        let required_end = self
53            .required_end_line
54            .max(required_start)
55            .min(file_end_line);
56        let desired_start = self.desired_start_line.max(1).min(file_end_line);
57        let desired_end = self.desired_end_line.max(desired_start).min(file_end_line);
58        let (start_line, end_line) = anchored_line_window(
59            desired_start,
60            desired_end,
61            required_start,
62            required_end,
63            self.max_lines,
64        );
65        Some(ResolvedStoredExcerptRequest {
66            file_id: self.file_id,
67            start_line,
68            end_line,
69        })
70    }
71}

94fn assemble_stored_excerpt(
95    request: ResolvedStoredExcerptRequest,
96    selected: &[crate::storage::ChunkRecord],
97) -> Option<StoredExcerpt> {
98    let first_chunk = selected.first()?;
99    let base_line = first_chunk.start_line;
100    let mut combined = String::new();
101    for chunk in selected {
102        combined.push_str(&chunk.content);
103    }
104    let local_start = request.start_line.saturating_sub(base_line) + 1;
105    let local_end = request.end_line.saturating_sub(base_line) + 1;
106    Some(StoredExcerpt {
107        content: crate::text::excerpt(&combined, local_start, local_end),
108        start_line: request.start_line,
109        end_line: request.end_line,
110    })
111}

126fn validate_outline_input(request: &OutlineRequest) -> Result<()> {
127    if request.paths.is_empty() {
128        return Err(Error::InvalidInput {
129            field: "paths",
130            reason: "must contain at least one path",
131        });
132    }
133    if request.paths.len() > MAX_INPUT_ITEMS {
134        return Err(Error::LimitExceeded);
135    }
136    for path in &request.paths {
137        validate_input(path, "path", MAX_PATH_BYTES)?;
138        validate_relative(path)?;
139    }
140    validate_optional_input(
141        request.symbol_name.as_deref(),
142        "symbol name",
143        MAX_PATTERN_BYTES,
144    )?;
145    validate_optional_input(
146        request.symbol_kind.as_deref(),
147        "symbol kind",
148        MAX_PATTERN_BYTES,
149    )?;
150    Ok(())
151}

153fn validate_read_input(request: &ReadRequest) -> Result<()> {
154    validate_input(&request.path, "path", MAX_PATH_BYTES)?;
155    if request.symbol.as_deref().is_some_and(str::is_empty) {
156        return Err(Error::InvalidInput {
157            field: "symbol",
158            reason: "must not be empty",
159        });
160    }
161    validate_optional_input(request.symbol.as_deref(), "symbol", MAX_PATTERN_BYTES)?;
162    validate_optional_input(request.expected_hash.as_deref(), "expected hash", 128)?;
163    validate_relative(&request.path)?;
164    if request.symbol.is_some() && (request.start_line.is_some() || request.end_line.is_some()) {
165        return Err(Error::InvalidInput {
166            field: "read target",
167            reason: "must use either a symbol or line range, not both",
168        });
169    }
170    if request.symbol.is_none() {
171        let start_line = request.start_line.unwrap_or(1);
172        if start_line == 0
173            || request
174                .end_line
175                .is_some_and(|end_line| end_line < start_line)
176        {
177            return Err(invalid_line_range());
178        }
179    }
180    Ok(())
181}

243    fn outline_sync(
244        &self,
245        request: OutlineRequest,
246        cancellation: &CancellationToken,
247    ) -> Result<OutlineResponse> {
248        check_cancelled(cancellation)?;
249        validate_outline_input(&request)?;
250        let limit = self.result_limit(request.max_results)?;
251        let token_limit = self.token_limit(request.max_tokens, self.config.default_read_tokens)?;
252        let (response, baseline_source_tokens) = self.consistent(|session, generation| {
253            let mut remaining = limit;
254            let mut emitted_tokens = 0usize;
255            let mut files = Vec::new();
256            for path in &request.paths {
257                check_cancelled(cancellation)?;
258                let file = session
259                    .find_file(path)?
260                    .ok_or_else(|| Error::NotIndexed(path.clone()))?;
261                let mut symbols = session
262                    .get_symbols_for_file_filtered(
263                        file.id,
264                        request.symbol_name.as_deref(),
265                        request.symbol_kind.as_deref(),
266                        remaining.max(1),
267                    )?
268                    .into_iter()
269                    .map(storage_symbol)
270                    .collect::<Vec<_>>();
271                symbols.retain(|symbol| {
272                    let cost = symbol
273                        .signature
274                        .as_deref()
275                        .map_or(1, |value| self.config.tokenizer.count(value));
276                    if remaining == 0 || emitted_tokens.saturating_add(cost) > token_limit {
277                        false
278                    } else {
279                        remaining -= 1;
280                        emitted_tokens += cost;
281                        true
282                    }
283                });
284                let mut imports = session
285                    .get_imports_for_file(file.id, limit)?
286                    .into_iter()
287                    .map(|import| Import {
288                        raw_target: import.raw_target,
289                        resolved_path: import.resolved_path,
290                        line: import.line,
291                    })
292                    .collect::<Vec<_>>();
293                imports.retain(|import| {
294                    let cost = self.config.tokenizer.count(&import.raw_target)
295                        + import
296                            .resolved_path
297                            .as_deref()
298                            .map_or(0, |value| self.config.tokenizer.count(value));
299                    if remaining == 0 || emitted_tokens.saturating_add(cost) > token_limit {
300                        false
301                    } else {
302                        remaining -= 1;
303                        emitted_tokens += cost;
304                        true
305                    }
306                });
307                files.push(OutlineFile {
308                    path: file.path,
309                    language: file.language.clone(),
310                    structurally_complete: file.structurally_complete,
311                    symbols,
312                    imports,
313                });
314                if remaining == 0 {
315                    break;
316                }
317            }
318            let paths = files
319                .iter()
320                .map(|file| file.path.clone())
321                .collect::<Vec<_>>();
322            let baseline_source_tokens =
323                session.whole_file_source_tokens(&paths, self.config.tokenizer.name())?;
324            Ok((
325                OutlineResponse {
326                    files,
327                    meta: self.meta(generation, emitted_tokens, None),
328                },
329                baseline_source_tokens,
330            ))
331        })?;
332        if let Some(baseline_source_tokens) = baseline_source_tokens {
333            self.record_token_savings(
334                TokenSavingsOperation::Outline,
335                baseline_source_tokens,
336                response.meta.emitted_tokens,
337            );
338        }
339        Ok(response)
340    }

342    fn read_sync(
343        &self,
344        request: ReadRequest,
345        cancellation: &CancellationToken,
346    ) -> Result<ReadResponse> {
347        check_cancelled(cancellation)?;
348        validate_read_input(&request)?;
349        let max_tokens = self.token_limit(request.max_tokens, self.config.default_read_tokens)?;
350        let (response, baseline_source_tokens) = self.consistent(|session, generation| {
351            check_cancelled(cancellation)?;
352            self.read_at_generation(session, &request, generation, max_tokens)
353        })?;
354        self.record_token_savings(
355            TokenSavingsOperation::Read,
356            baseline_source_tokens,
357            response.meta.emitted_tokens,
358        );
359        Ok(response)
360    }

362    fn read_at_generation(
363        &self,
364        session: &ReadSession,
365        request: &ReadRequest,
366        generation: u64,
367        max_tokens: usize,
368    ) -> Result<(ReadResponse, usize)> {
369        let indexed = session
370            .find_file(&request.path)?
371            .ok_or_else(|| Error::NotIndexed(request.path.clone()))?;
372        let path = resolve_existing(&self.config.root, &request.path)?;
373        let target = resolve_read_target(session, indexed.id, request)?;
374
375        // Stream the file through a BufReader for the full-file hash so the
376        // entire file does not need to be held in memory simultaneously. The
377        // content range is extracted by a bounded line-oriented reader.
378        let file = fs::File::open(&path)?;
379        let full_hash = stream_hash(&file)?;
380        let range = read_live_range(&file, target)?;
381        let baseline_source_tokens = self.config.tokenizer.count(&range.content);
382        let start_line = range.start_line;
383        let (content, emitted_tokens) = self.config.tokenizer.truncate(&range.content, max_tokens);
384        let returned_lines = content
385            .lines()
386            .count()
387            .max(usize::from(!content.is_empty()));
388        let end_line = if returned_lines == 0 {
389            start_line
390        } else {
391            (start_line + returned_lines - 1).min(range.end_line)
392        };
393        let content_hash = hash(content);
394        let index_stale = indexed.content_hash != full_hash;
395        let indexed_hash = Some(indexed.content_hash);
396        let not_modified = request.expected_hash.as_deref() == Some(content_hash.as_str());
397
398        Ok((
399            ReadResponse {
400                path: request.path.clone(),
401                status: if not_modified {
402                    ReadStatus::NotModified
403                } else {
404                    ReadStatus::Content
405                },
406                start_line,
407                end_line,
408                content: (!not_modified).then(|| content.to_string()),
409                content_hash,
410                indexed_hash,
411                index_stale,
412                meta: self.meta(
413                    generation,
414                    if not_modified { 0 } else { emitted_tokens },
415                    None,
416                ),
417            },
418            baseline_source_tokens,
419        ))
420    }

423fn resolve_read_target(
424    session: &ReadSession,
425    file_id: i64,
426    request: &ReadRequest,
427) -> Result<ResolvedReadTarget> {
428    let target = if let Some(symbol_name) = &request.symbol {
429        let symbol = session
430            .find_symbol(file_id, symbol_name)?
431            .ok_or_else(|| Error::NotIndexed(format!("{}::{symbol_name}", request.path)))?;
432        ResolvedReadTarget {
433            start_line: symbol.start_line,
434            end_line: Some(symbol.end_line),
435        }
436    } else {
437        ResolvedReadTarget {
438            start_line: request.start_line.unwrap_or(1),
439            end_line: request.end_line,
440        }
441    };
442
443    if target.start_line == 0
444        || target
445            .end_line
446            .is_some_and(|end_line| end_line < target.start_line)
447    {
448        return Err(invalid_line_range());
449    }
450    Ok(target)
451}

453/// Stream a file through a BufReader and compute the content hash without
454/// loading the entire file into memory.
455fn stream_hash(file: &File) -> Result<String> {
456    let mut reader = BufReader::new(file.try_clone()?);
457    let mut hasher = blake3::Hasher::new();
458    let mut buf = [0u8; 65_536];
459    loop {
460        let n = reader.read(&mut buf)?;
461        if n == 0 {
462            break;
463        }
464        hasher.update(&buf[..n]);
465    }
466    Ok(hasher.finalize().to_hex()[..crate::text::CONTENT_FINGERPRINT_HEX_LEN].to_string())
467}

469/// Read a resolved range without changing its original line terminators.
470fn read_live_range(file: &File, target: ResolvedReadTarget) -> Result<LiveReadRange> {
471    let mut file = file.try_clone()?;
472    file.seek(SeekFrom::Start(0))?;
473    let mut reader = BufReader::new(file);
474    let mut selected = Vec::new();
475    let mut line = Vec::new();
476    let mut current_line = 0usize;
477    let mut selected_end = None;
478
479    loop {
480        line.clear();
481        if reader.read_until(b'\n', &mut line)? == 0 {
482            break;
483        }
484        current_line += 1;
485        if current_line >= target.start_line
486            && target
487                .end_line
488                .is_none_or(|end_line| current_line <= end_line)
489        {
490            selected.extend_from_slice(&line);
491            selected_end = Some(current_line);
492        }
493        if target.end_line == Some(current_line) {
494            break;
495        }
496    }
497
498    let logical_line_count = current_line.max(1);
499    if target.start_line > logical_line_count {
500        return Err(invalid_line_range());
501    }
502    let content = String::from_utf8(selected).map_err(|_| Error::InvalidInput {
503        field: "path",
504        reason: "must identify UTF-8 text",
505    })?;
506    Ok(LiveReadRange {
507        content,
508        start_line: target.start_line,
509        end_line: selected_end.unwrap_or(target.start_line),
510    })
511}

548    pub(super) fn stored_excerpts(
549        &self,
550        session: &ReadSession,
551        requests: &[StoredExcerptRequest],
552    ) -> Result<Vec<Option<StoredExcerpt>>> {
553        let file_ids = requests
554            .iter()
555            .map(|request| request.file_id)
556            .collect::<Vec<_>>();
557        let file_end_lines = session.file_end_lines_batch(&file_ids)?;
558        let mut resolved = Vec::new();
559        let mut ranges = Vec::new();
560        for (index, (request, file_end_line)) in requests.iter().zip(file_end_lines).enumerate() {
561            let Some(request) = request.resolve(file_end_line) else {
562                continue;
563            };
564            ranges.push((request.file_id, request.start_line, request.end_line));
565            resolved.push((index, request));
566        }
567        let chunks = session.get_chunks_overlapping_batch(&ranges)?;
568        let mut excerpts = vec![None; requests.len()];
569        for ((index, request), chunks) in resolved.into_iter().zip(chunks) {
570            excerpts[index] = assemble_stored_excerpt(request, &chunks);
571        }
572        Ok(excerpts)
573    }

575    #[cfg(test)]
576    pub(super) fn adaptive_context_excerpt(
577        &self,
578        session: &ReadSession,
579        file_id: i64,
580        declaration_start: usize,
581        declaration_end: usize,
582        matched_line: usize,
583        token_budget: usize,
584    ) -> Result<Option<StoredExcerpt>> {
585        let Some(full) =
586            self.stored_excerpt(session, file_id, declaration_start, declaration_end, 0, 0)?
587        else {
588            return Ok(None);
589        };
590        let full_tokens = self.config.tokenizer.count(&full.content).max(1);
591        if full_tokens <= token_budget {
592            return Ok(Some(full));
593        }
594
595        let declaration_lines = declaration_end
596            .saturating_sub(declaration_start)
597            .saturating_add(1);
598        let proportional_lines = declaration_lines
599            .saturating_mul(token_budget)
600            .saturating_div(full_tokens)
601            .clamp(MIN_CONTEXT_RANGE_LINES, MAX_CONTEXT_RANGE_LINES)
602            .min(declaration_lines);
603        let before = proportional_lines / 3;
604        let mut start = matched_line.saturating_sub(before).max(declaration_start);
605        let mut end = start
606            .saturating_add(proportional_lines.saturating_sub(1))
607            .min(declaration_end);
608        if end.saturating_sub(start).saturating_add(1) < proportional_lines {
609            start = end
610                .saturating_add(1)
611                .saturating_sub(proportional_lines)
612                .max(declaration_start);
613        }
614        end = start
615            .saturating_add(proportional_lines.saturating_sub(1))
616            .min(declaration_end);
617        self.stored_excerpt(session, file_id, start, end, 0, 0)
618    }

620    pub(super) fn adaptive_context_excerpts(
621        &self,
622        session: &ReadSession,
623        requests: &[AdaptiveExcerptRequest],
624    ) -> Result<Vec<Option<StoredExcerpt>>> {
625        let full_requests = requests
626            .iter()
627            .map(|request| StoredExcerptRequest {
628                file_id: request.file_id,
629                desired_start_line: request.declaration_start,
630                desired_end_line: request.declaration_end,
631                required_start_line: request.matched_line,
632                required_end_line: request.matched_line,
633                max_lines: 0,
634            })
635            .collect::<Vec<_>>();
636        let mut excerpts = self.stored_excerpts(session, &full_requests)?;
637        let mut narrowed_indices = Vec::new();
638        let mut narrowed_requests = Vec::new();
639        for (index, (request, excerpt)) in requests.iter().zip(&excerpts).enumerate() {
640            let Some(excerpt) = excerpt else {
641                continue;
642            };
643            let full_tokens = self.config.tokenizer.count(&excerpt.content).max(1);
644            if full_tokens <= request.token_budget {
645                continue;
646            }
647            let declaration_lines = request
648                .declaration_end
649                .saturating_sub(request.declaration_start)
650                .saturating_add(1);
651            let proportional_lines = declaration_lines
652                .saturating_mul(request.token_budget)
653                .saturating_div(full_tokens)
654                .clamp(MIN_CONTEXT_RANGE_LINES, MAX_CONTEXT_RANGE_LINES)
655                .min(declaration_lines);
656            let before = proportional_lines / 3;
657            let mut start = request
658                .matched_line
659                .saturating_sub(before)
660                .max(request.declaration_start);
661            let mut end = start
662                .saturating_add(proportional_lines.saturating_sub(1))
663                .min(request.declaration_end);
664            if end.saturating_sub(start).saturating_add(1) < proportional_lines {
665                start = end
666                    .saturating_add(1)
667                    .saturating_sub(proportional_lines)
668                    .max(request.declaration_start);
669            }
670            end = start
671                .saturating_add(proportional_lines.saturating_sub(1))
672                .min(request.declaration_end);
673            narrowed_indices.push(index);
674            narrowed_requests.push(StoredExcerptRequest {
675                file_id: request.file_id,
676                desired_start_line: start,
677                desired_end_line: end,
678                required_start_line: request.matched_line,
679                required_end_line: request.matched_line,
680                max_lines: 0,
681            });
682        }
683        let narrowed = self.stored_excerpts(session, &narrowed_requests)?;
684        for (index, excerpt) in narrowed_indices.into_iter().zip(narrowed) {
685            excerpts[index] = excerpt;
686        }
687        Ok(excerpts)
688    }
689}
