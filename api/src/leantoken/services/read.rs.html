49impl StoredExcerptRequest {
50    fn resolve(&self, file_end_line: Option<usize>) -> Option<ResolvedStoredExcerptRequest> {
51        let file_end_line = file_end_line?;
52        let required_start = self.required_start_line.max(1);
53        if required_start > file_end_line {
54            return None;
55        }
56        let required_end = self
57            .required_end_line
58            .max(required_start)
59            .min(file_end_line);
60        let desired_start = self.desired_start_line.max(1).min(file_end_line);
61        let desired_end = self.desired_end_line.max(desired_start).min(file_end_line);
62        let (start_line, end_line) = anchored_line_window(
63            desired_start,
64            desired_end,
65            required_start,
66            required_end,
67            self.max_lines,
68        );
69        Some(ResolvedStoredExcerptRequest {
70            file_id: self.file_id,
71            start_line,
72            end_line,
73        })
74    }
75}

98fn assemble_stored_excerpt(
99    request: ResolvedStoredExcerptRequest,
100    selected: &[crate::storage::ChunkRecord],
101) -> Option<StoredExcerpt> {
102    let first_chunk = selected.first()?;
103    let base_line = first_chunk.start_line;
104    let mut combined = String::new();
105    for chunk in selected {
106        combined.push_str(&chunk.content);
107    }
108    let local_start = request.start_line.saturating_sub(base_line) + 1;
109    let local_end = request.end_line.saturating_sub(base_line) + 1;
110    Some(StoredExcerpt {
111        content: crate::text::excerpt(&combined, local_start, local_end),
112        start_line: request.start_line,
113        end_line: request.end_line,
114    })
115}

130fn validate_outline_input(request: &OutlineRequest) -> Result<()> {
131    if request.paths.is_empty() {
132        return Err(Error::InvalidInput {
133            field: "paths",
134            reason: "must contain at least one path",
135        });
136    }
137    if request.paths.len() > MAX_INPUT_ITEMS {
138        return Err(Error::LimitExceeded);
139    }
140    for path in &request.paths {
141        validate_input(path, "path", MAX_PATH_BYTES)?;
142        validate_relative(path)?;
143    }
144    validate_optional_input(
145        request.symbol_name.as_deref(),
146        "symbol name",
147        MAX_PATTERN_BYTES,
148    )?;
149    validate_optional_input(
150        request.symbol_kind.as_deref(),
151        "symbol kind",
152        MAX_PATTERN_BYTES,
153    )?;
154    Ok(())
155}

157fn validate_read_input(request: &ReadRequest) -> Result<()> {
158    validate_input(&request.path, "path", MAX_PATH_BYTES)?;
159    if request.symbol.as_deref().is_some_and(str::is_empty) {
160        return Err(Error::InvalidInput {
161            field: "symbol",
162            reason: "must not be empty",
163        });
164    }
165    validate_optional_input(request.symbol.as_deref(), "symbol", MAX_PATTERN_BYTES)?;
166    validate_optional_input(request.expected_hash.as_deref(), "expected hash", 128)?;
167    validate_relative(&request.path)?;
168    if request.symbol.is_some() && (request.start_line.is_some() || request.end_line.is_some()) {
169        return Err(Error::InvalidInput {
170            field: "read target",
171            reason: "must use either a symbol or line range, not both",
172        });
173    }
174    if request.symbol.is_none() {
175        let start_line = request.start_line.unwrap_or(1);
176        if start_line == 0
177            || request
178                .end_line
179                .is_some_and(|end_line| end_line < start_line)
180        {
181            return Err(invalid_line_range());
182        }
183    }
184    Ok(())
185}

247    fn outline_sync(
248        &self,
249        mut request: OutlineRequest,
250        cancellation: &CancellationToken,
251    ) -> Result<OutlineResponse> {
252        check_cancelled(cancellation)?;
253        validate_outline_input(&request)?;
254        request.paths = request
255            .paths
256            .iter()
257            .map(|path| normalize_relative(path))
258            .collect::<Result<Vec<_>>>()?;
259        let limit = self.result_limit(request.max_results)?;
260        let token_limit = self.token_limit(request.max_tokens, self.config.default_read_tokens)?;
261        let (response, baseline_source_tokens) = self.consistent(|session, generation| {
262            let mut remaining = limit;
263            let mut emitted_tokens = 0usize;
264            let mut files = Vec::new();
265            for path in &request.paths {
266                check_cancelled(cancellation)?;
267                let file = session
268                    .find_file(path)?
269                    .ok_or_else(|| Error::NotIndexed(path.clone()))?;
270                let mut symbols = session
271                    .get_symbols_for_file_filtered(
272                        file.id,
273                        request.symbol_name.as_deref(),
274                        request.symbol_kind.as_deref(),
275                        remaining.max(1),
276                    )?
277                    .into_iter()
278                    .map(storage_symbol)
279                    .collect::<Vec<_>>();
280                symbols.retain(|symbol| {
281                    let cost = symbol
282                        .signature
283                        .as_deref()
284                        .map_or(1, |value| self.config.tokenizer.count(value));
285                    if remaining == 0 || emitted_tokens.saturating_add(cost) > token_limit {
286                        false
287                    } else {
288                        remaining -= 1;
289                        emitted_tokens += cost;
290                        true
291                    }
292                });
293                let mut imports = session
294                    .get_imports_for_file(file.id, limit)?
295                    .into_iter()
296                    .map(|import| Import {
297                        raw_target: import.raw_target,
298                        resolved_path: import.resolved_path,
299                        line: import.line,
300                    })
301                    .collect::<Vec<_>>();
302                imports.retain(|import| {
303                    let cost = self.config.tokenizer.count(&import.raw_target)
304                        + import
305                            .resolved_path
306                            .as_deref()
307                            .map_or(0, |value| self.config.tokenizer.count(value));
308                    if remaining == 0 || emitted_tokens.saturating_add(cost) > token_limit {
309                        false
310                    } else {
311                        remaining -= 1;
312                        emitted_tokens += cost;
313                        true
314                    }
315                });
316                files.push(OutlineFile {
317                    path: file.path,
318                    language: file.language.clone(),
319                    structurally_complete: file.structurally_complete,
320                    symbols,
321                    imports,
322                });
323                if remaining == 0 {
324                    break;
325                }
326            }
327            let paths = files
328                .iter()
329                .map(|file| file.path.clone())
330                .collect::<Vec<_>>();
331            let baseline_source_tokens =
332                session.whole_file_source_tokens(&paths, self.config.tokenizer.name())?;
333            Ok((
334                OutlineResponse {
335                    files,
336                    meta: self.meta(generation, emitted_tokens, None),
337                },
338                baseline_source_tokens,
339            ))
340        })?;
341        if let Some(baseline_source_tokens) = baseline_source_tokens {
342            self.record_token_savings(
343                TokenSavingsOperation::Outline,
344                baseline_source_tokens,
345                response.meta.emitted_tokens,
346            );
347        }
348        Ok(response)
349    }

351    fn read_sync(
352        &self,
353        mut request: ReadRequest,
354        cancellation: &CancellationToken,
355    ) -> Result<ReadResponse> {
356        check_cancelled(cancellation)?;
357        validate_read_input(&request)?;
358        request.path = normalize_relative(&request.path)?;
359        let max_tokens = self.token_limit(request.max_tokens, self.config.default_read_tokens)?;
360        let (response, baseline_source_tokens) = self.consistent(|session, generation| {
361            check_cancelled(cancellation)?;
362            self.read_at_generation(session, &request, generation, max_tokens)
363        })?;
364        self.record_token_savings(
365            TokenSavingsOperation::Read,
366            baseline_source_tokens,
367            response.meta.emitted_tokens,
368        );
369        Ok(response)
370    }

372    fn read_at_generation(
373        &self,
374        session: &ReadSession,
375        request: &ReadRequest,
376        generation: u64,
377        max_tokens: usize,
378    ) -> Result<(ReadResponse, usize)> {
379        let indexed = session
380            .find_file(&request.path)?
381            .ok_or_else(|| Error::NotIndexed(request.path.clone()))?;
382        let path = resolve_existing(&self.config.root, &request.path)?;
383        let target = resolve_read_target(session, indexed.id, request)?;
384
385        // Stream the file through a BufReader for the full-file hash so the
386        // entire file does not need to be held in memory simultaneously. The
387        // content range is extracted by a bounded line-oriented reader.
388        let file = fs::File::open(&path)?;
389        let full_hash = stream_hash(&file)?;
390        let range = read_live_range(&file, target, max_tokens, self.config.tokenizer)?;
391        let baseline_source_tokens = self.config.tokenizer.count(&range.content);
392        let start_line = range.start_line;
393        let (content, emitted_tokens) = self.config.tokenizer.truncate(&range.content, max_tokens);
394        let returned_lines = content
395            .lines()
396            .count()
397            .max(usize::from(!content.is_empty()));
398        let end_line = if returned_lines == 0 {
399            start_line
400        } else {
401            (start_line + returned_lines - 1).min(range.end_line)
402        };
403        let content_hash = hash(content);
404        let index_stale = indexed.content_hash != full_hash;
405        let indexed_hash = Some(indexed.content_hash);
406        let not_modified = request.expected_hash.as_deref() == Some(content_hash.as_str());
407
408        Ok((
409            ReadResponse {
410                path: request.path.clone(),
411                status: if not_modified {
412                    ReadStatus::NotModified
413                } else {
414                    ReadStatus::Content
415                },
416                start_line,
417                end_line,
418                content: (!not_modified).then(|| content.to_string()),
419                content_hash,
420                indexed_hash,
421                index_stale,
422                meta: self.meta(
423                    generation,
424                    if not_modified { 0 } else { emitted_tokens },
425                    None,
426                ),
427            },
428            baseline_source_tokens,
429        ))
430    }

433fn resolve_read_target(
434    session: &ReadSession,
435    file_id: i64,
436    request: &ReadRequest,
437) -> Result<ResolvedReadTarget> {
438    let target = if let Some(symbol_name) = &request.symbol {
439        let symbol = session
440            .find_symbol(file_id, symbol_name)?
441            .ok_or_else(|| Error::NotIndexed(format!("{}::{symbol_name}", request.path)))?;
442        ResolvedReadTarget {
443            start_line: symbol.start_line,
444            end_line: Some(symbol.end_line),
445        }
446    } else {
447        ResolvedReadTarget {
448            start_line: request.start_line.unwrap_or(1),
449            end_line: request.end_line,
450        }
451    };
452
453    if target.start_line == 0
454        || target
455            .end_line
456            .is_some_and(|end_line| end_line < target.start_line)
457    {
458        return Err(invalid_line_range());
459    }
460    Ok(target)
461}

463/// Stream a file through a BufReader and compute the content hash without
464/// loading the entire file into memory.
465fn stream_hash(file: &File) -> Result<String> {
466    let mut reader = BufReader::new(file.try_clone()?);
467    let mut hasher = blake3::Hasher::new();
468    let mut buf = [0u8; 65_536];
469    loop {
470        let n = reader.read(&mut buf)?;
471        if n == 0 {
472            break;
473        }
474        hasher.update(&buf[..n]);
475    }
476    Ok(hasher.finalize().to_hex()[..crate::text::CONTENT_FINGERPRINT_HEX_LEN].to_string())
477}

479/// Read a resolved range without changing its original line terminators.
480fn read_live_range(
481    file: &File,
482    target: ResolvedReadTarget,
483    max_tokens: usize,
484    tokenizer: crate::tokens::Tokenizer,
485) -> Result<LiveReadRange> {
486    let mut file = file.try_clone()?;
487    file.seek(SeekFrom::Start(0))?;
488    let mut reader = BufReader::new(file);
489    let mut selected = Vec::with_capacity(LIVE_READ_TOKEN_CHECK_BYTES);
490    let mut current_line = 1usize;
491    let mut selected_end = None;
492    let mut target_finished = false;
493    let mut token_bound_reached = false;
494    let mut last_byte_was_newline = false;
495    let mut next_token_check = LIVE_READ_TOKEN_CHECK_BYTES;
496    let mut utf8_pending = Vec::new();
497
498    while !target_finished {
499        let buffer = reader.fill_buf()?;
500        if buffer.is_empty() {
501            break;
502        }
503
504        let mut consumed = 0usize;
505        let mut validation_chunk = Vec::new();
506        for &byte in buffer {
507            let in_target = current_line >= target.start_line
508                && target
509                    .end_line
510                    .is_none_or(|end_line| current_line <= end_line);
511            if in_target {
512                validation_chunk.push(byte);
513            }
514            if in_target && !token_bound_reached {
515                selected_end = Some(current_line);
516                selected.push(byte);
517            }
518            consumed += 1;
519            last_byte_was_newline = byte == b'\n';
520            if byte == b'\n' {
521                if target.end_line == Some(current_line) {
522                    target_finished = true;
523                    break;
524                }
525                current_line = current_line.saturating_add(1);
526            }
527        }
528        reader.consume(consumed);
529        validate_utf8_chunk(&mut utf8_pending, &validation_chunk, target_finished)?;
530
531        if !token_bound_reached
532            && (target_finished
533                || selected.len() >= next_token_check
534                || selected.len() >= MAX_LIVE_READ_BYTES)
535        {
536            match std::str::from_utf8(&selected) {
537                Ok(content) if tokenizer.count(content) > max_tokens => {
538                    token_bound_reached = true;
539                }
540                Ok(_) => {
541                    if selected.len() >= MAX_LIVE_READ_BYTES {
542                        return Err(Error::LimitExceeded);
543                    }
544                    next_token_check = selected.len().saturating_add(LIVE_READ_TOKEN_CHECK_BYTES);
545                }
546                Err(error) if error.error_len().is_none() => {
547                    if target_finished || selected.len() >= MAX_LIVE_READ_BYTES {
548                        return Err(Error::InvalidInput {
549                            field: "path",
550                            reason: "must identify UTF-8 text",
551                        });
552                    }
553                }
554                Err(_) => {
555                    return Err(Error::InvalidInput {
556                        field: "path",
557                        reason: "must identify UTF-8 text",
558                    });
559                }
560            }
561        }
562        if token_bound_reached && target.end_line.is_none() {
563            target_finished = true;
564        }
565    }
566
567    if !utf8_pending.is_empty() {
568        return Err(Error::InvalidInput {
569            field: "path",
570            reason: "must identify UTF-8 text",
571        });
572    }
573
574    let logical_line_count = if current_line == 1 || last_byte_was_newline {
575        current_line.saturating_sub(usize::from(current_line > 1))
576    } else {
577        current_line
578    };
579    if selected_end.is_none() && target.start_line > logical_line_count.max(1) {
580        return Err(invalid_line_range());
581    }
582    let content = String::from_utf8(selected).map_err(|_| Error::InvalidInput {
583        field: "path",
584        reason: "must identify UTF-8 text",
585    })?;
586    Ok(LiveReadRange {
587        content,
588        start_line: target.start_line,
589        end_line: selected_end
590            .unwrap_or(target.start_line)
591            .min(target.end_line.unwrap_or(usize::MAX)),
592    })
593}

595fn validate_utf8_chunk(pending: &mut Vec<u8>, bytes: &[u8], final_chunk: bool) -> Result<()> {
596    pending.extend_from_slice(bytes);
597    match std::str::from_utf8(pending) {
598        Ok(_) => {
599            pending.clear();
600            Ok(())
601        }
602        Err(error) if error.error_len().is_none() && !final_chunk => {
603            let valid_up_to = error.valid_up_to();
604            pending.drain(..valid_up_to);
605            Ok(())
606        }
607        Err(_) => Err(Error::InvalidInput {
608            field: "path",
609            reason: "must identify UTF-8 text",
610        }),
611    }
612}

648impl Services {
649    pub(super) fn stored_excerpts(
650        &self,
651        session: &ReadSession,
652        requests: &[StoredExcerptRequest],
653    ) -> Result<Vec<Option<StoredExcerpt>>> {
654        let file_ids = requests
655            .iter()
656            .map(|request| request.file_id)
657            .collect::<Vec<_>>();
658        let file_end_lines = session.file_end_lines_batch(&file_ids)?;
659        let mut resolved = Vec::new();
660        let mut ranges = Vec::new();
661        for (index, (request, file_end_line)) in requests.iter().zip(file_end_lines).enumerate() {
662            let Some(request) = request.resolve(file_end_line) else {
663                continue;
664            };
665            ranges.push((request.file_id, request.start_line, request.end_line));
666            resolved.push((index, request));
667        }
668        let chunks = session.get_chunks_overlapping_batch(&ranges)?;
669        let mut excerpts = vec![None; requests.len()];
670        for ((index, request), chunks) in resolved.into_iter().zip(chunks) {
671            excerpts[index] = assemble_stored_excerpt(request, &chunks);
672        }
673        Ok(excerpts)
674    }

685    ) -> Result<Option<StoredExcerpt>> {
686        let Some(full) =
687            self.stored_excerpt(session, file_id, declaration_start, declaration_end, 0, 0)?
688        else {
689            return Ok(None);
690        };
691        let full_tokens = self.config.tokenizer.count(&full.content).max(1);
692        if full_tokens <= token_budget {
693            return Ok(Some(full));
694        }
695
696        let declaration_lines = declaration_end
697            .saturating_sub(declaration_start)
698            .saturating_add(1);
699        let proportional_lines = declaration_lines
700            .saturating_mul(token_budget)
701            .saturating_div(full_tokens)
702            .clamp(MIN_CONTEXT_RANGE_LINES, MAX_CONTEXT_RANGE_LINES)
703            .min(declaration_lines);
704        let before = proportional_lines / 3;
705        let mut start = matched_line.saturating_sub(before).max(declaration_start);
706        let mut end = start
707            .saturating_add(proportional_lines.saturating_sub(1))
708            .min(declaration_end);
709        if end.saturating_sub(start).saturating_add(1) < proportional_lines {
710            start = end
711                .saturating_add(1)
712                .saturating_sub(proportional_lines)
713                .max(declaration_start);
714        }
715        end = start
716            .saturating_add(proportional_lines.saturating_sub(1))
717            .min(declaration_end);
718        self.stored_excerpt(session, file_id, start, end, 0, 0)
719    }
720
721    pub(super) fn adaptive_context_excerpts(
722        &self,
723        session: &ReadSession,
724        requests: &[AdaptiveExcerptRequest],
725    ) -> Result<Vec<Option<StoredExcerpt>>> {
726        let full_requests = requests
727            .iter()
728            .map(|request| StoredExcerptRequest {
729                file_id: request.file_id,
730                desired_start_line: request.declaration_start,
731                desired_end_line: request.declaration_end,
732                required_start_line: request.matched_line,
733                required_end_line: request.matched_line,
734                max_lines: 0,
735            })
736            .collect::<Vec<_>>();
737        let mut excerpts = self.stored_excerpts(session, &full_requests)?;
738        let mut narrowed_indices = Vec::new();
739        let mut narrowed_requests = Vec::new();
740        for (index, (request, excerpt)) in requests.iter().zip(&excerpts).enumerate() {
741            let Some(excerpt) = excerpt else {
742                continue;
743            };
744            let full_tokens = self.config.tokenizer.count(&excerpt.content).max(1);
745            if full_tokens <= request.token_budget {
746                continue;
747            }
748            let declaration_lines = request
749                .declaration_end
750                .saturating_sub(request.declaration_start)
751                .saturating_add(1);
752            let proportional_lines = declaration_lines
753                .saturating_mul(request.token_budget)
754                .saturating_div(full_tokens)
755                .clamp(MIN_CONTEXT_RANGE_LINES, MAX_CONTEXT_RANGE_LINES)
756                .min(declaration_lines);
757            let before = proportional_lines / 3;
758            let mut start = request
759                .matched_line
760                .saturating_sub(before)
761                .max(request.declaration_start);
762            let mut end = start
763                .saturating_add(proportional_lines.saturating_sub(1))
764                .min(request.declaration_end);
765            if end.saturating_sub(start).saturating_add(1) < proportional_lines {
766                start = end
767                    .saturating_add(1)
768                    .saturating_sub(proportional_lines)
769                    .max(request.declaration_start);
770            }
771            end = start
772                .saturating_add(proportional_lines.saturating_sub(1))
773                .min(request.declaration_end);
774            narrowed_indices.push(index);
775            narrowed_requests.push(StoredExcerptRequest {
776                file_id: request.file_id,
777                desired_start_line: start,
778                desired_end_line: end,
779                required_start_line: request.matched_line,
780                required_end_line: request.matched_line,
781                max_lines: 0,
782            });
783        }
784        let narrowed = self.stored_excerpts(session, &narrowed_requests)?;
785        for (index, excerpt) in narrowed_indices.into_iter().zip(narrowed) {
786            excerpts[index] = excerpt;
787        }
788        Ok(excerpts)
789    }
