95    fn outline_sync(
96        &self,
97        request: OutlineRequest,
98        cancellation: &CancellationToken,
99    ) -> Result<OutlineResponse> {
100        check_cancelled(cancellation)?;
101        if request.paths.is_empty() {
102            return Err(Error::InvalidRequest(
103                "outline requires at least one path".into(),
104            ));
105        }
106        if request.paths.len() > MAX_INPUT_ITEMS {
107            return Err(Error::LimitExceeded);
108        }
109        for path in &request.paths {
110            validate_input(path, "path", MAX_PATH_BYTES)?;
111        }
112        validate_optional_input(
113            request.symbol_name.as_deref(),
114            "symbol name",
115            MAX_PATTERN_BYTES,
116        )?;
117        validate_optional_input(
118            request.symbol_kind.as_deref(),
119            "symbol kind",
120            MAX_PATTERN_BYTES,
121        )?;
122        self.consistent(|session, generation| {
123            let limit = self.result_limit(request.max_results);
124            let token_limit = self.token_limit(request.max_tokens, self.config.default_read_tokens);
125            let mut remaining = limit;
126            let mut emitted_tokens = 0usize;
127            let mut files = Vec::new();
128            for path in &request.paths {
129                check_cancelled(cancellation)?;
130                validate_relative(path)?;
131                let file = session
132                    .find_file(path)?
133                    .ok_or_else(|| Error::NotIndexed(path.clone()))?;
134                let mut symbols = session
135                    .get_symbols_for_file_filtered(
136                        file.id,
137                        request.symbol_name.as_deref(),
138                        request.symbol_kind.as_deref(),
139                        remaining.max(1),
140                    )?
141                    .into_iter()
142                    .map(storage_symbol)
143                    .collect::<Vec<_>>();
144                symbols.retain(|symbol| {
145                    let cost = symbol
146                        .signature
147                        .as_deref()
148                        .map_or(1, |value| self.config.tokenizer.count(value));
149                    if remaining == 0 || emitted_tokens.saturating_add(cost) > token_limit {
150                        false
151                    } else {
152                        remaining -= 1;
153                        emitted_tokens += cost;
154                        true
155                    }
156                });
157                let mut imports = session
158                    .get_imports_for_file(file.id, limit)?
159                    .into_iter()
160                    .map(|import| Import {
161                        raw_target: import.raw_target,
162                        resolved_path: import.resolved_path,
163                        line: import.line,
164                    })
165                    .collect::<Vec<_>>();
166                imports.retain(|import| {
167                    let cost = self.config.tokenizer.count(&import.raw_target)
168                        + import
169                            .resolved_path
170                            .as_deref()
171                            .map_or(0, |value| self.config.tokenizer.count(value));
172                    if remaining == 0 || emitted_tokens.saturating_add(cost) > token_limit {
173                        false
174                    } else {
175                        remaining -= 1;
176                        emitted_tokens += cost;
177                        true
178                    }
179                });
180                files.push(OutlineFile {
181                    path: file.path,
182                    language: file.language.clone(),
183                    structurally_complete: file.structurally_complete,
184                    symbols,
185                    imports,
186                });
187                if remaining == 0 {
188                    break;
189                }
190            }
191            Ok(OutlineResponse {
192                files,
193                meta: self.meta(generation, emitted_tokens, None),
194            })
195        })
196    }

198    fn read_sync(
199        &self,
200        request: ReadRequest,
201        cancellation: &CancellationToken,
202    ) -> Result<ReadResponse> {
203        check_cancelled(cancellation)?;
204        validate_input(&request.path, "path", MAX_PATH_BYTES)?;
205        validate_optional_input(request.symbol.as_deref(), "symbol", MAX_PATTERN_BYTES)?;
206        validate_optional_input(request.expected_hash.as_deref(), "expected hash", 128)?;
207        validate_relative(&request.path)?;
208        if request.symbol.is_some() && (request.start_line.is_some() || request.end_line.is_some())
209        {
210            return Err(Error::InvalidRequest(
211                "read accepts either symbol or line range, not both".into(),
212            ));
213        }
214        self.consistent(|session, generation| {
215            check_cancelled(cancellation)?;
216            self.read_at_generation(session, &request, generation)
217        })
218    }

220    fn read_at_generation(
221        &self,
222        session: &ReadSession,
223        request: &ReadRequest,
224        generation: u64,
225    ) -> Result<ReadResponse> {
226        let indexed = session
227            .find_file(&request.path)?
228            .ok_or_else(|| Error::NotIndexed(request.path.clone()))?;
229        let path = resolve_existing(&self.config.root, &request.path)?;
230        let bytes = fs::read(path)?;
231        let source = std::str::from_utf8(&bytes)
232            .map_err(|_| Error::InvalidRequest("requested file is not UTF-8 text".into()))?;
233        let line_count = source.lines().count().max(1);
234
235        let (start_line, requested_end) = if let Some(symbol_name) = &request.symbol {
236            let symbol = session
237                .find_symbol(indexed.id, symbol_name)?
238                .ok_or_else(|| Error::NotIndexed(format!("{}::{symbol_name}", request.path)))?;
239            (symbol.start_line, symbol.end_line)
240        } else {
241            (
242                request.start_line.unwrap_or(1),
243                request.end_line.unwrap_or(line_count),
244            )
245        };
246        if start_line == 0 || requested_end < start_line || start_line > line_count {
247            return Err(Error::InvalidRequest(
248                "invalid or out-of-range line range".into(),
249            ));
250        }
251        let requested_end = requested_end.min(line_count);
252        let content = crate::text::excerpt(source, start_line, requested_end);
253        let max_tokens = self.token_limit(request.max_tokens, self.config.default_read_tokens);
254        let (content, emitted_tokens) = self.config.tokenizer.truncate(&content, max_tokens);
255        let returned_lines = content
256            .lines()
257            .count()
258            .max(usize::from(!content.is_empty()));
259        let end_line = if returned_lines == 0 {
260            start_line
261        } else {
262            start_line + returned_lines - 1
263        };
264        let content_hash = hash(content);
265        let full_hash = crate::text::hash_bytes(&bytes);
266        let index_stale = indexed.content_hash != full_hash;
267        let indexed_hash = Some(indexed.content_hash);
268        let not_modified = request.expected_hash.as_deref() == Some(content_hash.as_str());
269
270        Ok(ReadResponse {
271            path: request.path.clone(),
272            status: if not_modified {
273                ReadStatus::NotModified
274            } else {
275                ReadStatus::Content
276            },
277            start_line,
278            end_line,
279            content: (!not_modified).then(|| content.to_string()),
280            content_hash,
281            indexed_hash,
282            index_stale,
283            meta: self.meta(
284                generation,
285                if not_modified { 0 } else { emitted_tokens },
286                None,
287            ),
288        })
289    }

291    pub(super) fn stored_excerpt(
292        &self,
293        session: &ReadSession,
294        file_id: i64,
295        start_line: usize,
296        end_line: usize,
297        context: usize,
298        max_lines: usize,
299    ) -> Result<Option<StoredExcerpt>> {
300        let first = start_line.saturating_sub(context).max(1);
301        let mut last = end_line.saturating_add(context);
302        if max_lines > 0 && last.saturating_sub(first).saturating_add(1) > max_lines {
303            last = first + max_lines - 1;
304        }
305        let selected = session.get_chunks_overlapping(file_id, first, last)?;
306        let (Some(first_chunk), Some(last_chunk)) = (selected.first(), selected.last()) else {
307            return Ok(None);
308        };
309        last = last.min(last_chunk.end_line);
310        let base_line = first_chunk.start_line;
311        let mut combined = String::new();
312        for chunk in &selected {
313            combined.push_str(&chunk.content);
314        }
315        let local_start = first.saturating_sub(base_line) + 1;
316        let local_end = last.saturating_sub(base_line) + 1;
317        Ok(Some(StoredExcerpt {
318            content: crate::text::excerpt(&combined, local_start, local_end),
319            start_line: first,
320            end_line: last,
321        }))
322    }

324    pub(super) fn adaptive_context_excerpt(
325        &self,
326        session: &ReadSession,
327        file_id: i64,
328        declaration_start: usize,
329        declaration_end: usize,
330        matched_line: usize,
331        token_budget: usize,
332    ) -> Result<Option<StoredExcerpt>> {
333        let Some(full) =
334            self.stored_excerpt(session, file_id, declaration_start, declaration_end, 0, 0)?
335        else {
336            return Ok(None);
337        };
338        let full_tokens = self.config.tokenizer.count(&full.content).max(1);
339        if full_tokens <= token_budget {
340            return Ok(Some(full));
341        }
342
343        let declaration_lines = declaration_end
344            .saturating_sub(declaration_start)
345            .saturating_add(1);
346        let proportional_lines = declaration_lines
347            .saturating_mul(token_budget)
348            .saturating_div(full_tokens)
349            .clamp(MIN_CONTEXT_RANGE_LINES, MAX_CONTEXT_RANGE_LINES)
350            .min(declaration_lines);
351        let before = proportional_lines / 3;
352        let mut start = matched_line.saturating_sub(before).max(declaration_start);
353        let mut end = start
354            .saturating_add(proportional_lines.saturating_sub(1))
355            .min(declaration_end);
356        if end.saturating_sub(start).saturating_add(1) < proportional_lines {
357            start = end
358                .saturating_add(1)
359                .saturating_sub(proportional_lines)
360                .max(declaration_start);
361        }
362        end = start
363            .saturating_add(proportional_lines.saturating_sub(1))
364            .min(declaration_end);
365        self.stored_excerpt(session, file_id, start, end, 0, 0)
366    }
