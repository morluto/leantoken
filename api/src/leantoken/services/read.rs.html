45fn assemble_stored_excerpt(
46    request: &StoredExcerptRequest,
47    selected: &[crate::storage::ChunkRecord],
48) -> Option<StoredExcerpt> {
49    let first = request.start_line.saturating_sub(request.context).max(1);
50    let mut last = request.end_line.saturating_add(request.context);
51    if request.max_lines > 0 && last.saturating_sub(first).saturating_add(1) > request.max_lines {
52        last = first + request.max_lines - 1;
53    }
54    let (Some(first_chunk), Some(last_chunk)) = (selected.first(), selected.last()) else {
55        return None;
56    };
57    last = last.min(last_chunk.end_line);
58    let base_line = first_chunk.start_line;
59    let mut combined = String::new();
60    for chunk in selected {
61        combined.push_str(&chunk.content);
62    }
63    let local_start = first.saturating_sub(base_line) + 1;
64    let local_end = last.saturating_sub(base_line) + 1;
65    Some(StoredExcerpt {
66        content: crate::text::excerpt(&combined, local_start, local_end),
67        start_line: first,
68        end_line: last,
69    })
70}

140    fn outline_sync(
141        &self,
142        request: OutlineRequest,
143        cancellation: &CancellationToken,
144    ) -> Result<OutlineResponse> {
145        check_cancelled(cancellation)?;
146        if request.paths.is_empty() {
147            return Err(Error::InvalidInput {
148                field: "paths",
149                reason: "must contain at least one path",
150            });
151        }
152        if request.paths.len() > MAX_INPUT_ITEMS {
153            return Err(Error::LimitExceeded);
154        }
155        for path in &request.paths {
156            validate_input(path, "path", MAX_PATH_BYTES)?;
157        }
158        validate_optional_input(
159            request.symbol_name.as_deref(),
160            "symbol name",
161            MAX_PATTERN_BYTES,
162        )?;
163        validate_optional_input(
164            request.symbol_kind.as_deref(),
165            "symbol kind",
166            MAX_PATTERN_BYTES,
167        )?;
168        self.consistent(|session, generation| {
169            let limit = self.result_limit(request.max_results);
170            let token_limit = self.token_limit(request.max_tokens, self.config.default_read_tokens);
171            let mut remaining = limit;
172            let mut emitted_tokens = 0usize;
173            let mut files = Vec::new();
174            for path in &request.paths {
175                check_cancelled(cancellation)?;
176                validate_relative(path)?;
177                let file = session
178                    .find_file(path)?
179                    .ok_or_else(|| Error::NotIndexed(path.clone()))?;
180                let mut symbols = session
181                    .get_symbols_for_file_filtered(
182                        file.id,
183                        request.symbol_name.as_deref(),
184                        request.symbol_kind.as_deref(),
185                        remaining.max(1),
186                    )?
187                    .into_iter()
188                    .map(storage_symbol)
189                    .collect::<Vec<_>>();
190                symbols.retain(|symbol| {
191                    let cost = symbol
192                        .signature
193                        .as_deref()
194                        .map_or(1, |value| self.config.tokenizer.count(value));
195                    if remaining == 0 || emitted_tokens.saturating_add(cost) > token_limit {
196                        false
197                    } else {
198                        remaining -= 1;
199                        emitted_tokens += cost;
200                        true
201                    }
202                });
203                let mut imports = session
204                    .get_imports_for_file(file.id, limit)?
205                    .into_iter()
206                    .map(|import| Import {
207                        raw_target: import.raw_target,
208                        resolved_path: import.resolved_path,
209                        line: import.line,
210                    })
211                    .collect::<Vec<_>>();
212                imports.retain(|import| {
213                    let cost = self.config.tokenizer.count(&import.raw_target)
214                        + import
215                            .resolved_path
216                            .as_deref()
217                            .map_or(0, |value| self.config.tokenizer.count(value));
218                    if remaining == 0 || emitted_tokens.saturating_add(cost) > token_limit {
219                        false
220                    } else {
221                        remaining -= 1;
222                        emitted_tokens += cost;
223                        true
224                    }
225                });
226                files.push(OutlineFile {
227                    path: file.path,
228                    language: file.language.clone(),
229                    structurally_complete: file.structurally_complete,
230                    symbols,
231                    imports,
232                });
233                if remaining == 0 {
234                    break;
235                }
236            }
237            Ok(OutlineResponse {
238                files,
239                meta: self.meta(generation, emitted_tokens, None),
240            })
241        })
242    }

244    fn read_sync(
245        &self,
246        request: ReadRequest,
247        cancellation: &CancellationToken,
248    ) -> Result<ReadResponse> {
249        check_cancelled(cancellation)?;
250        validate_input(&request.path, "path", MAX_PATH_BYTES)?;
251        validate_optional_input(request.symbol.as_deref(), "symbol", MAX_PATTERN_BYTES)?;
252        validate_optional_input(request.expected_hash.as_deref(), "expected hash", 128)?;
253        validate_relative(&request.path)?;
254        if request.symbol.is_some() && (request.start_line.is_some() || request.end_line.is_some())
255        {
256            return Err(Error::InvalidInput {
257                field: "read target",
258                reason: "must use either a symbol or line range, not both",
259            });
260        }
261        self.consistent(|session, generation| {
262            check_cancelled(cancellation)?;
263            self.read_at_generation(session, &request, generation)
264        })
265    }

267    fn read_at_generation(
268        &self,
269        session: &ReadSession,
270        request: &ReadRequest,
271        generation: u64,
272    ) -> Result<ReadResponse> {
273        let indexed = session
274            .find_file(&request.path)?
275            .ok_or_else(|| Error::NotIndexed(request.path.clone()))?;
276        let path = resolve_existing(&self.config.root, &request.path)?;
277
278        // Stream the file through a BufReader for the full-file hash so the
279        // entire file does not need to be held in memory simultaneously. The
280        // content range is extracted by a bounded line-oriented reader.
281        let file = fs::File::open(&path)?;
282        let full_hash = stream_hash(&file)?;
283
284        // For the content range, read only the needed lines. When the symbol or
285        // line range is bounded, a line-oriented reader avoids loading the
286        // whole file into a String.
287        let (content, line_count) = read_range(&file, request, session, &indexed)?;
288        let line_count = line_count.max(1);
289
290        let (start_line, requested_end) = if let Some(symbol_name) = &request.symbol {
291            let symbol = session
292                .find_symbol(indexed.id, symbol_name)?
293                .ok_or_else(|| Error::NotIndexed(format!("{}::{symbol_name}", request.path)))?;
294            (symbol.start_line, symbol.end_line)
295        } else {
296            (
297                request.start_line.unwrap_or(1),
298                request.end_line.unwrap_or(line_count),
299            )
300        };
301        if start_line == 0 || requested_end < start_line || start_line > line_count {
302            return Err(Error::InvalidInput {
303                field: "line range",
304                reason: "must be ordered and within the requested file",
305            });
306        }
307        let requested_end = requested_end.min(line_count);
308        let content = crate::text::excerpt(&content, start_line, requested_end);
309        let max_tokens = self.token_limit(request.max_tokens, self.config.default_read_tokens);
310        let (content, emitted_tokens) = self.config.tokenizer.truncate(&content, max_tokens);
311        let returned_lines = content
312            .lines()
313            .count()
314            .max(usize::from(!content.is_empty()));
315        let end_line = if returned_lines == 0 {
316            start_line
317        } else {
318            start_line + returned_lines - 1
319        };
320        let content_hash = hash(content);
321        let index_stale = indexed.content_hash != full_hash;
322        let indexed_hash = Some(indexed.content_hash);
323        let not_modified = request.expected_hash.as_deref() == Some(content_hash.as_str());
324
325        Ok(ReadResponse {
326            path: request.path.clone(),
327            status: if not_modified {
328                ReadStatus::NotModified
329            } else {
330                ReadStatus::Content
331            },
332            start_line,
333            end_line,
334            content: (!not_modified).then(|| content.to_string()),
335            content_hash,
336            indexed_hash,
337            index_stale,
338            meta: self.meta(
339                generation,
340                if not_modified { 0 } else { emitted_tokens },
341                None,
342            ),
343        })
344    }

347/// Stream a file through a BufReader and compute the content hash without
348/// loading the entire file into memory.
349fn stream_hash(file: &File) -> Result<String> {
350    let mut reader = BufReader::new(file.try_clone()?);
351    let mut hasher = blake3::Hasher::new();
352    let mut buf = [0u8; 65_536];
353    loop {
354        let n = reader.read(&mut buf)?;
355        if n == 0 {
356            break;
357        }
358        hasher.update(&buf[..n]);
359    }
360    Ok(hasher.finalize().to_hex()[..crate::text::CONTENT_FINGERPRINT_HEX_LEN].to_string())
361}

363/// Read the file content needed to satisfy a read request. When a bounded line
364/// range is requested, only lines up to `end_line` are read. When no end is
365/// specified, the whole file is read as a String. Returns the content and the
366/// total line count.
367fn read_range(
368    file: &File,
369    request: &ReadRequest,
370    session: &ReadSession,
371    indexed: &crate::storage::FileRecord,
372) -> Result<(String, usize)> {
373    let mut file = file.try_clone()?;
374    // If a symbol is requested, we need the symbol's line range first.
375    if let Some(symbol_name) = &request.symbol {
376        let symbol = session
377            .find_symbol(indexed.id, symbol_name)?
378            .ok_or_else(|| Error::NotIndexed(format!("{}::{symbol_name}", request.path)))?;
379        return read_bounded_range(&mut file, symbol.start_line, symbol.end_line);
380    }
381
382    // If both start and end are specified, read only the needed range.
383    if let (Some(start), Some(end)) = (request.start_line, request.end_line) {
384        return read_bounded_range(&mut file, start, end);
385    }
386
387    // Otherwise, read the whole file (needed for line_count or default range).
388    file.seek(SeekFrom::Start(0))?;
389    let mut reader = BufReader::new(file);
390    let mut content = String::new();
391    reader.read_to_string(&mut content)?;
392    let line_count = content.lines().count();
393    Ok((content, line_count))
394}

396/// Read lines `start_line` through `end_line` (1-based, inclusive) from the
397/// file without loading the entire file into memory.
398fn read_bounded_range(
399    file: &mut File,
400    start_line: usize,
401    end_line: usize,
402) -> Result<(String, usize)> {
403    file.seek(SeekFrom::Start(0))?;
404    let reader = BufReader::new(file);
405    let mut content = String::new();
406    let mut line_count = 0usize;
407    let mut current_line = 0usize;
408    let effective_end = if end_line == 0 { usize::MAX } else { end_line };
409
410    for line_result in reader.lines() {
411        current_line += 1;
412        let line = line_result?;
413        if current_line >= start_line && current_line <= effective_end {
414            content.push_str(&line);
415            content.push('\n');
416        }
417        if current_line > effective_end {
418            break;
419        }
420    }
421    line_count = current_line.max(line_count);
422
423    // We need the total line count for validation, but we stopped early.
424    // For bounded reads, the count we have is at least the end line.
425    // The caller clamps to line_count, so returning current_line is safe.
426    Ok((content, line_count))
427}

458    pub(super) fn stored_excerpts(
459        &self,
460        session: &ReadSession,
461        requests: &[StoredExcerptRequest],
462    ) -> Result<Vec<Option<StoredExcerpt>>> {
463        let ranges = requests
464            .iter()
465            .map(|request| {
466                let first = request.start_line.saturating_sub(request.context).max(1);
467                let mut last = request.end_line.saturating_add(request.context);
468                if request.max_lines > 0
469                    && last.saturating_sub(first).saturating_add(1) > request.max_lines
470                {
471                    last = first + request.max_lines - 1;
472                }
473                (request.file_id, first, last)
474            })
475            .collect::<Vec<_>>();
476        let chunks = session.get_chunks_overlapping_batch(&ranges)?;
477        Ok(requests
478            .iter()
479            .zip(chunks)
480            .map(|(request, chunks)| assemble_stored_excerpt(request, &chunks))
481            .collect())
482    }

484    #[cfg(test)]
485    pub(super) fn adaptive_context_excerpt(
486        &self,
487        session: &ReadSession,
488        file_id: i64,
489        declaration_start: usize,
490        declaration_end: usize,
491        matched_line: usize,
492        token_budget: usize,
493    ) -> Result<Option<StoredExcerpt>> {
494        let Some(full) =
495            self.stored_excerpt(session, file_id, declaration_start, declaration_end, 0, 0)?
496        else {
497            return Ok(None);
498        };
499        let full_tokens = self.config.tokenizer.count(&full.content).max(1);
500        if full_tokens <= token_budget {
501            return Ok(Some(full));
502        }
503
504        let declaration_lines = declaration_end
505            .saturating_sub(declaration_start)
506            .saturating_add(1);
507        let proportional_lines = declaration_lines
508            .saturating_mul(token_budget)
509            .saturating_div(full_tokens)
510            .clamp(MIN_CONTEXT_RANGE_LINES, MAX_CONTEXT_RANGE_LINES)
511            .min(declaration_lines);
512        let before = proportional_lines / 3;
513        let mut start = matched_line.saturating_sub(before).max(declaration_start);
514        let mut end = start
515            .saturating_add(proportional_lines.saturating_sub(1))
516            .min(declaration_end);
517        if end.saturating_sub(start).saturating_add(1) < proportional_lines {
518            start = end
519                .saturating_add(1)
520                .saturating_sub(proportional_lines)
521                .max(declaration_start);
522        }
523        end = start
524            .saturating_add(proportional_lines.saturating_sub(1))
525            .min(declaration_end);
526        self.stored_excerpt(session, file_id, start, end, 0, 0)
527    }

529    pub(super) fn adaptive_context_excerpts(
530        &self,
531        session: &ReadSession,
532        requests: &[AdaptiveExcerptRequest],
533    ) -> Result<Vec<Option<StoredExcerpt>>> {
534        let full_requests = requests
535            .iter()
536            .map(|request| StoredExcerptRequest {
537                file_id: request.file_id,
538                start_line: request.declaration_start,
539                end_line: request.declaration_end,
540                context: 0,
541                max_lines: 0,
542            })
543            .collect::<Vec<_>>();
544        let mut excerpts = self.stored_excerpts(session, &full_requests)?;
545        let mut narrowed_indices = Vec::new();
546        let mut narrowed_requests = Vec::new();
547        for (index, (request, excerpt)) in requests.iter().zip(&excerpts).enumerate() {
548            let Some(excerpt) = excerpt else {
549                continue;
550            };
551            let full_tokens = self.config.tokenizer.count(&excerpt.content).max(1);
552            if full_tokens <= request.token_budget {
553                continue;
554            }
555            let declaration_lines = request
556                .declaration_end
557                .saturating_sub(request.declaration_start)
558                .saturating_add(1);
559            let proportional_lines = declaration_lines
560                .saturating_mul(request.token_budget)
561                .saturating_div(full_tokens)
562                .clamp(MIN_CONTEXT_RANGE_LINES, MAX_CONTEXT_RANGE_LINES)
563                .min(declaration_lines);
564            let before = proportional_lines / 3;
565            let mut start = request
566                .matched_line
567                .saturating_sub(before)
568                .max(request.declaration_start);
569            let mut end = start
570                .saturating_add(proportional_lines.saturating_sub(1))
571                .min(request.declaration_end);
572            if end.saturating_sub(start).saturating_add(1) < proportional_lines {
573                start = end
574                    .saturating_add(1)
575                    .saturating_sub(proportional_lines)
576                    .max(request.declaration_start);
577            }
578            end = start
579                .saturating_add(proportional_lines.saturating_sub(1))
580                .min(request.declaration_end);
581            narrowed_indices.push(index);
582            narrowed_requests.push(StoredExcerptRequest {
583                file_id: request.file_id,
584                start_line: start,
585                end_line: end,
586                context: 0,
587                max_lines: 0,
588            });
589        }
590        let narrowed = self.stored_excerpts(session, &narrowed_requests)?;
591        for (index, excerpt) in narrowed_indices.into_iter().zip(narrowed) {
592            excerpts[index] = excerpt;
593        }
594        Ok(excerpts)
595    }
