32fn cached_file(
33    session: &ReadSession,
34    cache: &mut HashMap<String, Option<FileRecord>>,
35    path: &str,
36) -> Result<Option<FileRecord>> {
37    if let Some(file) = cache.get(path) {
38        return Ok(file.clone());
39    }
40    let file = session.find_file(path)?;
41    cache.insert(path.to_owned(), file.clone());
42    Ok(file)
43}

45fn context_path_score(path: &str, terms: &[String], task: &str) -> f64 {
46    let path = path.to_lowercase();
47    let mut score = terms
48        .iter()
49        .filter(|term| path.contains(term.to_ascii_lowercase().as_str()))
50        .count() as f64;
51    for code_token in facets::legacy_code_tokens(task)
52        .into_iter()
53        .filter(|token| {
54            token.contains("::")
55                || token
56                    .split('.')
57                    .any(|part| part.chars().next().is_some_and(char::is_uppercase))
58        })
59    {
60        let matched_parts = expand_terms(&code_token)
61            .into_iter()
62            .map(|part| part.to_ascii_lowercase())
63            .filter(|part| part.chars().count() >= 2 && path.contains(part))
64            .collect::<HashSet<_>>()
65            .len();
66        if matched_parts >= 2 {
67            #[allow(clippy::cast_precision_loss)]
68            {
69                score += (matched_parts * matched_parts) as f64;
70            }
71        }
72    }
73    for (language, component) in [
74        ("javascript", "/js/"),
75        ("typescript", "/ts/"),
76        ("python", "/python/"),
77        ("rust", "/rust/"),
78        ("go", "/go/"),
79    ] {
80        if task_mentions_language(task, language) && format!("/{path}/").contains(component) {
81            // An explicit language name in the task is strong repository-scope
82            // evidence. Keep this above an exact-name match in another
83            // language so common names such as `Point` do not dominate.
84            score += 12.0;
85        }
86    }
87    score
88}

96fn qualified_symbol_match(
97    concept: &str,
98    name: &str,
99    parent: Option<&str>,
100    signature: Option<&str>,
101) -> f64 {
102    if !concept.contains(['.', ':']) {
103        return 0.0;
104    }
105    let parts = concept
106        .split(|character: char| !character.is_alphanumeric() && character != '_')
107        .flat_map(identifier_words)
108        .map(|part| part.to_ascii_lowercase())
109        .filter(|part| part.chars().count() >= 2)
110        .collect::<HashSet<_>>();
111    if parts.len() < 2 {
112        return 0.0;
113    }
114    let haystack = format!(
115        "{} {} {}",
116        name,
117        parent.unwrap_or_default(),
118        signature.unwrap_or_default()
119    )
120    .to_ascii_lowercase();
121    f64::from(parts.iter().all(|part| haystack.contains(part)))
122}

124fn record_query_hit(
125    fusion: &mut HashMap<String, HashMap<String, f64>>,
126    path: &str,
127    fusion_key: &str,
128    weight: f64,
129    rank: usize,
130) {
131    if weight < 0.65 {
132        return;
133    }
134    const RRF_K: f64 = 60.0;
135    #[allow(clippy::cast_precision_loss)]
136    let score = weight * RRF_K / (RRF_K + rank as f64 + 1.0);
137    fusion
138        .entry(path.to_owned())
139        .or_default()
140        .entry(fusion_key.to_owned())
141        .and_modify(|current| *current = current.max(score))
142        .or_insert(score);
143}

145fn apply_query_fusion(
146    candidates: &mut [Candidate],
147    fusion: &HashMap<String, HashMap<String, f64>>,
148) {
149    for candidate in candidates {
150        let Some(matches) = fusion.get(&candidate.path) else {
151            continue;
152        };
153        if matches.len() > 1 {
154            let total = matches.values().sum::<f64>();
155            let strongest = matches.values().copied().fold(0.0, f64::max);
156            candidate.path_score += (total - strongest).min(0.2);
157            if !candidate
158                .match_kinds
159                .iter()
160                .any(|kind| kind == "multi-query")
161            {
162                candidate.match_kinds.push("multi-query".into());
163            }
164        }
165    }
166}

168fn annotate_candidate(
169    mut candidate: Candidate,
170    query: &ContextQuery,
171    channel: &str,
172    rank: usize,
173) -> Candidate {
174    for facet in query.facet_names() {
175        candidate = candidate.facet(facet, &query.fusion_key);
176    }
177    candidate.channel(channel, rank)
178}

180fn task_mentions_language(task: &str, language: &str) -> bool {
181    task.split(|character: char| !character.is_alphanumeric())
182        .filter(|word| !word.is_empty())
183        .any(|word| {
184            if language == "go" {
185                word == "Go" || word.eq_ignore_ascii_case("golang")
186            } else {
187                word.eq_ignore_ascii_case(language)
188            }
189        })
190}

192impl Services {
193    fn file_change_boost(
194        file: Option<&FileRecord>,
195        path: &str,
196        changed_paths: &HashSet<String>,
197        prior_generation: Option<u64>,
198    ) -> f64 {
199        let mut boost = 0.0;
200
201        if let Some(prior) = prior_generation
202            && file.is_some_and(|f| f.generation > prior)
203        {
204            boost += 1.0;
205        }
206
207        if changed_paths.contains(path) {
208            boost += 1.0;
209        }
210
211        boost
212    }

232    pub async fn context_cancellable(
233        &self,
234        request: ContextRequest,
235        cancellation: CancellationToken,
236    ) -> Result<ContextResponse> {
237        let this = self.clone();
238        tokio::task::spawn_blocking(move || {
239            this.context_sync(request, &cancellation, CandidateDiagnostics::Omit)
240                .map(|evaluation| evaluation.response)
241        })
242        .await?
243    }

245    /// Retrieve context and expose pre-selection candidate paths for evaluation.
246    ///
247    /// Production adapters should use [`Self::context`]. This method exists for
248    /// frozen retrieval benchmarks and does not alter the MCP response schema.
249    pub async fn context_evaluation(&self, request: ContextRequest) -> Result<ContextEvaluation> {
250        let this = self.clone();
251        let cancellation = CancellationToken::new();
252        tokio::task::spawn_blocking(move || {
253            this.context_sync(request, &cancellation, CandidateDiagnostics::Collect)
254        })
255        .await?
256    }

258    fn context_sync(
259        &self,
260        request: ContextRequest,
261        cancellation: &CancellationToken,
262        diagnostics: CandidateDiagnostics,
263    ) -> Result<ContextEvaluation> {
264        check_cancelled(cancellation)?;
265        if request.task.trim().is_empty() || request.token_budget == 0 {
266            return Err(Error::InvalidRequest(
267                "context requires a task and positive token budget".into(),
268            ));
269        }
270        validate_input(&request.task, "task", MAX_QUERY_BYTES)?;
271        validate_patterns(&request.focus_paths)?;
272        validate_patterns(&request.exclude_paths)?;
273        if request.focus_symbols.len() > MAX_INPUT_ITEMS {
274            return Err(Error::LimitExceeded);
275        }
276        for symbol in &request.focus_symbols {
277            validate_input(symbol, "focus symbol", MAX_PATTERN_BYTES)?;
278        }
279        if request.token_budget > self.config.max_output_tokens {
280            return Err(Error::LimitExceeded);
281        }
282        if request.known_hashes.len() > MAX_INPUT_ITEMS {
283            return Err(Error::LimitExceeded);
284        }
285        for hash in &request.known_hashes {
286            validate_input(hash, "known hash", 128)?;
287        }
288        let changed_paths = git_changed_paths(&self.config.root, GIT_CHANGED_PATHS_MAX)
289            .unwrap_or_else(|error| {
290                tracing::debug!(%error, "working-tree signal unavailable");
291                HashSet::new()
292            });
293        self.consistent(|session, generation| {
294            let facet_plan = facets::plan(&request.task, MAX_CONTEXT_QUERIES);
295            let queries = facet_plan.queries;
296            let terms = queries
297                .iter()
298                .map(|query| query.value.clone())
299                .collect::<Vec<_>>();
300            let mut file_cache = HashMap::<String, Option<FileRecord>>::new();
301            let mut candidates = Vec::new();
302            let mut query_fusion = HashMap::<String, HashMap<String, f64>>::new();
303
304            // Workflow words such as `test` are useful path priors but terrible
305            // retrieval queries: nearly every test function becomes a high-
306            // scoring symbol candidate. Keep them out of candidate generation.
307            for query in queries
308                .iter()
309                .filter(|query| !query.has_facet(FacetKind::TestIntent))
310            {
311                let term = &query.value;
312                let concept = query.fusion_key.as_str();
313                check_cancelled(cancellation)?;
314                for (rank, hit) in session
315                    .search_symbols(term, false, MAX_CONTEXT_HITS_PER_SOURCE)?
316                    .into_iter()
317                    .enumerate()
318                {
319                    check_cancelled(cancellation)?;
320                    if !path_allowed(&hit.path, &[], &request.exclude_paths)? {
321                        continue;
322                    }
323                    let Some(excerpt) = self.adaptive_context_excerpt(
324                        session,
325                        hit.symbol.file_id,
326                        hit.symbol.start_line,
327                        hit.symbol.end_line,
328                        hit.symbol.start_line,
329                        request.token_budget,
330                    )?
331                    else {
332                        continue;
333                    };
334                    let exact = f64::from(hit.symbol.name.eq_ignore_ascii_case(term));
335                    let qualified = qualified_symbol_match(
336                        concept,
337                        &hit.symbol.name,
338                        hit.symbol.parent.as_deref(),
339                        hit.symbol.signature.as_deref(),
340                    );
341                    if query.fuse {
342                        record_query_hit(
343                            &mut query_fusion,
344                            &hit.path,
345                            &query.fusion_key,
346                            query.weight,
347                            rank,
348                        );
349                    }
350                    let file = cached_file(session, &mut file_cache, &hit.path)?;
351                    let change_boost = Self::file_change_boost(
352                        file.as_ref(),
353                        &hit.path,
354                        &changed_paths,
355                        request.prior_repository_generation,
356                    );
357                    let candidate = Candidate::new(
358                        &hit.path,
359                        excerpt.start_line,
360                        excerpt.end_line,
361                        excerpt.content,
362                    )
363                    .match_kind("symbol")
364                    .concept(concept, query.concept_weight)
365                    .representation("symbol")
366                    .symbol_name(hit.symbol.name)
367                    .exact(exact + qualified * 1.5)
368                    .symbol(1.0)
369                    .path_score(context_path_score(&hit.path, &terms, &request.task))
370                    .change_boost(change_boost);
371                    candidates.push(annotate_candidate(candidate, query, "symbol", rank));
372                }
373                for (rank, hit) in session
374                    .search_references(term, false, MAX_CONTEXT_HITS_PER_SOURCE)?
375                    .into_iter()
376                    .enumerate()
377                {
378                    check_cancelled(cancellation)?;
379                    if !path_allowed(&hit.path, &[], &request.exclude_paths)? {
380                        continue;
381                    }
382                    let excerpt = if let Some(symbol) = session
383                        .find_enclosing_symbol(hit.reference.file_id, hit.reference.start_line)?
384                    {
385                        self.adaptive_context_excerpt(
386                            session,
387                            hit.reference.file_id,
388                            symbol.start_line,
389                            symbol.end_line,
390                            hit.reference.start_line,
391                            request.token_budget,
392                        )?
393                    } else {
394                        None
395                    };
396                    let excerpt = if excerpt.is_some() {
397                        excerpt
398                    } else {
399                        self.stored_excerpt(
400                            session,
401                            hit.reference.file_id,
402                            hit.reference.start_line,
403                            hit.reference.end_line,
404                            2,
405                            12,
406                        )?
407                    };
408                    let Some(excerpt) = excerpt else {
409                        continue;
410                    };
411                    if query.fuse {
412                        record_query_hit(
413                            &mut query_fusion,
414                            &hit.path,
415                            &query.fusion_key,
416                            query.weight,
417                            rank,
418                        );
419                    }
420                    let file = cached_file(session, &mut file_cache, &hit.path)?;
421                    let change_boost = Self::file_change_boost(
422                        file.as_ref(),
423                        &hit.path,
424                        &changed_paths,
425                        request.prior_repository_generation,
426                    );
427                    let candidate = Candidate::new(
428                        &hit.path,
429                        excerpt.start_line,
430                        excerpt.end_line,
431                        excerpt.content,
432                    )
433                    .match_kind("reference")
434                    .concept(concept, query.concept_weight)
435                    .symbol_name(hit.reference.name)
436                    .reference(1.0)
437                    .path_score(context_path_score(&hit.path, &terms, &request.task))
438                    .change_boost(change_boost);
439                    candidates.push(annotate_candidate(candidate, query, "reference", rank));
440                }
441                let lexical = if term.chars().count() >= 3 {
442                    session.search_trigram(term, MAX_CONTEXT_LEXICAL_HITS)?
443                } else {
444                    session.search_word(&fts_quote(term), MAX_CONTEXT_LEXICAL_HITS)?
445                };
446                for (rank, hit) in lexical.into_iter().enumerate() {
447                    check_cancelled(cancellation)?;
448                    if !path_allowed(&hit.path, &[], &request.exclude_paths)? {
449                        continue;
450                    }
451                    let Some(search_hit) = chunk_search_hit(hit.clone(), term, false, 2, None)?
452                    else {
453                        continue;
454                    };
455                    let matched_line =
456                        matching_line(&hit, term, false).unwrap_or(search_hit.start_line);
457                    let excerpt = if let Some(symbol) =
458                        session.find_enclosing_symbol(hit.file_id, matched_line)?
459                    {
460                        self.adaptive_context_excerpt(
461                            session,
462                            hit.file_id,
463                            symbol.start_line,
464                            symbol.end_line,
465                            matched_line,
466                            request.token_budget,
467                        )?
468                    } else {
469                        None
470                    }
471                    .unwrap_or(StoredExcerpt {
472                        content: search_hit.excerpt.clone(),
473                        start_line: search_hit.start_line,
474                        end_line: search_hit.end_line,
475                    });
476                    if query.fuse {
477                        record_query_hit(
478                            &mut query_fusion,
479                            &search_hit.path,
480                            &query.fusion_key,
481                            query.weight,
482                            rank,
483                        );
484                    }
485                    let occurrences = hit
486                        .content
487                        .to_lowercase()
488                        .matches(&term.to_lowercase())
489                        .count();
490                    let file = cached_file(session, &mut file_cache, &search_hit.path)?;
491                    let change_boost = Self::file_change_boost(
492                        file.as_ref(),
493                        &search_hit.path,
494                        &changed_paths,
495                        request.prior_repository_generation,
496                    );
497                    let candidate = Candidate::new(
498                        &search_hit.path,
499                        excerpt.start_line,
500                        excerpt.end_line,
501                        excerpt.content,
502                    )
503                    .match_kind("text")
504                    .concept(concept, query.concept_weight)
505                    .exact(query.weight)
506                    .bm25((-hit.score).max(0.0) * 1_000_000.0)
507                    .path_score(context_path_score(&search_hit.path, &terms, &request.task))
508                    .lexical_frequency_penalty(
509                        (occurrences.saturating_sub(5) as f64 / 20.0).min(1.0),
510                    )
511                    .change_boost(change_boost);
512                    candidates.push(annotate_candidate(candidate, query, "text", rank));
513                }
514            }
515
516            apply_query_fusion(&mut candidates, &query_fusion);
517
518            let seed_paths = candidates
519                .iter()
520                .map(|candidate| candidate.path.clone())
521                .collect::<BTreeSet<_>>();
522            let mut neighbor_count = 0usize;
523            for seed_path in seed_paths.iter().take(24) {
524                check_cancelled(cancellation)?;
525                let Some(seed_file) = cached_file(session, &mut file_cache, seed_path)? else {
526                    continue;
527                };
528                for import in session.get_imports_for_file(seed_file.id, 32)? {
529                    check_cancelled(cancellation)?;
530                    let Some(target_path) = import.resolved_path else {
531                        continue;
532                    };
533                    if !path_allowed(&target_path, &[], &request.exclude_paths)? {
534                        continue;
535                    }
536                    let Some(target_file) = cached_file(session, &mut file_cache, &target_path)?
537                    else {
538                        continue;
539                    };
540                    let Some(chunk) = session
541                        .get_chunks_for_file(target_file.id, 1)?
542                        .into_iter()
543                        .next()
544                    else {
545                        continue;
546                    };
547                    let end_line = chunk.end_line.min(chunk.start_line + 29);
548                    let content = crate::text::excerpt(
549                        &chunk.content,
550                        1,
551                        end_line.saturating_sub(chunk.start_line) + 1,
552                    );
553                    let change_boost = Self::file_change_boost(
554                        Some(&target_file),
555                        &target_path,
556                        &changed_paths,
557                        request.prior_repository_generation,
558                    );
559                    candidates.push(
560                        Candidate::new(&target_path, chunk.start_line, end_line, content)
561                            .match_kind("import")
562                            .concept(seed_path, 0.2)
563                            .representation("import_neighbor")
564                            .path_score(context_path_score(&target_path, &terms, &request.task))
565                            .import_boost(1.0)
566                            .change_boost(change_boost),
567                    );
568                    neighbor_count += 1;
569                    if neighbor_count >= 24 {
570                        break;
571                    }
572                }
573                if neighbor_count >= 24 {
574                    break;
575                }
576            }
577
578            let generated_candidate_paths = if diagnostics == CandidateDiagnostics::Collect {
579                candidates
580                    .iter()
581                    .map(|candidate| candidate.path.clone())
582                    .collect::<BTreeSet<_>>()
583                    .into_iter()
584                    .collect()
585            } else {
586                Vec::new()
587            };
588            let generated_candidates = if diagnostics == CandidateDiagnostics::Collect {
589                candidates
590                    .iter()
591                    .map(|candidate| {
592                        let token_count = candidate.token_count_with(self.config.tokenizer).max(1);
593                        ContextCandidateEvaluation {
594                            path: candidate.path.clone(),
595                            start_line: candidate.start_line,
596                            end_line: candidate.end_line,
597                            representation: candidate.representation.clone(),
598                            match_kinds: candidate.match_kinds.clone(),
599                            concepts: candidate.concepts.clone(),
600                            concept_weight: candidate.concept_weight,
601                            score: candidate.score(&ranking::Weights::default(), token_count),
602                            token_count,
603                        }
604                    })
605                    .collect()
606            } else {
607                Vec::new()
608            };
609            let mut response = ranking::select_with_tokenizer(
610                candidates,
611                &request,
612                generation,
613                self.config.tokenizer,
614            );
615            response.meta.freshness = self.freshness();
616            if response.fragments.is_empty() {
617                response
618                    .warnings
619                    .push("no relevant indexed evidence found".into());
620            }
621            Ok(ContextEvaluation {
622                response,
623                generated_candidate_paths,
624                generated_candidates,
625            })
626        })
627    }
