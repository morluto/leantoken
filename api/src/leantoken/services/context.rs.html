48    const fn token_cap(self) -> usize {
49        match self {
50            Self::Symbol => SYMBOL_CONTEXT_TOKEN_CAP,
51            Self::Reference => REFERENCE_CONTEXT_TOKEN_CAP,
52            Self::Text => TEXT_CONTEXT_TOKEN_CAP,

62fn cached_file(
63    session: &ReadSession,
64    cache: &mut HashMap<String, Option<FileRecord>>,
65    path: &str,
66) -> Result<Option<FileRecord>> {
67    if let Some(file) = cache.get(path) {
68        return Ok(file.clone());
69    }
70    let file = session.find_file(path)?;
71    cache.insert(path.to_owned(), file.clone());
72    Ok(file)
73}

75fn context_path_score(path: &str, terms: &[String], task: &str) -> f64 {
76    let path = path.to_lowercase();
77    let mut score = terms
78        .iter()
79        .filter(|term| path.contains(term.to_ascii_lowercase().as_str()))
80        .count() as f64;
81    for code_token in facets::legacy_code_tokens(task)
82        .into_iter()
83        .filter(|token| {
84            token.contains("::")
85                || token
86                    .split('.')
87                    .any(|part| part.chars().next().is_some_and(char::is_uppercase))
88        })
89    {
90        let matched_parts = expand_terms(&code_token)
91            .into_iter()
92            .map(|part| part.to_ascii_lowercase())
93            .filter(|part| part.chars().count() >= 2 && path.contains(part))
94            .collect::<HashSet<_>>()
95            .len();
96        if matched_parts >= 2 {
97            #[allow(clippy::cast_precision_loss)]
98            {
99                score += (matched_parts * matched_parts) as f64;
100            }
101        }
102    }
103    for (language, component) in [
104        ("javascript", "/js/"),
105        ("typescript", "/ts/"),
106        ("python", "/python/"),
107        ("rust", "/rust/"),
108        ("go", "/go/"),
109    ] {
110        if task_mentions_language(task, language) && format!("/{path}/").contains(component) {
111            // An explicit language name in the task is strong repository-scope
112            // evidence. Keep this above an exact-name match in another
113            // language so common names such as `Point` do not dominate.
114            score += 12.0;
115        }
116    }
117    score
118}

126fn qualified_symbol_match(
127    concept: &str,
128    name: &str,
129    parent: Option<&str>,
130    signature: Option<&str>,
131) -> f64 {
132    if !concept.contains(['.', ':']) {
133        return 0.0;
134    }
135    let parts = concept
136        .split(|character: char| !character.is_alphanumeric() && character != '_')
137        .flat_map(identifier_words)
138        .map(|part| part.to_ascii_lowercase())
139        .filter(|part| part.chars().count() >= 2)
140        .collect::<HashSet<_>>();
141    if parts.len() < 2 {
142        return 0.0;
143    }
144    let haystack = format!(
145        "{} {} {}",
146        name,
147        parent.unwrap_or_default(),
148        signature.unwrap_or_default()
149    )
150    .to_ascii_lowercase();
151    f64::from(parts.iter().all(|part| haystack.contains(part)))
152}

154fn record_query_hit(
155    fusion: &mut HashMap<String, HashMap<String, f64>>,
156    path: &str,
157    fusion_key: &str,
158    weight: f64,
159    rank: usize,
160) {
161    if weight < MIN_CORROBORATED_QUERY_WEIGHT {
162        return;
163    }
164    const RRF_K: f64 = 60.0;
165    #[allow(clippy::cast_precision_loss)]
166    let score = weight * RRF_K / (RRF_K + rank as f64 + 1.0);
167    fusion
168        .entry(path.to_owned())
169        .or_default()
170        .entry(fusion_key.to_owned())
171        .and_modify(|current| *current = current.max(score))
172        .or_insert(score);
173}

175fn apply_query_fusion(
176    candidates: &mut [Candidate],
177    fusion: &HashMap<String, HashMap<String, f64>>,
178) {
179    for candidate in candidates {
180        let Some(matches) = fusion.get(&candidate.path) else {
181            continue;
182        };
183        if matches.len() > 1 {
184            let total = matches.values().sum::<f64>();
185            let strongest = matches.values().copied().fold(0.0, f64::max);
186            candidate.path_score += (total - strongest).min(0.2);
187            if !candidate
188                .match_kinds
189                .iter()
190                .any(|kind| kind == "multi-query")
191            {
192                candidate.match_kinds.push("multi-query".into());
193            }
194        }
195    }
196}

210fn low_cardinality_exact_query(queries: &[ContextQuery]) -> bool {
211    queries
212        .iter()
213        .filter(|query| query.has_facet(FacetKind::ExactAtom))
214        .map(|query| query.fusion_key.as_str())
215        .collect::<BTreeSet<_>>()
216        .len()
217        == 1
218}

220fn corroborated_import_symbol<'a>(
221    symbols: Vec<SymbolRecord>,
222    queries: &'a [ContextQuery],
223    seed_concepts: &BTreeSet<String>,
224) -> Option<(SymbolRecord, &'a ContextQuery, f64)> {
225    let mut best: Option<(usize, usize, usize, SymbolRecord, &ContextQuery, f64)> = None;
226    for (query_rank, query) in queries.iter().enumerate() {
227        if query.concept_weight < MIN_CORROBORATED_QUERY_WEIGHT
228            || !seed_concepts.contains(&query.fusion_key)
229            || !(query.has_facet(FacetKind::ExactAtom)
230                || query.has_facet(FacetKind::Symbol)
231                || query.has_facet(FacetKind::Configuration))
232        {
233            continue;
234        }
235        for symbol in &symbols {
236            let exact = symbol.name.eq_ignore_ascii_case(&query.value);
237            let qualified = qualified_symbol_match(
238                &query.fusion_key,
239                &symbol.name,
240                symbol.parent.as_deref(),
241                symbol.signature.as_deref(),
242            ) > 0.0;
243            if !exact && !qualified {
244                continue;
245            }
246            let class = usize::from(qualified) * 2 + usize::from(exact);
247            let evidence = f64::from(exact) + f64::from(qualified) * 1.5;
248            let candidate = (
249                class,
250                usize::MAX - query_rank,
251                usize::MAX - symbol.start_line,
252                symbol.clone(),
253                query,
254                evidence,
255            );
256            if best.as_ref().is_none_or(|current| {
257                (candidate.0, candidate.1, candidate.2) > (current.0, current.1, current.2)
258            }) {
259                best = Some(candidate);
260            }
261        }
262    }
263    best.map(|(_, _, _, symbol, query, evidence)| (symbol, query, evidence))
264}

266fn import_seed_paths(
267    candidates: &[Candidate],
268    queries: &[ContextQuery],
269    tokenizer: crate::tokens::Tokenizer,
270) -> Vec<(String, BTreeSet<String>)> {
271    if low_cardinality_exact_query(queries) {
272        return Vec::new();
273    }
274    let mut paths = BTreeMap::<String, (f64, BTreeSet<String>)>::new();
275    for candidate in candidates {
276        if candidate.concept_weight < MIN_CORROBORATED_QUERY_WEIGHT || candidate.concepts.is_empty()
277        {
278            continue;
279        }
280        let token_count = candidate.token_count_with(tokenizer).max(1);
281        let score = candidate.score(&ranking::Weights::default(), token_count);
282        let entry = paths
283            .entry(candidate.path.clone())
284            .or_insert_with(|| (score, BTreeSet::new()));
285        entry.0 = entry.0.max(score);
286        entry.1.extend(candidate.concepts.iter().cloned());
287    }
288    let mut paths = paths.into_iter().collect::<Vec<_>>();
289    paths.sort_by(|left, right| {
290        right
291            .1
292            .0
293            .total_cmp(&left.1.0)
294            .then_with(|| left.0.cmp(&right.0))
295    });
296    paths
297        .into_iter()
298        .map(|(path, (_, concepts))| (path, concepts))
299        .collect()
300}

311fn task_mentions_language(task: &str, language: &str) -> bool {
312    task.split(|character: char| !character.is_alphanumeric())
313        .filter(|word| !word.is_empty())
314        .any(|word| {
315            if language == "go" {
316                word == "Go" || word.eq_ignore_ascii_case("golang")
317            } else {
318                word.eq_ignore_ascii_case(language)
319            }
320        })
321}

324    fn file_change_boost(
325        file: Option<&FileRecord>,
326        path: &str,
327        changed_paths: &HashSet<String>,
328        prior_generation: Option<u64>,
329    ) -> f64 {
330        let mut boost = 0.0;
331
332        if let Some(prior) = prior_generation
333            && file.is_some_and(|f| f.generation > prior)
334        {
335            boost += 1.0;
336        }
337
338        if changed_paths.contains(path) {
339            boost += 1.0;
340        }
341
342        boost
343    }

345    fn append_import_symbol_candidates(
346        &self,
347        expansion: ImportExpansion<'_>,
348        file_cache: &mut HashMap<String, Option<FileRecord>>,
349        candidates: &mut Vec<Candidate>,
350    ) -> Result<()> {
351        let seed_paths = import_seed_paths(candidates, expansion.queries, self.config.tokenizer);
352        let mut neighbor_count = 0usize;
353        let mut neighbor_ranges = BTreeSet::new();
354        for (seed_path, seed_concepts) in seed_paths.iter().take(24) {
355            check_cancelled(expansion.cancellation)?;
356            let Some(seed_file) = cached_file(expansion.session, file_cache, seed_path)? else {
357                continue;
358            };
359            for import in expansion.session.get_imports_for_file(seed_file.id, 32)? {
360                check_cancelled(expansion.cancellation)?;
361                let Some(target_path) = import.resolved_path else {
362                    continue;
363                };
364                if !path_allowed(&target_path, &[], &expansion.request.exclude_paths)? {
365                    continue;
366                }
367                let Some(target_file) = cached_file(expansion.session, file_cache, &target_path)?
368                else {
369                    continue;
370                };
371                let Some((symbol, query, exact)) = corroborated_import_symbol(
372                    expansion
373                        .session
374                        .get_symbols_for_file(target_file.id, MAX_IMPORT_SYMBOLS)?,
375                    expansion.queries,
376                    seed_concepts,
377                ) else {
378                    continue;
379                };
380                let Some(excerpt) = self.adaptive_context_excerpt(
381                    expansion.session,
382                    target_file.id,
383                    symbol.start_line,
384                    symbol.end_line,
385                    symbol.start_line,
386                    excerpt_budget(
387                        expansion.request.token_budget,
388                        ContextExcerptKind::ImportSymbol,
389                    ),
390                )?
391                else {
392                    continue;
393                };
394                if !neighbor_ranges.insert((
395                    target_path.clone(),
396                    excerpt.start_line,
397                    excerpt.end_line,
398                )) {
399                    continue;
400                }
401                let change_boost = Self::file_change_boost(
402                    Some(&target_file),
403                    &target_path,
404                    expansion.changed_paths,
405                    expansion.request.prior_repository_generation,
406                );
407                let candidate = Candidate::new(
408                    &target_path,
409                    excerpt.start_line,
410                    excerpt.end_line,
411                    excerpt.content,
412                )
413                .match_kind("import")
414                .match_kind("symbol")
415                .concept(&query.fusion_key, query.concept_weight)
416                .representation("import_symbol")
417                .symbol_name(symbol.name)
418                .exact(exact)
419                .symbol(1.0)
420                .path_score(context_path_score(
421                    &target_path,
422                    expansion.terms,
423                    &expansion.request.task,
424                ))
425                .import_boost(1.0)
426                .change_boost(change_boost);
427                candidates.push(annotate_candidate(
428                    candidate,
429                    query,
430                    "import_symbol",
431                    neighbor_count,
432                ));
433                neighbor_count += 1;
434                if neighbor_count >= 24 {
435                    break;
436                }
437            }
438            if neighbor_count >= 24 {
439                break;
440            }
441        }
442        Ok(())
443    }

463    pub async fn context_cancellable(
464        &self,
465        request: ContextRequest,
466        cancellation: CancellationToken,
467    ) -> Result<ContextResponse> {
468        let this = self.clone();
469        tokio::task::spawn_blocking(move || {

489    fn context_sync(
490        &self,
491        request: ContextRequest,
492        cancellation: &CancellationToken,
493        diagnostics: CandidateDiagnostics,
494    ) -> Result<ContextEvaluation> {
495        check_cancelled(cancellation)?;
496        if request.task.trim().is_empty() || request.token_budget == 0 {
497            return Err(Error::InvalidRequest(
498                "context requires a task and positive token budget".into(),
499            ));
500        }
501        validate_input(&request.task, "task", MAX_QUERY_BYTES)?;
502        validate_patterns(&request.focus_paths)?;
503        validate_patterns(&request.exclude_paths)?;
504        if request.focus_symbols.len() > MAX_INPUT_ITEMS {
505            return Err(Error::LimitExceeded);
506        }
507        for symbol in &request.focus_symbols {
508            validate_input(symbol, "focus symbol", MAX_PATTERN_BYTES)?;
509        }
510        if request.token_budget > self.config.max_output_tokens {
511            return Err(Error::LimitExceeded);
512        }
513        if request.known_hashes.len() > MAX_INPUT_ITEMS {
514            return Err(Error::LimitExceeded);
515        }
516        for hash in &request.known_hashes {
517            validate_input(hash, "known hash", 128)?;
518        }
519        let changed_paths = git_changed_paths(&self.config.root, GIT_CHANGED_PATHS_MAX)
520            .unwrap_or_else(|error| {
521                tracing::debug!(%error, "working-tree signal unavailable");
522                HashSet::new()
523            });
524        self.consistent(|session, generation| {
525            let facet_plan = facets::plan(&request.task, MAX_CONTEXT_QUERIES);
526            let queries = facet_plan.queries;
527            let terms = queries
528                .iter()
529                .map(|query| query.value.clone())
530                .collect::<Vec<_>>();
531            let mut file_cache = HashMap::<String, Option<FileRecord>>::new();
532            let mut candidates = Vec::new();
533            let mut query_fusion = HashMap::<String, HashMap<String, f64>>::new();
534
535            // Workflow words such as `test` are useful path priors but terrible
536            // retrieval queries: nearly every test function becomes a high-
537            // scoring symbol candidate. Keep them out of candidate generation.
538            for query in queries
539                .iter()
540                .filter(|query| !query.has_facet(FacetKind::TestIntent))
541            {
542                let term = &query.value;
543                let concept = query.fusion_key.as_str();
544                check_cancelled(cancellation)?;
545                for (rank, hit) in session
546                    .search_symbols(term, false, MAX_CONTEXT_HITS_PER_SOURCE)?
547                    .into_iter()
548                    .enumerate()
549                {
550                    check_cancelled(cancellation)?;
551                    if !path_allowed(&hit.path, &[], &request.exclude_paths)? {
552                        continue;
553                    }
554                    let Some(excerpt) = self.adaptive_context_excerpt(
555                        session,
556                        hit.symbol.file_id,
557                        hit.symbol.start_line,
558                        hit.symbol.end_line,
559                        hit.symbol.start_line,
560                        excerpt_budget(request.token_budget, ContextExcerptKind::Symbol),
561                    )?
562                    else {
563                        continue;
564                    };
565                    let exact = f64::from(hit.symbol.name.eq_ignore_ascii_case(term));
566                    let qualified = qualified_symbol_match(
567                        concept,
568                        &hit.symbol.name,
569                        hit.symbol.parent.as_deref(),
570                        hit.symbol.signature.as_deref(),
571                    );
572                    if query.fuse {
573                        record_query_hit(
574                            &mut query_fusion,
575                            &hit.path,
576                            &query.fusion_key,
577                            query.weight,
578                            rank,
579                        );
580                    }
581                    let file = cached_file(session, &mut file_cache, &hit.path)?;
582                    let change_boost = Self::file_change_boost(
583                        file.as_ref(),
584                        &hit.path,
585                        &changed_paths,
586                        request.prior_repository_generation,
587                    );
588                    let candidate = Candidate::new(
589                        &hit.path,
590                        excerpt.start_line,
591                        excerpt.end_line,
592                        excerpt.content,
593                    )
594                    .match_kind("symbol")
595                    .concept(concept, query.concept_weight)
596                    .representation("symbol")
597                    .symbol_name(hit.symbol.name)
598                    .exact(exact + qualified * 1.5)
599                    .symbol(1.0)
600                    .path_score(context_path_score(&hit.path, &terms, &request.task))
601                    .change_boost(change_boost);
602                    candidates.push(annotate_candidate(candidate, query, "symbol", rank));
603                }
604                for (rank, hit) in session
605                    .search_references(term, false, MAX_CONTEXT_HITS_PER_SOURCE)?
606                    .into_iter()
607                    .enumerate()
608                {
609                    check_cancelled(cancellation)?;
610                    if !path_allowed(&hit.path, &[], &request.exclude_paths)? {
611                        continue;
612                    }
613                    let excerpt = if let Some(symbol) = session
614                        .find_enclosing_symbol(hit.reference.file_id, hit.reference.start_line)?
615                    {
616                        self.adaptive_context_excerpt(
617                            session,
618                            hit.reference.file_id,
619                            symbol.start_line,
620                            symbol.end_line,
621                            hit.reference.start_line,
622                            excerpt_budget(request.token_budget, ContextExcerptKind::Reference),
623                        )?
624                    } else {
625                        None
626                    };
627                    let excerpt = if excerpt.is_some() {
628                        excerpt
629                    } else {
630                        self.stored_excerpt(
631                            session,
632                            hit.reference.file_id,
633                            hit.reference.start_line,
634                            hit.reference.end_line,
635                            2,
636                            12,
637                        )?
638                    };
639                    let Some(excerpt) = excerpt else {
640                        continue;
641                    };
642                    if query.fuse {
643                        record_query_hit(
644                            &mut query_fusion,
645                            &hit.path,
646                            &query.fusion_key,
647                            query.weight,
648                            rank,
649                        );
650                    }
651                    let file = cached_file(session, &mut file_cache, &hit.path)?;
652                    let change_boost = Self::file_change_boost(
653                        file.as_ref(),
654                        &hit.path,
655                        &changed_paths,
656                        request.prior_repository_generation,
657                    );
658                    let candidate = Candidate::new(
659                        &hit.path,
660                        excerpt.start_line,
661                        excerpt.end_line,
662                        excerpt.content,
663                    )
664                    .match_kind("reference")
665                    .concept(concept, query.concept_weight)
666                    .symbol_name(hit.reference.name)
667                    .reference(1.0)
668                    .path_score(context_path_score(&hit.path, &terms, &request.task))
669                    .change_boost(change_boost);
670                    candidates.push(annotate_candidate(candidate, query, "reference", rank));
671                }
672                let lexical = if term.chars().count() >= 3 {
673                    session.search_trigram(term, MAX_CONTEXT_LEXICAL_HITS)?
674                } else {
675                    session.search_word(&fts_quote(term), MAX_CONTEXT_LEXICAL_HITS)?
676                };
677                for (rank, hit) in lexical.into_iter().enumerate() {
678                    check_cancelled(cancellation)?;
679                    if !path_allowed(&hit.path, &[], &request.exclude_paths)? {
680                        continue;
681                    }
682                    let Some(search_hit) = chunk_search_hit(hit.clone(), term, false, 2, None)?
683                    else {
684                        continue;
685                    };
686                    let matched_line =
687                        matching_line(&hit, term, false).unwrap_or(search_hit.start_line);
688                    let excerpt = if let Some(symbol) =
689                        session.find_enclosing_symbol(hit.file_id, matched_line)?
690                    {
691                        self.adaptive_context_excerpt(
692                            session,
693                            hit.file_id,
694                            symbol.start_line,
695                            symbol.end_line,
696                            matched_line,
697                            excerpt_budget(request.token_budget, ContextExcerptKind::Text),
698                        )?
699                    } else {
700                        None
701                    }
702                    .unwrap_or(StoredExcerpt {
703                        content: search_hit.excerpt.clone(),
704                        start_line: search_hit.start_line,
705                        end_line: search_hit.end_line,
706                    });
707                    if query.fuse {
708                        record_query_hit(
709                            &mut query_fusion,
710                            &search_hit.path,
711                            &query.fusion_key,
712                            query.weight,
713                            rank,
714                        );
715                    }
716                    let occurrences = hit
717                        .content
718                        .to_lowercase()
719                        .matches(&term.to_lowercase())
720                        .count();
721                    let file = cached_file(session, &mut file_cache, &search_hit.path)?;
722                    let change_boost = Self::file_change_boost(
723                        file.as_ref(),
724                        &search_hit.path,
725                        &changed_paths,
726                        request.prior_repository_generation,
727                    );
728                    let candidate = Candidate::new(
729                        &search_hit.path,
730                        excerpt.start_line,
731                        excerpt.end_line,
732                        excerpt.content,
733                    )
734                    .match_kind("text")
735                    .concept(concept, query.concept_weight)
736                    .exact(query.weight)
737                    .bm25((-hit.score).max(0.0) * 1_000_000.0)
738                    .path_score(context_path_score(&search_hit.path, &terms, &request.task))
739                    .lexical_frequency_penalty(
740                        (occurrences.saturating_sub(5) as f64 / 20.0).min(1.0),
741                    )
742                    .change_boost(change_boost);
743                    candidates.push(annotate_candidate(candidate, query, "text", rank));
744                }
745            }
746
747            apply_query_fusion(&mut candidates, &query_fusion);
748
749            self.append_import_symbol_candidates(
750                ImportExpansion {
751                    session,
752                    request: &request,
753                    queries: &queries,
754                    terms: &terms,
755                    changed_paths: &changed_paths,
756                    cancellation,
757                },
758                &mut file_cache,
759                &mut candidates,
760            )?;
761
762            let generated_candidate_paths = if diagnostics == CandidateDiagnostics::Collect {
763                candidates
764                    .iter()
765                    .map(|candidate| candidate.path.clone())
766                    .collect::<BTreeSet<_>>()
767                    .into_iter()
768                    .collect()
769            } else {
770                Vec::new()
771            };
772            let generated_candidates = if diagnostics == CandidateDiagnostics::Collect {
773                candidates
774                    .iter()
775                    .map(|candidate| {
776                        let token_count = candidate.token_count_with(self.config.tokenizer).max(1);
777                        ContextCandidateEvaluation {
778                            path: candidate.path.clone(),
779                            start_line: candidate.start_line,
780                            end_line: candidate.end_line,
781                            representation: candidate.representation.clone(),
782                            match_kinds: candidate.match_kinds.clone(),
783                            concepts: candidate.concepts.clone(),
784                            concept_weight: candidate.concept_weight,
785                            score: candidate.score(&ranking::Weights::default(), token_count),
786                            token_count,
787                        }
788                    })
789                    .collect()
790            } else {
791                Vec::new()
792            };
793            let mut response = ranking::select_with_tokenizer(
794                candidates,
795                &request,
796                generation,
797                self.config.tokenizer,
798            );
799            response.meta.freshness = self.freshness();
800            if response.fragments.is_empty() {
801                response
802                    .warnings
803                    .push("no relevant indexed evidence found".into());
804            }
805            Ok(ContextEvaluation {
806                response,
807                generated_candidate_paths,
808                generated_candidates,
809            })
810        })
811    }
