77fn context_path_score(path: &str, terms: &[String], task: &str) -> f64 {
78    let path = path.to_lowercase();
79    let mut score = terms
80        .iter()
81        .filter(|term| path.contains(term.to_ascii_lowercase().as_str()))
82        .count() as f64;
83    for code_token in facets::legacy_code_tokens(task)
84        .into_iter()
85        .filter(|token| {
86            token.contains("::")
87                || token
88                    .split('.')
89                    .any(|part| part.chars().next().is_some_and(char::is_uppercase))
90        })
91    {
92        let matched_parts = expand_terms(&code_token)
93            .into_iter()
94            .map(|part| part.to_ascii_lowercase())
95            .filter(|part| part.chars().count() >= 2 && path.contains(part))
96            .collect::<HashSet<_>>()
97            .len();
98        if matched_parts >= 2 {
99            #[allow(clippy::cast_precision_loss)]
100            {
101                score += (matched_parts * matched_parts) as f64;
102            }
103        }
104    }
105    for (language, component, extensions) in [
106        ("javascript", "/js/", &["js", "jsx", "mjs", "cjs"][..]),
107        ("typescript", "/ts/", &["ts", "tsx"][..]),
108        ("python", "/python/", &["py", "pyw"][..]),
109        ("rust", "/rust/", &["rs"][..]),
110        ("go", "/go/", &["go"][..]),
111    ] {
112        let extension_matches = path
113            .rsplit_once('.')
114            .is_some_and(|(_, extension)| extensions.contains(&extension));
115        if task_mentions_language(task, language)
116            && (extension_matches || format!("/{path}/").contains(component))
117        {
118            // An explicit language name in the task is strong repository-scope
119            // evidence. Keep this above an exact-name match in another
120            // language so common names such as `Point` do not dominate.
121            score += 12.0;
122        }
123    }
124    score
125}

127fn resolve_context_workflow(requested: ContextWorkflow, task: &str) -> ContextWorkflow {
128    if requested != ContextWorkflow::Auto {
129        return requested;
130    }
131    let task = task.to_ascii_lowercase();
132    if ["pull request", "contribution", "contributing"]
133        .iter()
134        .any(|signal| task.contains(signal))
135    {
136        ContextWorkflow::Contribution
137    } else if ["code review", "review this", "review the changes"]
138        .iter()
139        .any(|signal| task.contains(signal))
140    {
141        ContextWorkflow::Review
142    } else if ["investigate", "root cause", "diagnose"]
143        .iter()
144        .any(|signal| task.contains(signal))
145    {
146        ContextWorkflow::Investigation
147    } else {
148        ContextWorkflow::Implementation
149    }

152fn workflow_path_role(path: &str, request: &ContextRequest) -> Option<(f64, &'static str)> {
153    let normalized = path.replace('\\', "/");
154    let lower = normalized.to_ascii_lowercase();
155    let mut best = WORKFLOW_PATHS
156        .matches(&normalized)
157        .into_iter()
158        .map(|index| WORKFLOW_PATH_ROLES[index])
159        .max_by(|left, right| left.0.total_cmp(&right.0));
160    if likely_owner_test(&lower, request) && best.is_none_or(|(score, _)| score < OWNER_TEST_SCORE)
161    {
162        best = Some((OWNER_TEST_SCORE, "owner_test"));
163    }
164    best
165}

187const fn workflow_path_roles() -> [(f64, &'static str); WORKFLOW_PATH_RULES.len()] {
188    let mut roles = [(0.0, ""); WORKFLOW_PATH_RULES.len()];
189    let mut index = 0;
190    while index < WORKFLOW_PATH_RULES.len() {
191        roles[index] = (WORKFLOW_PATH_RULES[index].1, WORKFLOW_PATH_RULES[index].2);
192        index += 1;
193    }
194    roles
195}

204                .build()

215fn owner_test_changed_path(path: &str, request: &ContextRequest) -> Option<String> {
216    let path = path.replace('\\', "/").to_ascii_lowercase();
217    let is_test = path.contains("/test")
218        || path.starts_with("test")
219        || path.contains("/spec")
220        || path.starts_with("spec");
221    if !is_test {
222        return None;
223    }
224    let test_name = path.rsplit('/').next().unwrap_or(&path);
225    let test_stem = test_name.split('.').next().unwrap_or(test_name);
226    request
227        .changed_paths
228        .iter()
229        .chain(&request.focus_paths)
230        .find_map(|changed| {
231            let normalized = changed.replace('\\', "/");
232            let stem = normalized
233                .rsplit('/')
234                .next()
235                .and_then(|name| name.split('.').next())?
236                .to_ascii_lowercase();
237            (stem.len() >= 3 && contains_filename_token(test_stem, &stem)).then(|| changed.clone())
238        })
239}

241fn contains_filename_token(path: &str, token: &str) -> bool {
242    path.match_indices(token).any(|(start, matched)| {
243        let end = start + matched.len();
244        let boundary =
245            |character: Option<char>| character.is_none_or(|value| !value.is_alphanumeric());
246        boundary(path[..start].chars().next_back()) && boundary(path[end..].chars().next())
247    })
248}

296fn qualified_symbol_match(
297    concept: &str,
298    name: &str,
299    parent: Option<&str>,
300    signature: Option<&str>,
301) -> f64 {
302    if !concept.contains(['.', ':']) {
303        return 0.0;
304    }
305    let parts = concept
306        .split(|character: char| !character.is_alphanumeric() && character != '_')
307        .flat_map(identifier_words)
308        .map(|part| part.to_ascii_lowercase())
309        .filter(|part| part.chars().count() >= 2)
310        .collect::<HashSet<_>>();
311    if parts.len() < 2 {
312        return 0.0;
313    }
314    let haystack = format!(
315        "{} {} {}",
316        name,
317        parent.unwrap_or_default(),
318        signature.unwrap_or_default()
319    )
320    .to_ascii_lowercase();
321    f64::from(parts.iter().all(|part| haystack.contains(part)))
322}

324fn record_query_hit(
325    fusion: &mut HashMap<String, HashMap<String, f64>>,
326    path: &str,
327    fusion_key: &str,
328    weight: f64,
329    rank: usize,
330) {
331    if weight < MIN_CORROBORATED_QUERY_WEIGHT {
332        return;
333    }
334    const RRF_K: f64 = 60.0;
335    #[allow(clippy::cast_precision_loss)]
336    let score = weight * RRF_K / (RRF_K + rank as f64 + 1.0);
337    fusion
338        .entry(path.to_owned())
339        .or_default()
340        .entry(fusion_key.to_owned())
341        .and_modify(|current| *current = current.max(score))
342        .or_insert(score);
343}

345fn apply_query_fusion(
346    candidates: &mut [Candidate],
347    fusion: &HashMap<String, HashMap<String, f64>>,
348) {
349    for candidate in candidates {
350        let Some(matches) = fusion.get(&candidate.path) else {
351            continue;
352        };
353        if matches.len() > 1 {
354            let total = matches.values().sum::<f64>();
355            let strongest = matches.values().copied().fold(0.0, f64::max);
356            candidate.path_score += (total - strongest).min(0.2);
357            if !candidate
358                .match_kinds
359                .iter()
360                .any(|kind| kind == "multi-query")
361            {
362                candidate.match_kinds.push("multi-query".into());
363            }
364        }
365    }
366}

390fn corroborated_import_symbol<'a>(
391    symbols: Vec<SymbolRecord>,
392    queries: &'a [ContextQuery],
393    seed_concepts: &BTreeSet<String>,
394) -> Option<(SymbolRecord, &'a ContextQuery, f64)> {
395    let mut best: Option<(usize, usize, usize, SymbolRecord, &ContextQuery, f64)> = None;
396    for (query_rank, query) in queries.iter().enumerate() {
397        if query.concept_weight < MIN_CORROBORATED_QUERY_WEIGHT
398            || !seed_concepts.contains(&query.fusion_key)
399            || !(query.has_facet(FacetKind::ExactAtom)
400                || query.has_facet(FacetKind::Symbol)
401                || query.has_facet(FacetKind::Configuration))
402        {
403            continue;
404        }
405        for symbol in &symbols {
406            let exact = symbol.name.eq_ignore_ascii_case(&query.value);
407            let qualified = qualified_symbol_match(
408                &query.fusion_key,
409                &symbol.name,
410                symbol.parent.as_deref(),
411                symbol.signature.as_deref(),
412            ) > 0.0;
413            if !exact && !qualified {
414                continue;
415            }
416            let class = usize::from(qualified) * 2 + usize::from(exact);
417            let evidence = f64::from(exact) + f64::from(qualified) * 1.5;
418            let candidate = (
419                class,
420                usize::MAX - query_rank,
421                usize::MAX - symbol.start_line,
422                symbol.clone(),
423                query,
424                evidence,
425            );
426            if best.as_ref().is_none_or(|current| {
427                (candidate.0, candidate.1, candidate.2) > (current.0, current.1, current.2)
428            }) {
429                best = Some(candidate);
430            }
431        }
432    }
433    best.map(|(_, _, _, symbol, query, evidence)| (symbol, query, evidence))
434}

436fn import_seed_paths(
437    candidates: &[Candidate],
438    queries: &[ContextQuery],
439    tokenizer: crate::tokens::Tokenizer,
440) -> Vec<(String, BTreeSet<String>)> {
441    if low_cardinality_exact_query(queries) {
442        return Vec::new();
443    }
444    let mut paths = BTreeMap::<String, (f64, BTreeSet<String>)>::new();
445    for candidate in candidates {
446        if candidate.concept_weight < MIN_CORROBORATED_QUERY_WEIGHT || candidate.concepts.is_empty()
447        {
448            continue;
449        }
450        let token_count = candidate.token_count_with(tokenizer).max(1);
451        let score = candidate.score(&ranking::Weights::default(), token_count);
452        let entry = paths
453            .entry(candidate.path.clone())
454            .or_insert_with(|| (score, BTreeSet::new()));
455        entry.0 = entry.0.max(score);
456        entry.1.extend(candidate.concepts.iter().cloned());
457    }
458    let mut paths = paths.into_iter().collect::<Vec<_>>();
459    paths.sort_by(|left, right| {
460        right
461            .1
462            .0
463            .total_cmp(&left.1.0)
464            .then_with(|| left.0.cmp(&right.0))
465    });
466    paths
467        .into_iter()
468        .map(|(path, (_, concepts))| (path, concepts))
469        .collect()
470}

481fn task_mentions_language(task: &str, language: &str) -> bool {
482    task.split(|character: char| !character.is_alphanumeric())
483        .filter(|word| !word.is_empty())
484        .any(|word| {
485            if language == "go" {
486                word == "Go" || word.eq_ignore_ascii_case("golang")
487            } else {
488                word.eq_ignore_ascii_case(language)
489            }
490        })

493impl Services {
494    fn validate_context_request(&self, request: &ContextRequest) -> Result<()> {
495        if request.task.trim().is_empty() {
496            return Err(Error::InvalidInput {
497                field: "task",
498                reason: "must not be empty",
499            });
500        }
501        self.token_budget_limit(request.token_budget)?;
502        validate_input(&request.task, "task", MAX_QUERY_BYTES)?;
503        validate_glob_patterns(&request.focus_paths)?;
504        validate_glob_patterns(&request.exclude_paths)?;
505        if request.focus_symbols.len() > MAX_INPUT_ITEMS {
506            return Err(Error::LimitExceeded);
507        }
508        for symbol in &request.focus_symbols {
509            validate_input(symbol, "focus symbol", MAX_PATTERN_BYTES)?;
510        }
511        if request.known_hashes.len() > MAX_INPUT_ITEMS {
512            return Err(Error::LimitExceeded);
513        }
514        for hash in &request.known_hashes {
515            validate_input(hash, "known hash", 128)?;
516        }
517        if request.changed_paths.len() > MAX_DIFF_CHANGED_PATHS {
518            return Err(Error::LimitExceeded);
519        }
520        for path in &request.changed_paths {
521            validate_input(path, "changed path", MAX_PATH_BYTES)?;
522            validate_relative(path)?;
523        }
524        if let Some(revision) = request
525            .base_revision
526            .as_deref()
527            .filter(|revision| !revision.trim().is_empty())
528        {
529            validate_input(revision, "base revision", MAX_BASE_REVISION_BYTES)?;
530        }
531        for query in facets::plan(&request.task, MAX_CONTEXT_QUERIES)
532            .queries
533            .iter()
534            .filter(|query| !query.has_facet(FacetKind::TestIntent))
535        {
536            compile_literal_regex(&query.value, false)?;
537        }
538        Ok(())
539    }

562    fn append_import_symbol_candidates(
563        &self,
564        expansion: ImportExpansion<'_>,
565        candidates: &mut Vec<Candidate>,
566    ) -> Result<()> {
567        let seed_paths = import_seed_paths(candidates, expansion.queries, self.config.tokenizer);
568        let requested_paths = seed_paths
569            .iter()
570            .take(24)
571            .map(|(path, _)| path.clone())
572            .collect::<Vec<_>>();
573        let targets =
574            expansion
575                .session
576                .import_symbol_targets(&requested_paths, 32, MAX_IMPORT_SYMBOLS)?;
577        let mut pending = Vec::new();
578        for target in targets {
579            check_cancelled(expansion.cancellation)?;
580            let Some((_, seed_concepts)) = seed_paths.get(target.seed_index) else {
581                continue;
582            };
583            let target_path = &target.target_file.path;
584            if !path_allowed(target_path, &[], &expansion.request.exclude_paths)? {
585                continue;
586            }
587            let Some((symbol, query, exact)) =
588                corroborated_import_symbol(target.symbols, expansion.queries, seed_concepts)
589            else {
590                continue;
591            };
592            pending.push((target.target_file, symbol, query.clone(), exact));
593        }
594        let excerpt_requests = pending
595            .iter()
596            .map(|(target_file, symbol, _, _)| AdaptiveExcerptRequest {
597                file_id: target_file.id,
598                declaration_start: symbol.start_line,
599                declaration_end: symbol.end_line,
600                matched_line: symbol.start_line,
601                token_budget: excerpt_budget(
602                    expansion.request.token_budget,
603                    ContextExcerptKind::ImportSymbol,
604                ),
605            })
606            .collect::<Vec<_>>();
607        let excerpts = self.adaptive_context_excerpts(expansion.session, &excerpt_requests)?;
608        let mut neighbor_count = 0usize;
609        let mut neighbor_ranges = BTreeSet::new();
610        for ((target_file, symbol, query, exact), excerpt) in pending.into_iter().zip(excerpts) {
611            check_cancelled(expansion.cancellation)?;
612            let Some(excerpt) = excerpt else { continue };
613            let target_path = target_file.path;
614            if !neighbor_ranges.insert((target_path.clone(), excerpt.start_line, excerpt.end_line))
615            {
616                continue;
617            }
618            let change_boost = Self::file_change_boost(
619                Some(target_file.generation),
620                &target_path,
621                expansion.changed_paths,
622                expansion.request.prior_repository_generation,
623            );
624            let candidate = Candidate::new(
625                &target_path,
626                excerpt.start_line,
627                excerpt.end_line,
628                excerpt.content,
629            )
630            .match_kind("import")
631            .match_kind("symbol")
632            .concept(&query.fusion_key, query.concept_weight)
633            .representation("import_symbol")
634            .symbol_name(symbol.name)
635            .exact(exact)
636            .symbol(1.0)
637            .path_score(context_path_score(
638                &target_path,
639                expansion.terms,
640                &expansion.request.task,
641            ))
642            .import_boost(1.0)
643            .change_boost(change_boost);
644            candidates.push(annotate_candidate(
645                candidate,
646                &query,
647                "import_symbol",
648                neighbor_count,
649            ));
650            neighbor_count += 1;
651            if neighbor_count >= 24 {
652                break;
653            }
654        }
655        Ok(())

659        &self,
660        session: &ReadSession,
661        queries: &[ContextQuery],
662        candidates: &mut [Candidate],
663    ) -> Result<()> {
664        let seed_paths = import_seed_paths(candidates, queries, self.config.tokenizer)

666            .take(24)

682                candidate.import_boost = candidate.import_boost.max(1.0);
683            }
684        }
685        Ok(())

688    /// Resolve a diff scope from the request into a receipt, if one is supplied.
689    ///
690    /// When `base_revision` is set, committed and working-tree paths since that
691    /// revision are resolved from the repository, including untracked files.
692    /// Explicit `changed_paths` are merged with that result. When neither input
693    /// is supplied, `None` is returned and task-only behavior is preserved.
694    fn resolve_diff_scope(&self, request: &ContextRequest) -> Result<Option<DiffScopeReceipt>> {
695        let has_base = request
696            .base_revision
697            .as_deref()
698            .is_some_and(|rev| !rev.trim().is_empty());
699        let has_paths = !request.changed_paths.is_empty();
700        if !has_base && !has_paths {
701            return Ok(None);
702        }
703        if let Some(ref revision) = request.base_revision
704            && !revision.trim().is_empty()
705        {
706            let git_result = git_diff_paths(&self.config.root, revision, MAX_DIFF_CHANGED_PATHS)?;
707            let mut changed_paths = request.changed_paths.clone();
708            let mut resolved_paths = git_result.changed_paths;
709            match git_changed_paths(&self.config.root, MAX_DIFF_CHANGED_PATHS) {
710                Ok(working_tree_paths) => resolved_paths.extend(working_tree_paths),
711                Err(error) => {
712                    tracing::debug!(%error, "working-tree diff scope unavailable");
713                }
714            }
715            resolved_paths.sort();
716            resolved_paths.dedup();
717            for path in resolved_paths {
718                if changed_paths.len() == MAX_DIFF_CHANGED_PATHS {
719                    break;
720                }
721                if !changed_paths.contains(&path) {
722                    changed_paths.push(path);
723                }
724            }
725            changed_paths.sort();
726            changed_paths.dedup();
727            return Ok(Some(DiffScopeReceipt {
728                base_revision: Some(git_result.base_revision),
729                head_revision: Some(git_result.head_revision),
730                changed_paths,
731                indexed_changed_paths: 0,
732                evidence: None,
733            }));
734        }
735        Ok(Some(DiffScopeReceipt {
736            base_revision: None,
737            head_revision: None,
738            changed_paths: request.changed_paths.clone(),
739            indexed_changed_paths: 0,
740            evidence: None,
741        }))
742    }

863    fn append_workflow_candidates(
864        &self,
865        session: &ReadSession,
866        request: &ContextRequest,
867        workflow: ContextWorkflow,
868        cancellation: &CancellationToken,
869        candidates: &mut Vec<Candidate>,
870    ) -> Result<Option<WorkflowReceipt>> {
871        if !matches!(
872            workflow,
873            ContextWorkflow::Contribution | ContextWorkflow::Review
874        ) {
875            return Ok(None);
876        }
877
878        let mut matches = Vec::new();
879        let mut cursor = None;
880        let mut scanned_files = 0;
881        let mut scan_truncated = false;
882        loop {
883            check_cancelled(cancellation)?;
884            let page = session.list_files(512, cursor)?;
885            let Some(last) = page.last() else {
886                break;
887            };
888            cursor = Some(last.id);
889            for file in page {
890                if scanned_files == MAX_WORKFLOW_SCAN_FILES {
891                    scan_truncated = true;
892                    break;
893                }
894                scanned_files += 1;
895                if let Some((score, family)) = workflow_path_role(&file.path, request) {
896                    matches.push((file, score, family));
897                }
898            }
899            if scan_truncated {
900                break;
901            }
902        }
903        let count = |family| {
904            matches
905                .iter()
906                .filter(|(_, _, candidate_family)| *candidate_family == family)
907                .count()
908        };
909        let guidance_candidates = count("guidance");
910        let template_candidates = count("template");
911        let validation_candidates = count("validation");
912        let owner_test_candidates = count("owner_test");
913        let mut missing_families = Vec::new();
914        for (family, candidates) in [
915            ("guidance", guidance_candidates),
916            ("template", template_candidates),
917            ("validation", validation_candidates),
918            ("owner_test", owner_test_candidates),
919        ] {
920            if candidates == 0 {
921                missing_families.push(family.to_owned());
922            }
923        }
924        if scan_truncated {
925            missing_families.push("repository_scan_truncated".to_owned());
926        }
927        matches.sort_by(|left, right| {
928            right
929                .1
930                .total_cmp(&left.1)
931                .then_with(|| left.0.path.cmp(&right.0.path))
932        });
933        matches.truncate(24);
934
935        let excerpt_requests = matches
936            .iter()
937            .map(|(file, _, _)| StoredExcerptRequest {
938                file_id: file.id,
939                desired_start_line: 1,
940                desired_end_line: 80,
941                required_start_line: 1,
942                required_end_line: 1,
943                max_lines: 80,
944            })
945            .collect::<Vec<_>>();
946        for ((file, score, family), excerpt) in matches
947            .into_iter()
948            .zip(self.stored_excerpts(session, &excerpt_requests)?)
949        {
950            let Some(excerpt) = excerpt else { continue };
951            candidates.push(
952                Candidate::new(
953                    file.path,
954                    excerpt.start_line,
955                    excerpt.end_line,
956                    excerpt.content,
957                )
958                .match_kind(format!("workflow_{family}"))
959                .representation("workflow")
960                .path_score(score)
961                .focus_boost(score),
962            );
963        }
964        Ok(Some(WorkflowReceipt {
965            guidance_candidates,
966            template_candidates,
967            validation_candidates,
968            owner_test_candidates,
969            missing_families,
970        }))
971    }

973    fn build_diff_evidence(
974        &self,
975        session: &ReadSession,
976        request: &ContextRequest,
977        scope: &DiffScopeReceipt,
978        cancellation: &CancellationToken,
979    ) -> Result<DiffEvidenceReceipt> {
980        let mut changed_symbols = Vec::new();
981        let mut relationships = BTreeSet::new();
982        let mut gaps = Vec::new();
983        let changed_hunks = if let Some(base_revision) = &scope.base_revision {
984            let mut hunks = git_diff_hunks(
985                &self.config.root,
986                base_revision,
987                MAX_DIFF_EVIDENCE_SYMBOLS + 1,
988            )?;
989            if hunks.len() > MAX_DIFF_EVIDENCE_SYMBOLS {
990                gaps.push("changed_hunk_evidence_truncated".into());
991                hunks.truncate(MAX_DIFF_EVIDENCE_SYMBOLS);
992            }
993            hunks
994                .into_iter()
995                .map(|hunk| DiffHunkEvidence {
996                    path: hunk.path,
997                    start_line: hunk.start_line,
998                    end_line: hunk.end_line,
999                })
1000                .collect::<Vec<_>>()
1001        } else {
1002            gaps.push("hunk_ranges_unavailable_for_explicit_paths".into());
1003            Vec::new()
1004        };
1005        let scoped_paths = scope
1006            .changed_paths
1007            .iter()
1008            .take(MAX_DIFF_EVIDENCE_PATHS)
1009            .collect::<Vec<_>>();
1010        if scope.changed_paths.len() > scoped_paths.len() {
1011            gaps.push("changed_path_evidence_truncated".into());
1012        }
1013
1014        for path in &scoped_paths {
1015            check_cancelled(cancellation)?;
1016            let Some(file) = session.find_file(path)? else {
1017                gaps.push(format!("{path}:not_indexed_or_deleted"));
1018                continue;
1019            };
1020            if !file.structurally_complete {
1021                gaps.push(format!("{path}:structural_coverage_incomplete"));
1022            }
1023            let symbols = session.get_symbols_for_file(file.id, MAX_DIFF_EVIDENCE_SYMBOLS)?;
1024            let path_hunks = changed_hunks
1025                .iter()
1026                .filter(|hunk| hunk.path == **path)
1027                .collect::<Vec<_>>();
1028            let symbols = symbols
1029                .into_iter()
1030                .filter(|symbol| {
1031                    path_hunks.is_empty()
1032                        || path_hunks.iter().any(|hunk| {
1033                            hunk.start_line <= hunk.end_line
1034                                && symbol.start_line <= hunk.end_line
1035                                && symbol.end_line >= hunk.start_line
1036                        })
1037                })
1038                .collect::<Vec<_>>();
1039            if symbols.is_empty() && file.structurally_complete {
1040                gaps.push(format!("{path}:no_indexed_definitions"));
1041            }
1042            for symbol in symbols {
1043                if changed_symbols.len() == MAX_DIFF_EVIDENCE_SYMBOLS {
1044                    gaps.push("changed_symbol_evidence_truncated".into());
1045                    break;
1046                }
1047                let mut references = session.search_references(
1048                    &symbol.name,
1049                    true,
1050                    MAX_REFERENCES_PER_CHANGED_SYMBOL + 1,
1051                )?;
1052                if references.len() > MAX_REFERENCES_PER_CHANGED_SYMBOL {
1053                    gaps.push(format!(
1054                        "{path}:{}:reference_evidence_truncated",
1055                        symbol.name
1056                    ));
1057                    references.truncate(MAX_REFERENCES_PER_CHANGED_SYMBOL);
1058                }
1059                for reference in references {
1060                    if reference.path != **path {
1061                        relationships.insert((
1062                            (**path).clone(),
1063                            reference.path,
1064                            "reference".to_owned(),
1065                        ));
1066                    }
1067                }
1068                changed_symbols.push(DiffSymbolEvidence {
1069                    path: (**path).clone(),
1070                    name: symbol.name,
1071                    kind: symbol.kind,
1072                    start_line: symbol.start_line,
1073                    end_line: symbol.end_line,
1074                });
1075            }
1076            for importer in session.affected_importers(&[(**path).clone()])? {
1077                if importer != **path {
1078                    relationships.insert(((**path).clone(), importer, "importer".to_owned()));
1079                }
1080            }
1081        }
1082
1083        let mut cursor = None;
1084        let mut scanned_owner_test_files = 0;
1085        let mut owner_test_scan_truncated = false;
1086        loop {
1087            check_cancelled(cancellation)?;
1088            let page = session.list_files(512, cursor)?;
1089            let Some(last) = page.last() else {
1090                break;
1091            };
1092            cursor = Some(last.id);
1093            for file in page {
1094                if scanned_owner_test_files == MAX_OWNER_TEST_SCAN_FILES {
1095                    owner_test_scan_truncated = true;
1096                    break;
1097                }
1098                scanned_owner_test_files += 1;
1099                if let Some(changed_path) = owner_test_changed_path(&file.path, request) {
1100                    relationships.insert((changed_path, file.path, "test_name_match".to_owned()));
1101                }
1102            }
1103            if owner_test_scan_truncated {
1104                break;
1105            }
1106        }
1107        if owner_test_scan_truncated {
1108            gaps.push("owner_test_scan_truncated".into());
1109        }
1110
1111        let relationship_count = relationships.len();
1112        let related_paths = relationships
1113            .into_iter()
1114            .take(MAX_DIFF_EVIDENCE_RELATIONSHIPS)
1115            .map(|(changed_path, related_path, signal)| DiffRelatedPath {
1116                changed_path,
1117                related_path,
1118                signal,
1119            })
1120            .collect();
1121        if relationship_count > MAX_DIFF_EVIDENCE_RELATIONSHIPS {
1122            gaps.push("related_path_evidence_truncated".into());
1123        }
1124        gaps.sort();
1125        gaps.dedup();
1126
1127        Ok(DiffEvidenceReceipt {
1128            changed_hunks,
1129            changed_symbols,
1130            related_paths,
1131            gaps,
1132        })
1133    }

1135    #[allow(clippy::cognitive_complexity)]
1136    fn context_sync(
1137        &self,
1138        mut request: ContextRequest,
1139        workflow: ContextWorkflow,
1140        cancellation: &CancellationToken,
1141        diagnostics: CandidateDiagnostics,
1142        signals: ContextSignals,
1143    ) -> Result<(ContextEvaluation, Option<usize>)> {
1144        check_cancelled(cancellation)?;
1145        self.validate_context_request(&request)?;
1146        request.changed_paths = request
1147            .changed_paths
1148            .iter()
1149            .map(|path| normalize_relative(path))
1150            .collect::<Result<Vec<_>>>()?;
1151        let diff_scope = self.resolve_diff_scope(&request)?;
1152        let mut scoped_request = request.clone();
1153        if let Some(scope) = &diff_scope {
1154            scoped_request.changed_paths = scope.changed_paths.clone();
1155        }
1156
1157        let mut changed_paths = git_changed_paths(&self.config.root, GIT_CHANGED_PATHS_MAX)
1158            .unwrap_or_else(|error| {
1159                tracing::debug!(%error, "working-tree signal unavailable");
1160                HashSet::new()
1161            });
1162        if let Some(ref scope) = diff_scope {
1163            changed_paths.extend(scope.changed_paths.iter().cloned());
1164        }
1165        self.consistent(|session, generation| {
1166            let facet_plan = facets::plan(&request.task, MAX_CONTEXT_QUERIES);
1167            let queries = facet_plan.queries;
1168            let terms = queries
1169                .iter()
1170                .map(|query| query.value.clone())
1171                .collect::<Vec<_>>();
1172            let mut candidates = Vec::new();
1173            let mut query_fusion = HashMap::<String, HashMap<String, f64>>::new();
1174
1175            // Workflow words such as `test` are useful path priors but terrible
1176            // retrieval queries: nearly every test function becomes a high-
1177            // scoring symbol candidate. Keep them out of candidate generation.
1178            for query in queries
1179                .iter()
1180                .filter(|query| !query.has_facet(FacetKind::TestIntent))
1181            {
1182                let term = &query.value;
1183                let concept = query.fusion_key.as_str();
1184                check_cancelled(cancellation)?;
1185                let mut symbol_hits = Vec::new();
1186                for (rank, hit) in session
1187                    .search_symbols(term, false, MAX_CONTEXT_HITS_PER_SOURCE)?
1188                    .into_iter()
1189                    .enumerate()
1190                {
1191                    check_cancelled(cancellation)?;
1192                    if path_allowed(&hit.path, &[], &request.exclude_paths)? {
1193                        symbol_hits.push((rank, hit));
1194                    }
1195                }
1196                let symbol_excerpt_requests = symbol_hits
1197                    .iter()
1198                    .map(|(_, hit)| AdaptiveExcerptRequest {
1199                        file_id: hit.symbol.file_id,
1200                        declaration_start: hit.symbol.start_line,
1201                        declaration_end: hit.symbol.end_line,
1202                        matched_line: hit.symbol.start_line,
1203                        token_budget: excerpt_budget(
1204                            request.token_budget,
1205                            ContextExcerptKind::Symbol,
1206                        ),
1207                    })
1208                    .collect::<Vec<_>>();
1209                for ((rank, hit), excerpt) in symbol_hits
1210                    .into_iter()
1211                    .zip(self.adaptive_context_excerpts(session, &symbol_excerpt_requests)?)
1212                {
1213                    check_cancelled(cancellation)?;
1214                    let Some(excerpt) = excerpt else { continue };
1215                    let exact = f64::from(hit.symbol.name.eq_ignore_ascii_case(term));
1216                    let qualified = qualified_symbol_match(
1217                        concept,
1218                        &hit.symbol.name,
1219                        hit.symbol.parent.as_deref(),
1220                        hit.symbol.signature.as_deref(),
1221                    );
1222                    if query.fuse {
1223                        record_query_hit(
1224                            &mut query_fusion,
1225                            &hit.path,
1226                            &query.fusion_key,
1227                            query.weight,
1228                            rank,
1229                        );
1230                    }
1231                    let change_boost = Self::file_change_boost(
1232                        Some(hit.generation),
1233                        &hit.path,
1234                        &changed_paths,
1235                        request.prior_repository_generation,
1236                    );
1237                    let candidate = Candidate::new(
1238                        &hit.path,
1239                        excerpt.start_line,
1240                        excerpt.end_line,
1241                        excerpt.content,
1242                    )
1243                    .match_kind("symbol")
1244                    .concept(concept, query.concept_weight)
1245                    .representation("symbol")
1246                    .symbol_name(hit.symbol.name)
1247                    .exact(exact + qualified * 1.5)
1248                    .symbol(1.0)
1249                    .path_score(context_path_score(&hit.path, &terms, &request.task))
1250                    .change_boost(change_boost);
1251                    candidates.push(annotate_candidate(candidate, query, "symbol", rank));
1252                }
1253                let reference_results = signals
1254                    .caller
1255                    .then(|| session.search_references(term, false, MAX_CONTEXT_HITS_PER_SOURCE))
1256                    .transpose()?
1257                    .unwrap_or_default();
1258                let mut reference_hits = Vec::new();
1259                for (rank, hit) in reference_results.into_iter().enumerate() {
1260                    check_cancelled(cancellation)?;
1261                    if path_allowed(&hit.path, &[], &request.exclude_paths)? {
1262                        reference_hits.push((rank, hit));
1263                    }
1264                }
1265                let reference_locations = reference_hits
1266                    .iter()
1267                    .map(|(_, hit)| (hit.reference.file_id, hit.reference.start_line))
1268                    .collect::<Vec<_>>();
1269                let enclosing = session.find_enclosing_symbols_batch(&reference_locations)?;
1270                let mut adaptive_indices = Vec::new();
1271                let mut adaptive_requests = Vec::new();
1272                for (index, ((_, hit), symbol)) in reference_hits.iter().zip(enclosing).enumerate()
1273                {
1274                    if let Some(symbol) = symbol {
1275                        adaptive_indices.push(index);
1276                        adaptive_requests.push(AdaptiveExcerptRequest {
1277                            file_id: hit.reference.file_id,
1278                            declaration_start: symbol.start_line,
1279                            declaration_end: symbol.end_line,
1280                            matched_line: hit.reference.start_line,
1281                            token_budget: excerpt_budget(
1282                                request.token_budget,
1283                                ContextExcerptKind::Reference,
1284                            ),
1285                        });
1286                    }
1287                }
1288                let mut adaptive_excerpts = vec![None; reference_hits.len()];
1289                for (index, excerpt) in adaptive_indices
1290                    .into_iter()
1291                    .zip(self.adaptive_context_excerpts(session, &adaptive_requests)?)
1292                {
1293                    adaptive_excerpts[index] = excerpt;
1294                }
1295                let fallback_requests = reference_hits
1296                    .iter()
1297                    .map(|(_, hit)| StoredExcerptRequest {
1298                        file_id: hit.reference.file_id,
1299                        desired_start_line: hit.reference.start_line.saturating_sub(2).max(1),
1300                        desired_end_line: hit.reference.end_line.saturating_add(2),
1301                        required_start_line: hit.reference.start_line,
1302                        required_end_line: hit.reference.end_line,
1303                        max_lines: 12,
1304                    })
1305                    .collect::<Vec<_>>();
1306                let fallback_excerpts = self.stored_excerpts(session, &fallback_requests)?;
1307                for (((rank, hit), adaptive), fallback) in reference_hits
1308                    .into_iter()
1309                    .zip(adaptive_excerpts)
1310                    .zip(fallback_excerpts)
1311                {
1312                    check_cancelled(cancellation)?;
1313                    let excerpt = adaptive.or(fallback);
1314                    let Some(excerpt) = excerpt else {
1315                        continue;
1316                    };
1317                    if query.fuse {
1318                        record_query_hit(
1319                            &mut query_fusion,
1320                            &hit.path,
1321                            &query.fusion_key,
1322                            query.weight,
1323                            rank,
1324                        );
1325                    }
1326                    let change_boost = Self::file_change_boost(
1327                        Some(hit.generation),
1328                        &hit.path,
1329                        &changed_paths,
1330                        request.prior_repository_generation,
1331                    );
1332                    let candidate = Candidate::new(
1333                        &hit.path,
1334                        excerpt.start_line,
1335                        excerpt.end_line,
1336                        excerpt.content,
1337                    )
1338                    .match_kind("reference")
1339                    .concept(concept, query.concept_weight)
1340                    .symbol_name(hit.reference.name)
1341                    .reference(1.0)
1342                    .path_score(context_path_score(&hit.path, &terms, &request.task))
1343                    .change_boost(change_boost);
1344                    candidates.push(annotate_candidate(candidate, query, "reference", rank));
1345                }
1346                let term_regex = compile_literal_regex(term, false)?;
1347                let lexical = if term.chars().count() >= 3 {
1348                    session.search_trigram(term, MAX_CONTEXT_LEXICAL_HITS)?
1349                } else {
1350                    session.search_word(&fts_quote(term), MAX_CONTEXT_LEXICAL_HITS)?
1351                };
1352                let mut lexical_hits = Vec::new();
1353                for (rank, hit) in lexical.into_iter().enumerate() {
1354                    check_cancelled(cancellation)?;
1355                    if !path_allowed(&hit.path, &[], &request.exclude_paths)? {
1356                        continue;
1357                    }
1358                    let Some(search_hit) =
1359                        chunk_search_hit(hit.clone(), term, false, 2, term_regex.as_ref(), false)?
1360                    else {
1361                        continue;
1362                    };
1363                    let matched_line = matching_line(&hit, term, false, term_regex.as_ref())
1364                        .unwrap_or(search_hit.start_line);
1365                    lexical_hits.push((rank, hit, search_hit, matched_line));
1366                }
1367                let lexical_locations = lexical_hits
1368                    .iter()
1369                    .map(|(_, hit, _, matched_line)| (hit.file_id, *matched_line))
1370                    .collect::<Vec<_>>();
1371                let enclosing = session.find_enclosing_symbols_batch(&lexical_locations)?;
1372                let mut adaptive_indices = Vec::new();
1373                let mut adaptive_requests = Vec::new();
1374                for (index, ((_, hit, _, matched_line), symbol)) in
1375                    lexical_hits.iter().zip(enclosing).enumerate()
1376                {
1377                    if let Some(symbol) = symbol {
1378                        adaptive_indices.push(index);
1379                        adaptive_requests.push(AdaptiveExcerptRequest {
1380                            file_id: hit.file_id,
1381                            declaration_start: symbol.start_line,
1382                            declaration_end: symbol.end_line,
1383                            matched_line: *matched_line,
1384                            token_budget: excerpt_budget(
1385                                request.token_budget,
1386                                ContextExcerptKind::Text,
1387                            ),
1388                        });
1389                    }
1390                }
1391                let mut adaptive_excerpts = vec![None; lexical_hits.len()];
1392                for (index, excerpt) in adaptive_indices
1393                    .into_iter()
1394                    .zip(self.adaptive_context_excerpts(session, &adaptive_requests)?)
1395                {
1396                    adaptive_excerpts[index] = excerpt;
1397                }
1398                for ((rank, hit, search_hit, _), adaptive) in
1399                    lexical_hits.into_iter().zip(adaptive_excerpts)
1400                {
1401                    check_cancelled(cancellation)?;
1402                    let excerpt = adaptive.unwrap_or(StoredExcerpt {
1403                        content: search_hit.excerpt.clone(),
1404                        start_line: search_hit.start_line,
1405                        end_line: search_hit.end_line,
1406                    });
1407                    if query.fuse {
1408                        record_query_hit(
1409                            &mut query_fusion,
1410                            &search_hit.path,
1411                            &query.fusion_key,
1412                            query.weight,
1413                            rank,
1414                        );
1415                    }
1416                    let occurrences = term_regex
1417                        .as_ref()
1418                        .map_or(0, |matcher| matcher.find_iter(&hit.content).count());
1419                    let change_boost = Self::file_change_boost(
1420                        Some(hit.generation),
1421                        &search_hit.path,
1422                        &changed_paths,
1423                        request.prior_repository_generation,
1424                    );
1425                    let candidate = Candidate::new(
1426                        &search_hit.path,
1427                        excerpt.start_line,
1428                        excerpt.end_line,
1429                        excerpt.content,
1430                    )
1431                    .match_kind("text")
1432                    .concept(concept, query.concept_weight)
1433                    .exact(query.weight)
1434                    .bm25((-hit.score).max(0.0) * 1_000_000.0)
1435                    .path_score(context_path_score(&search_hit.path, &terms, &request.task))
1436                    .lexical_frequency_penalty(
1437                        (occurrences.saturating_sub(5) as f64 / 20.0).min(1.0),
1438                    )
1439                    .change_boost(change_boost);
1440                    candidates.push(annotate_candidate(candidate, query, "text", rank));
1441                }
1442            }
1443
1444            apply_query_fusion(&mut candidates, &query_fusion);
1445            let resolved_workflow = resolve_context_workflow(workflow, &request.task);
1446            let workflow_receipt = self.append_workflow_candidates(
1447                session,
1448                &scoped_request,
1449                resolved_workflow,
1450                cancellation,
1451                &mut candidates,
1452            )?;
1453
1454            signals
1455                .import_neighbor
1456                .then(|| {
1457                    self.append_import_symbol_candidates(
1458                        ImportExpansion {
1459                            session,
1460                            request: &request,
1461                            queries: &queries,
1462                            terms: &terms,
1463                            changed_paths: &changed_paths,
1464                            cancellation,
1465                        },
1466                        &mut candidates,
1467                    )
1468                })
1469                .transpose()?;
1470            signals
1471                .reverse_dependency
1472                .then(|| self.apply_reverse_dependency_boost(session, &queries, &mut candidates))
1473                .transpose()?;
1474
1475            let generated_candidate_paths = if diagnostics == CandidateDiagnostics::Collect {
1476                candidates
1477                    .iter()
1478                    .map(|candidate| candidate.path.clone())
1479                    .collect::<BTreeSet<_>>()
1480                    .into_iter()
1481                    .collect()
1482            } else {
1483                Vec::new()
1484            };
1485            let generated_candidates = if diagnostics == CandidateDiagnostics::Collect {
1486                candidates
1487                    .iter()
1488                    .map(|candidate| {
1489                        let token_count = candidate.token_count_with(self.config.tokenizer).max(1);
1490                        ContextCandidateEvaluation {
1491                            path: candidate.path.clone(),
1492                            start_line: candidate.start_line,
1493                            end_line: candidate.end_line,
1494                            representation: candidate.representation.clone(),
1495                            match_kinds: candidate.match_kinds.clone(),
1496                            concepts: candidate.concepts.clone(),
1497                            concept_weight: candidate.concept_weight,
1498                            score: candidate.score(&ranking::Weights::default(), token_count),
1499                            token_count,
1500                        }
1501                    })
1502                    .collect()
1503            } else {
1504                Vec::new()
1505            };
1506            let mut response = ranking::select_with_tokenizer(
1507                candidates,
1508                &request,
1509                generation,
1510                self.config.tokenizer,
1511            );
1512            response.workflow = resolved_workflow;
1513            response.workflow_receipt = workflow_receipt;
1514            response.meta.freshness = self.freshness();
1515            response.meta.repository_id = self.repository_id();
1516            if let Some(mut scope) = diff_scope.clone() {
1517                let mut indexed = 0usize;
1518                for path in &scope.changed_paths {
1519                    if session.find_file(path)?.is_some() {
1520                        indexed += 1;
1521                    }
1522                }
1523                scope.indexed_changed_paths = indexed;
1524                scope.evidence = Some(self.build_diff_evidence(
1525                    session,
1526                    &scoped_request,
1527                    &scope,
1528                    cancellation,
1529                )?);
1530                response.diff_scope = Some(scope);
1531            }
1532            if response.fragments.is_empty() {
1533                response
1534                    .warnings
1535                    .push("no relevant indexed evidence found".into());
1536            }
1537            let paths = response
1538                .fragments
1539                .iter()
1540                .map(|fragment| fragment.path.clone())
1541                .collect::<BTreeSet<_>>()
1542                .into_iter()
1543                .collect::<Vec<_>>();
1544            let baseline_source_tokens =
1545                session.whole_file_source_tokens(&paths, self.config.tokenizer.name())?;
1546            Ok((
1547                ContextEvaluation {
1548                    response,
1549                    generated_candidate_paths,
1550                    generated_candidates,
1551                },
1552                baseline_source_tokens,
1553            ))
1554        })
1555    }
