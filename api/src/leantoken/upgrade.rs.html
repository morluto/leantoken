3use std::{
4    env,

9use serde::Serialize;

11use crate::{Error, Result};

13const NPM_PACKAGE: &str = "leantoken@latest";
14const GIT_REPOSITORY: &str = "https://github.com/morluto/leantoken";
15
16#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize)]
17#[serde(rename_all = "lowercase")]
18enum InstallMethod {
19    Npm,
20    Cargo,
21}
22
23#[derive(Debug, Serialize)]
24struct UpgradeReport {
25    status: &'static str,
26    method: InstallMethod,
27}
28
29/// Update LeanToken through the package manager that owns the executable.
30///
31/// # Errors
32///
33/// Returns an error when the installation method cannot be identified or the
34/// package manager cannot install the latest release.
35pub fn run(json: bool) -> Result<()> {
36    let executable = env::current_exe()?.canonicalize()?;
37    let method = detect_install_method(&executable, env::var_os("npm_execpath").is_some())
38        .ok_or_else(unknown_install_error)?;
39
40    let (program, arguments): (&str, &[&str]) = match method {
41        InstallMethod::Npm => ("npm", &["install", "--global", NPM_PACKAGE]),
42        InstallMethod::Cargo => ("cargo", &["install", "--git", GIT_REPOSITORY, "--force"]),
43    };
44
45    let output = Command::new(program)
46        .args(arguments)
47        .output()
48        .map_err(|error| Error::InvalidRequest(format!("failed to run {program}: {error}")))?;
49    require_success(program, arguments, &output)?;
50
51    let report = UpgradeReport {
52        status: "updated",
53        method,
54    };
55    if json {
56        println!("{}", serde_json::to_string(&report)?);
57    } else {
58        println!("LeanToken update completed through {program}.");
59    }
60    Ok(())
61}
62
63fn detect_install_method(executable: &Path, npm_environment: bool) -> Option<InstallMethod> {
64    if npm_environment || path_contains(executable, "node_modules") {
65        return Some(InstallMethod::Npm);
66    }
67    if path_contains(executable, ".cargo") {
68        return Some(InstallMethod::Cargo);
69    }
70    None
71}
72
73fn path_contains(path: &Path, component: &str) -> bool {
74    path.components()
75        .any(|part| part.as_os_str() == std::ffi::OsStr::new(component))
76}
77
78fn unknown_install_error() -> Error {
79    Error::InvalidRequest(format!(
80        "could not identify the LeanToken installation method; update manually with \
81         `npm install --global {NPM_PACKAGE}` or \
82         `cargo install --git {GIT_REPOSITORY} --force`"
83    ))
84}
85
86fn require_success(program: &str, arguments: &[&str], output: &Output) -> Result<()> {
87    if output.status.success() {
88        return Ok(());
89    }
90    let stderr = String::from_utf8_lossy(&output.stderr);
91    let detail = stderr.trim();
92    let command = std::iter::once(program)
93        .chain(arguments.iter().copied())
94        .collect::<Vec<_>>()
95        .join(" ");
96    Err(Error::InvalidRequest(if detail.is_empty() {
97        format!("update command failed: {command}")
98    } else {
99        format!("update command failed: {command}: {detail}")
100    }))
101}

103#[cfg(test)]
104mod tests {
105    use super::*;
106
107    #[test]
108    fn detects_npm_from_environment_or_package_path() {
109        assert_eq!(
110            detect_install_method(Path::new("/tmp/leantoken"), true),
111            Some(InstallMethod::Npm)
112        );
113        assert_eq!(
114            detect_install_method(
115                Path::new("/usr/lib/node_modules/leantoken/bin/leantoken"),
116                false
117            ),
118            Some(InstallMethod::Npm)
119        );
120    }
121
122    #[test]
123    fn detects_cargo_and_rejects_unknown_paths() {
124        assert_eq!(
125            detect_install_method(Path::new("/home/me/.cargo/bin/leantoken"), false),
126            Some(InstallMethod::Cargo)
127        );
128        assert_eq!(
129            detect_install_method(Path::new("/usr/local/bin/leantoken"), false),
130            None
131        );
132    }
133
134    #[cfg(unix)]
135    #[test]
136    fn failed_commands_include_stderr() {
137        use std::os::unix::process::ExitStatusExt;
138        use std::process::ExitStatus;
139
140        let output = Output {
141            status: ExitStatus::from_raw(1 << 8),
142            stdout: Vec::new(),
143            stderr: b"permission denied\n".to_vec(),
144        };
145        let error = require_success("npm", &["install"], &output).unwrap_err();
146        assert!(error.to_string().contains("permission denied"));
147    }
