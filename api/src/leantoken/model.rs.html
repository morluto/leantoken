449    #[test]
450    fn compact_context_response_round_trips_with_defaults() {
451        let response = ContextResponse {
452            fragments: vec![ContextFragment {
453                path: "src/lib.rs".into(),
454                start_line: 1,
455                end_line: 2,
456                representation: "source".into(),
457                content: "fn answer() {}".into(),
458                content_hash: "receipt-hash".into(),
459                score: 2.0,
460                reason: "symbol".into(),
461                token_count: 4,
462            }],
463            receipt: EvidenceReceipt {
464                task_fingerprint: "task".into(),
465                fragment_hashes: vec!["receipt-hash".into()],
466            },
467            omitted: Vec::new(),
468            warnings: Vec::new(),
469            meta: ResponseMeta {
470                repository_generation: 7,
471                freshness: Freshness::Current,
472                emitted_tokens: 4,
473                token_count_exact: true,
474                next_cursor: None,
475            },
476        };
477
478        let value = serde_json::to_value(&response).expect("serialize response");
479        assert!(value["fragments"][0].get("representation").is_none());
480        assert!(value["fragments"][0].get("content_hash").is_none());
481        assert!(value["receipt"].get("task_fingerprint").is_none());
482        assert_eq!(value["meta"]["freshness"], "current");
483        assert_eq!(value["meta"]["token_count_exact"], true);
484        assert!(value.get("omitted").is_none());
485        assert!(value.get("warnings").is_none());
486
487        let round_trip: ContextResponse =
488            serde_json::from_value(value).expect("deserialize compact response");
489        assert_eq!(round_trip.fragments[0].representation, "source");
490        assert_eq!(round_trip.fragments[0].content_hash, "");
491        assert!(round_trip.receipt.task_fingerprint.is_empty());
492        assert_eq!(round_trip.meta.freshness, Freshness::Current);
493        assert!(round_trip.meta.token_count_exact);
494    }
