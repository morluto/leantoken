460/// Bounded aggregate counts for files skipped during index preparation.
461#[derive(Debug, Clone, Default, PartialEq, Eq, Serialize, Deserialize, JsonSchema)]
462pub struct IndexSkipReasonCounts {
463    /// Files detected as binary during preparation.
464    pub binary: usize,
465    /// Files admitted by discovery that exceeded the byte limit before reading completed.
466    pub oversized_during_read: usize,
467    /// Files whose preparation failed before searchable content could be produced.
468    pub failed: usize,
469}
470
471impl IndexSkipReasonCounts {
472    /// Return the total number of preparation skips across every public reason.
473    #[must_use]
474    pub fn total(&self) -> usize {
475        self.binary
476            .saturating_add(self.oversized_during_read)
477            .saturating_add(self.failed)
478    }
479}

496    #[serde(flatten)]
497    pub response: IndexResponse,
498    /// Known aggregate preparation skip counts whose sum equals `files_skipped`.
499    ///
500    /// Legacy deserialized responses omit this field because their reason
501    /// breakdown is unknown. Responses produced by this version always include

507impl IndexReport {

511        Self {
512            response,
513            skip_reasons: Some(skip_reasons),
514        }
515    }
516
517    /// Discard additive details and return the compatible index response.
518    #[must_use]
519    pub fn into_response(self) -> IndexResponse {
520        self.response
521    }

595        let report = IndexReport::with_skip_reasons(response, skip_reasons);
