3/// Repository indexing resource whose configured hard limit was exceeded.
4#[derive(Debug, Clone, Copy, PartialEq, Eq)]
5pub enum IndexLimitKind {
6    /// Filesystem entries yielded by repository traversal.
7    WalkEntries,
8    /// Files admitted to the source index.
9    Files,
10    /// Aggregate bytes of files admitted to the source index.
11    TotalSourceBytes,
12    /// Repository-relative traversal depth below the root.
13    Depth,
14}
15
16impl std::fmt::Display for IndexLimitKind {
17    fn fmt(&self, formatter: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
18        match self {
19            Self::WalkEntries => formatter.write_str("walk entries"),
20            Self::Files => formatter.write_str("source files"),
21            Self::TotalSourceBytes => formatter.write_str("total source bytes"),
22            Self::Depth => formatter.write_str("repository depth"),
23        }
24    }
25}

27/// Repository operation that may be retried after concurrent state changes.
28#[derive(Debug, Clone, Copy, PartialEq, Eq)]
29pub enum RetryableOperation {
30    /// Preparing and publishing an index generation.
31    Reconciliation,
32    /// Reading one consistent committed generation.
33    Retrieval,
34}
35
36impl std::fmt::Display for RetryableOperation {
37    fn fmt(&self, formatter: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
38        match self {
39            Self::Reconciliation => formatter.write_str("reconciliation"),
40            Self::Retrieval => formatter.write_str("retrieval"),
41        }
42    }
43}

45/// Errors returned by LeanToken operations.
46///
47/// Callers should match the variants they can recover from and retain a
48/// fallback arm for errors added by later releases.
49#[non_exhaustive]
50#[derive(Debug, thiserror::Error)]
51pub enum Error {
52    #[error("repository root does not exist: {0}")]
53    RootNotFound(PathBuf),
54    /// Automatic indexing refused a filesystem root, home directory, or parent
55    /// of the current user's home directory.
56    #[error(
57        "repository root is too broad for automatic indexing: {0}; pass --allow-broad-root to override"
58    )]
59    UnsafeRepositoryRoot(PathBuf),
60    /// Repository discovery stopped instead of returning a truncated index.
61    #[error("index {kind} limit exceeded: observed {observed}, limit {limit}")]
62    IndexLimitExceeded {
63        /// Resource whose configured bound was crossed.
64        kind: IndexLimitKind,
65        /// First observed value outside the configured bound.
66        observed: u64,
67        /// Configured inclusive maximum.
68        limit: u64,
69    },
70    #[error("path escapes repository root: {0}")]
71    PathOutsideRoot(PathBuf),
72    /// Filesystem path cannot be represented by the UTF-8 retrieval API.
73    #[error("repository path is not valid UTF-8: {0}")]
74    UnsupportedPathEncoding(PathBuf),
75    /// Ignore-aware traversal failed before complete membership was known.
76    #[error("repository traversal failed: {0}")]
77    RepositoryTraversal(#[from] ignore::Error),
78    #[error("path is not indexed: {0}")]
79    NotIndexed(String),
80    /// Requested symbol was absent from an indexed file.
81    #[error("symbol is not indexed in {path}: {symbol}")]
82    SymbolNotFound {
83        /// Repository-relative indexed file.
84        path: String,
85        /// Exact symbol identity requested by the caller.
86        symbol: String,
87    },
88    #[error("requested content exceeds the configured limit")]
89    LimitExceeded,
90    /// Caller-controlled response limit crossed its configured maximum.
91    #[error("{field} exceeds its configured limit: requested {requested}, limit {limit}")]
92    RequestLimitExceeded {
93        /// Safe public request field name.
94        field: &'static str,
95        /// Caller-provided value.
96        requested: usize,
97        /// Configured inclusive maximum.
98        limit: usize,
99    },
100    #[error("unsupported structured language for {0}")]
101    UnsupportedLanguage(String),
102    /// Request input failed a validation rule described entirely by static text.
103    #[error("invalid {field}: {reason}")]
104    InvalidInput {
105        /// Safe public field label.
106        field: &'static str,
107        /// Safe public validation rule.
108        reason: &'static str,
109    },
110    /// Request input crossed its configured byte bound.
111    #[error("{field} exceeds {max_bytes} bytes")]
112    InputTooLong {
113        /// Safe public field label.
114        field: &'static str,
115        /// Inclusive byte bound.
116        max_bytes: usize,
117    },
118    #[error("invalid request: {0}")]
119    InvalidRequest(String),
120    /// Internal operation failure retaining its historical CLI rendering.
121    #[error("invalid request: {0}")]
122    InternalFailure(String),
123    /// A doctor probe failed at a named integration boundary.
124    #[error("doctor {stage} check failed: {message}")]
125    DoctorFailure {
126        /// Stable integration stage suitable for structured diagnostics.
127        stage: &'static str,
128        /// Human-readable failure detail.
129        message: String,
130    },
131    #[error("invalid repository configuration: {0}")]
132    InvalidConfiguration(String),
133    #[error(
134        "SQLite index {database} belongs to repository {expected_repository}, not {actual_repository}"
135    )]
136    RepositoryMismatch {
137        database: PathBuf,
138        expected_repository: String,
139        actual_repository: PathBuf,
140    },
141    /// Retrieval request was intended for a different bound repository.
142    #[error("repository identity mismatch: expected {expected}, actual {actual}")]
143    RepositoryIdentityMismatch {
144        /// Opaque identity supplied by the caller.
145        expected: String,
146        /// Opaque identity of this server's canonical root.
147        actual: String,
148    },
149    #[error("stale cursor")]
150    StaleCursor,
151    #[error("request cancelled")]
152    Cancelled,
153    #[error("repository index is not ready")]
154    IndexNotReady,
155    #[error(
156        "reconciliation plan is stale: expected generation {expected}, found generation {actual}"
157    )]
158    StaleReconciliation { expected: u64, actual: u64 },
159    #[error("repository {0} could not stabilize because the index changed repeatedly; retry")]
160    RetryableConflict(RetryableOperation),
161    #[error("MCP indexing runtime stopped unexpectedly")]
162    McpRuntimeStopped,
163    #[error("required runtime capability is unavailable: {capability}")]
164    RuntimeCapabilityUnavailable {
165        capability: &'static str,
166        #[source]
167        source: Option<rusqlite::Error>,
168    },
169    #[error("I/O error: {0}")]
170    Io(#[from] std::io::Error),
171    #[error("SQLite error: {0}")]
172    Sqlite(#[from] rusqlite::Error),
173    #[error("database migration error: {0}")]
174    Migration(#[from] rusqlite_migration::Error),
175    #[error("SQLite connection pool error: {0}")]
176    ConnectionPool(#[from] r2d2::Error),
177    #[error("tree-sitter language error: {0}")]
178    TreeSitterLanguage(#[from] tree_sitter::LanguageError),
179    #[error("tree-sitter query error: {0}")]
180    TreeSitterQuery(#[from] tree_sitter::QueryError),
181    #[error("regex error: {0}")]
182    Regex(#[from] regex::Error),
183    #[error("glob error: {0}")]
184    Glob(#[from] globset::Error),
185    #[error("serialization error: {0}")]
186    Json(#[from] serde_json::Error),
187    #[error("background task failed: {0}")]
188    Join(#[from] tokio::task::JoinError),
189    #[error("index worker pool failed: {0}")]
190    ThreadPoolBuild(#[from] rayon::ThreadPoolBuildError),
191}
