84/// Detect the parser language from a file path based on its extension.
85pub fn language_by_path(path: impl AsRef<Path>) -> Option<String> {
86    let ext = path.as_ref().extension()?.to_str()?;
87    Some(match ext.to_lowercase().as_str() {
88        "c" => "c".to_string(),
89        "cc" | "cpp" | "cxx" | "h" | "hh" | "hpp" | "hxx" | "inl" | "ipp" | "tpp" => {
90            "cpp".to_string()
91        }
92        "java" => "java".to_string(),
93        "rs" => "rust".to_string(),
94        "py" | "pyi" => "python".to_string(),
95        "php" => "php".to_string(),
96        "rb" => "ruby".to_string(),
97        "js" | "jsx" | "mjs" | "cjs" => "javascript".to_string(),
98        "ts" | "mts" | "cts" => "typescript".to_string(),
99        "tsx" => "tsx".to_string(),
100        "go" => "go".to_string(),
101        _ => return None,
102    })
103}

105/// Parse a source file given its repository path and full text.
106///
107/// Files whose language is not supported still return `Ok`, but with an empty
108/// parse and `language: None` so callers can fall back to plain text indexing.
109pub fn parse(path: impl AsRef<Path>, source: &str) -> Result<ParseOutput> {
110    parse_with_cancellation(path, source, &CancellationToken::new())
111}

113pub(crate) fn parse_with_cancellation(
114    path: impl AsRef<Path>,
115    source: &str,
116    cancellation: &CancellationToken,
117) -> Result<ParseOutput> {
118    match language_by_path(path) {
119        Some(lang) => {
120            parse_language_with_cancellation(&lang, source, || cancellation.is_cancelled())
121        }
122        None if cancellation.is_cancelled() => Err(Error::Cancelled),
123        None => Ok(empty_parse()),
124    }
125}

127/// Parse source text for a known language name.
128pub fn parse_language(language: &str, source: &str) -> Result<ParseOutput> {
129    parse_language_with_cancellation(language, source, || false)

132// Per-thread cache of a configured tree-sitter `Parser` and compiled query
133// objects, keyed by language name. `Parser` and `Query` are not `Send`, so
134// they cannot be shared across the rayon worker pool. A thread-local avoids
135// recreating and recompiling them for every source file parsed on the same
136// thread.
137thread_local! {
138    static PARSER_CACHE: RefCell<Option<ParserCache>> = const { RefCell::new(None) };
139}
140
141struct ParserCache {
142    language: String,
143    parser: Parser,
144    tags_query: Query,
145    import_query: Option<Query>,
146}
147
148fn parse_language_with_cancellation(
149    language: &str,
150    source: &str,
151    mut is_cancelled: impl FnMut() -> bool,
152) -> Result<ParseOutput> {
153    let lang = language_object(language)
154        .ok_or_else(|| Error::UnsupportedLanguage(language.to_string()))?;
155
156    PARSER_CACHE.with(|cell| {
157        let mut cache = cell.borrow_mut();
158        let needs_init = match &*cache {
159            Some(c) => c.language != language,
160            None => true,
161        };
162        if needs_init {
163            let mut parser = Parser::new();
164            parser
165                .set_language(&lang)
166                .map_err(Error::TreeSitterLanguage)?;
167            let tags_query = build_tags_query(language, &lang)?;
168            let import_query = build_import_query(language, &lang)?;
169            *cache = Some(ParserCache {
170                language: language.to_string(),
171                parser,
172                tags_query,
173                import_query,
174            });
175        }
176        let cache = cache.as_mut().expect("cache was just initialized");
177
178        let tree = parse_tree(&mut cache.parser, source, &mut is_cancelled)?;
179        let root = tree.root_node();
180        let structurally_complete = !root.has_error();
181
182        let mut symbols = Vec::new();
183        let mut references = Vec::new();
184        let mut imports = Vec::new();
185
186        run_query(source, &cache.tags_query, root, &mut is_cancelled, |qm| {
187            process_tags_match(
188                language,
189                source,
190                &cache.tags_query,
191                qm,
192                &mut symbols,
193                &mut references,
194            );
195        })?;
196        if let Some(import_query) = &cache.import_query {
197            run_query(source, import_query, root, &mut is_cancelled, |qm| {
198                process_imports_match(source, import_query, qm, &mut imports);
199            })?;
200        }
201
202        if is_cancelled() {
203            return Err(Error::Cancelled);
204        }
205
206        deduplicate_symbols(&mut symbols);
207        compute_symbol_parents(&mut symbols);
208        compute_reference_enclosing(&symbols, &mut references);
209
210        Ok(ParseOutput {
211            language: Some(language.to_string()),
212            structurally_complete,
213            symbols,
214            references,
215            imports,
216        })
217    })
218}

230fn parse_tree(
231    parser: &mut Parser,
232    source: &str,
233    is_cancelled: &mut impl FnMut() -> bool,
234) -> Result<Tree> {
235    if is_cancelled() {
236        return Err(Error::Cancelled);
237    }
238
239    let bytes = source.as_bytes();
240    let mut input = |offset: usize, _| bytes.get(offset..).unwrap_or_default();
241    let tree = {
242        let mut progress = |_: &tree_sitter::ParseState| {
243            if is_cancelled() {
244                ControlFlow::Break(())
245            } else {
246                ControlFlow::Continue(())
247            }
248        };
249        let options = ParseOptions::new().progress_callback(&mut progress);
250        parser.parse_with_options(&mut input, None, Some(options))
251    };
252
253    match tree {
254        Some(tree) => Ok(tree),
255        None if is_cancelled() => Err(Error::Cancelled),
256        None => Err(Error::InternalFailure("parser returned None".into())),
257    }
258}

277fn build_tags_query(language: &str, lang: &Language) -> Result<Query> {
278    let base = match language {
279        "c" => tree_sitter_c::TAGS_QUERY,
280        "cpp" => tree_sitter_cpp::TAGS_QUERY,
281        "java" => tree_sitter_java::TAGS_QUERY,
282        "rust" => tree_sitter_rust::TAGS_QUERY,
283        "python" => tree_sitter_python::TAGS_QUERY,
284        "php" => tree_sitter_php::TAGS_QUERY,
285        "ruby" => tree_sitter_ruby::TAGS_QUERY,
286        "javascript" => tree_sitter_javascript::TAGS_QUERY,
287        // The TypeScript crate's query contains only TypeScript-specific
288        // additions. Its grammar inherits JavaScript definitions, so both
289        // query sets are required.
290        "typescript" | "tsx" => tree_sitter_javascript::TAGS_QUERY,
291        "go" => tree_sitter_go::TAGS_QUERY,
292        _ => return Err(Error::UnsupportedLanguage(language.to_string())),
293    };
294
295    let mut source = base.to_string();
296    match language {
297        "rust" => source.push_str(RUST_DEFS_QUERY),
298        "go" => source.push_str(GO_DEFS_QUERY),
299        "php" => source.push_str(PHP_REFS_QUERY),
300        "typescript" | "tsx" => source.push_str(tree_sitter_typescript::TAGS_QUERY),
301        _ => {}
302    }
303
304    Query::new(lang, &source).map_err(Error::TreeSitterQuery)
305}

307fn build_import_query(language: &str, lang: &Language) -> Result<Option<Query>> {
308    let src = match language {
309        "rust" => RUST_IMPORT_QUERY,
310        "python" => PYTHON_IMPORT_QUERY,
311        "javascript" | "typescript" | "tsx" => JS_IMPORT_QUERY,
312        "go" => GO_IMPORT_QUERY,
313        "c" | "cpp" | "java" | "php" | "ruby" => return Ok(None),
314        _ => return Err(Error::UnsupportedLanguage(language.to_string())),
315    };
316
317    Query::new(lang, src)
318        .map(Some)
319        .map_err(Error::TreeSitterQuery)
320}

322fn run_query<F>(
323    source: &str,
324    query: &Query,
325    root: Node,
326    is_cancelled: &mut impl FnMut() -> bool,
327    mut f: F,
328) -> Result<()>
329where
330    F: FnMut(&QueryMatch),
331{
332    if is_cancelled() {
333        return Err(Error::Cancelled);
334    }
335
336    let mut cursor = QueryCursor::new();
337    {
338        let mut progress = |_: &tree_sitter::QueryCursorState| {
339            if is_cancelled() {
340                ControlFlow::Break(())
341            } else {
342                ControlFlow::Continue(())
343            }
344        };
345        let options = QueryCursorOptions::new().progress_callback(&mut progress);
346        let mut matches = cursor.matches_with_options(query, root, source.as_bytes(), options);
347        while let Some(qm) = matches.next() {
348            f(qm);
349        }
350    }
351
352    if is_cancelled() {
353        Err(Error::Cancelled)
354    } else {
355        Ok(())
356    }
357}

359fn process_tags_match(
360    language: &str,
361    source: &str,
362    query: &Query,
363    qm: &QueryMatch,
364    symbols: &mut Vec<Symbol>,
365    references: &mut Vec<Reference>,
366) {
367    let capture_names = query.capture_names();
368    let mut name_node: Option<Node> = None;
369    let mut kind_captures = Vec::new();
370
371    for cap in qm.captures {
372        let cap_name = capture_names[cap.index as usize];
373        if cap_name == "name" {
374            name_node = Some(cap.node);
375        } else if let Some(prefix) = cap_name.strip_prefix("definition.") {
376            kind_captures.push((true, prefix, cap.node));
377        } else if let Some(prefix) = cap_name.strip_prefix("reference.") {
378            kind_captures.push((false, prefix, cap.node));
379        }
380    }
381
382    let Some(name_node) = name_node else {
383        return;
384    };
385
386    let name = node_text(source, name_node);
387
388    for (is_definition, kind, kind_node) in kind_captures {
389        if is_definition {
390            let kind_node = definition_extent(kind_node);
391            let (kind, parent) = canonical_definition(language, source, kind, kind_node);
392            let (start_line, end_line, start_byte, end_byte) = range_from_node(kind_node);
393            symbols.push(Symbol {
394                name: name.clone(),
395                kind,
396                parent,
397                signature: signature_from_node(source, kind_node),
398                start_line,
399                end_line,
400                start_byte,
401                end_byte,
402            });
403        } else {
404            let (start_line, end_line, start_byte, end_byte) = range_from_node(name_node);
405            references.push(Reference {
406                name: name.clone(),
407                kind: kind.to_string(),
408                role: ReferenceRole::Reference,
409                enclosing_symbol: None,
410                start_line,
411                end_line,
412                start_byte,
413                end_byte,
414            });
415        }
416    }
417}

419fn canonical_definition(
420    language: &str,
421    source: &str,
422    kind: &str,
423    node: Node<'_>,
424) -> (String, Option<String>) {
425    match (language, node.kind()) {
426        ("rust", "function_item") => rust_function_identity(source, node),
427        ("go", "method_declaration") => ("method".into(), go_method_owner(source, node)),
428        _ => (kind.to_string(), None),
429    }
430}

432fn rust_function_identity(source: &str, node: Node<'_>) -> (String, Option<String>) {
433    let Some(declarations) = node
434        .parent()
435        .filter(|parent| parent.kind() == "declaration_list")
436    else {
437        return ("function".into(), None);
438    };
439    match declarations.parent() {
440        Some(owner) if owner.kind() == "impl_item" => {
441            let owner = owner
442                .child_by_field_name("type")
443                .and_then(|node| base_type_name(source, node));
444            ("method".into(), owner)
445        }
446        Some(owner) if owner.kind() == "trait_item" => {
447            let owner = owner
448                .child_by_field_name("name")
449                .and_then(|node| base_type_name(source, node));
450            ("method".into(), owner)
451        }
452        _ => ("function".into(), None),
453    }
454}

456fn go_method_owner(source: &str, node: Node<'_>) -> Option<String> {
457    let receiver = node.child_by_field_name("receiver")?;
458    let mut cursor = receiver.walk();
459    receiver.named_children(&mut cursor).find_map(|parameter| {
460        parameter
461            .child_by_field_name("type")
462            .and_then(|node| base_type_name(source, node))
463    })
464}

466fn base_type_name(source: &str, node: Node<'_>) -> Option<String> {
467    match node.kind() {
468        "type_identifier" | "identifier" | "field_identifier" | "primitive_type" => {
469            let name = node_text(source, node);
470            (!name.is_empty()).then_some(name)
471        }
472        "generic_type" => node
473            .child_by_field_name("type")
474            .and_then(|node| base_type_name(source, node)),
475        "scoped_type_identifier" | "qualified_type" => node
476            .child_by_field_name("name")
477            .and_then(|node| base_type_name(source, node)),
478        "pointer_type" | "reference_type" | "parenthesized_type" | "bracketed_type"
479        | "abstract_type" => {
480            let mut cursor = node.walk();
481            node.named_children(&mut cursor)
482                .find_map(|child| base_type_name(source, child))
483        }
484        _ => node
485            .child_by_field_name("type")
486            .and_then(|node| base_type_name(source, node)),
487    }
488}

490fn definition_extent(node: Node<'_>) -> Node<'_> {
491    if node.kind() != "function_declarator" {
492        return node;
493    }
494    let mut current = node;
495    while let Some(parent) = current.parent() {
496        if parent.kind() == "function_definition" {
497            return parent;
498        }
499        if matches!(parent.kind(), "declaration" | "translation_unit") {
500            break;
501        }
502        current = parent;
503    }
504    node
505}

507fn signature_from_node(source: &str, node: Node) -> Option<String> {
508    const MAX_SIGNATURE_CHARS: usize = 512;
509
510    let end = first_body_start(node).unwrap_or_else(|| node.end_byte());
511    let raw = source.get(node.start_byte()..end)?.trim();
512    if raw.is_empty() {
513        return None;
514    }
515
516    let mut compact = String::with_capacity(raw.len().min(MAX_SIGNATURE_CHARS));
517    for part in raw.split_whitespace() {
518        if !compact.is_empty() {
519            compact.push(' ');
520        }
521        let remaining = MAX_SIGNATURE_CHARS.saturating_sub(compact.chars().count());
522        if remaining == 0 {
523            break;
524        }
525        compact.extend(part.chars().take(remaining));
526    }
527    (!compact.is_empty()).then_some(compact)
528}

530fn first_body_start(node: Node<'_>) -> Option<usize> {
531    let mut pending = vec![node];
532    let mut earliest = None;
533    while let Some(current) = pending.pop() {
534        if let Some(body) = current.child_by_field_name("body") {
535            earliest = Some(earliest.map_or(body.start_byte(), |start: usize| {
536                start.min(body.start_byte())
537            }));
538            continue;
539        }
540        let mut cursor = current.walk();
541        pending.extend(current.named_children(&mut cursor));
542    }
543    earliest
544}

546fn process_imports_match(source: &str, query: &Query, qm: &QueryMatch, imports: &mut Vec<Import>) {
547    let capture_names = query.capture_names();
548    let mut python_module = None;
549    let mut python_members = Vec::new();
550    let mut python_wildcard = false;
551    for cap in qm.captures {
552        let cap_name = capture_names[cap.index as usize];
553        let raw = node_text(source, cap.node);
554        let raw = unquote(&raw);
555        if raw.is_empty() {
556            continue;
557        }
558        match cap_name {
559            "raw" => push_import(imports, raw, cap.node.start_position().row + 1),
560            "python_module" => {
561                python_module = Some((raw.to_string(), cap.node.start_position().row + 1));
562            }
563            "python_member" => {
564                let member = cap.node.child_by_field_name("name").unwrap_or(cap.node);
565                python_members.push((node_text(source, member), member.start_position().row + 1));
566            }
567            "python_wildcard" => python_wildcard = true,
568            _ => {}
569        }
570    }
571
572    if let Some((module, line)) = python_module {
573        if python_wildcard {
574            push_import(imports, &module, line);
575        } else if module.bytes().all(|byte| byte == b'.') {
576            for (member, member_line) in python_members {
577                push_import(imports, &format!("{module}{member}"), member_line);
578            }
579        } else {
580            push_import(imports, &module, line);
581        }
582    }
583}

585fn push_import(imports: &mut Vec<Import>, raw_target: &str, line: usize) {
586    if imports
587        .iter()
588        .any(|import| import.line == line && import.raw_target == raw_target)
589    {
590        return;
591    }
592    imports.push(Import {
593        raw_target: raw_target.to_string(),
594        resolved_path: None,
595        line,
596    });
597}

599fn compute_symbol_parents(symbols: &mut [Symbol]) {
600    if symbols.is_empty() {
601        return;
602    }
603
604    let mut indices: Vec<usize> = (0..symbols.len()).collect();
605    indices.sort_by(|&a, &b| {
606        symbols[a]
607            .start_byte
608            .cmp(&symbols[b].start_byte)
609            .then_with(|| symbols[b].end_byte.cmp(&symbols[a].end_byte))
610    });
611
612    let mut stack: Vec<usize> = Vec::new();
613    for i in indices {
614        while let Some(&top) = stack.last() {
615            if strictly_contains(&symbols[top], &symbols[i]) {
616                break;
617            }
618            stack.pop();
619        }
620        if symbols[i].parent.is_none() {
621            symbols[i].parent = stack.last().map(|&top| symbols[top].name.clone());
622        }
623        stack.push(i);
624    }
625
626    symbols.sort_by(|a, b| {
627        a.start_byte
628            .cmp(&b.start_byte)
629            .then_with(|| a.end_byte.cmp(&b.end_byte))
630    });
631}

633fn deduplicate_symbols(symbols: &mut Vec<Symbol>) {
634    let mut seen = HashSet::with_capacity(symbols.len());
635    symbols.retain(|symbol| {
636        seen.insert((
637            symbol.kind.clone(),
638            symbol.name.clone(),
639            symbol.start_byte,
640            symbol.end_byte,
641        ))
642    });
643}

645fn strictly_contains(parent: &Symbol, child: &Symbol) -> bool {
646    parent.start_byte <= child.start_byte
647        && parent.end_byte >= child.end_byte
648        && (parent.start_byte < child.start_byte || parent.end_byte > child.end_byte)
649}

651fn compute_reference_enclosing(symbols: &[Symbol], references: &mut [Reference]) {
652    if symbols.is_empty() || references.is_empty() {
653        references.sort_by(|a, b| {
654            a.start_byte
655                .cmp(&b.start_byte)
656                .then_with(|| a.end_byte.cmp(&b.end_byte))
657        });
658        return;
659    }
660
661    let mut sym_indices: Vec<usize> = (0..symbols.len()).collect();
662    sym_indices.sort_by(|&a, &b| {
663        symbols[a]
664            .start_byte
665            .cmp(&symbols[b].start_byte)
666            .then_with(|| symbols[b].end_byte.cmp(&symbols[a].end_byte))
667    });
668
669    let mut ref_indices: Vec<usize> = (0..references.len()).collect();
670    ref_indices.sort_by_key(|&i| references[i].start_byte);
671
672    let mut sym_idx = 0;
673    let mut stack: Vec<usize> = Vec::new();
674
675    for ri in ref_indices {
676        let ref_start = references[ri].start_byte;
677        let ref_end = references[ri].end_byte;
678
679        while let Some(&top) = stack.last() {
680            if symbols[top].end_byte <= ref_start {
681                stack.pop();
682            } else {
683                break;
684            }
685        }
686
687        while sym_idx < sym_indices.len() {
688            let si = sym_indices[sym_idx];
689            if symbols[si].start_byte > ref_start {
690                break;
691            }
692            if symbols[si].end_byte > ref_start {
693                stack.push(si);
694            }
695            sym_idx += 1;
696        }
697
698        while let Some(&top) = stack.last() {
699            if symbols[top].end_byte < ref_end {
700                stack.pop();
701            } else {
702                break;
703            }
704        }
705
706        if let Some(&top) = stack.last() {
707            references[ri].enclosing_symbol = Some(symbols[top].name.clone());
708        }
709    }
710
711    references.sort_by(|a, b| {
712        a.start_byte
713            .cmp(&b.start_byte)
714            .then_with(|| a.end_byte.cmp(&b.end_byte))
715    });
716}

718fn range_from_node(node: Node) -> (usize, usize, usize, usize) {
719    let start = node.start_position();
720    let end = node.end_position();
721    (
722        start.row + 1,
723        end.row + 1,
724        node.start_byte(),
725        node.end_byte(),
726    )
727}
728
729fn node_text(source: &str, node: Node) -> String {
730    let bytes = source.as_bytes();
731    let start = node.start_byte();
732    let end = node.end_byte();
733    if start <= end && end <= bytes.len() {
734        String::from_utf8_lossy(&bytes[start..end]).into_owned()
735    } else {
736        String::new()
737    }
738}

740fn unquote(s: &str) -> &str {
741    let mut chars = s.chars();
742    if let (Some(first), Some(last)) = (chars.next(), chars.next_back())
743        && first == last
744        && (first == '"' || first == '\'' || first == '`')
745    {
746        return chars.as_str();
747    }
748    s
749}
