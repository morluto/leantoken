156#[serde(tag = "kind", rename_all = "snake_case", deny_unknown_fields)]
157enum FilesMcpOperation {
158    /// Return a compact hierarchy, optionally below one repository-relative directory.
159    Tree {
160        /// Optional repository-relative directory.
161        #[serde(default)]
162        #[schemars(length(max = 4096))]
163        path: Option<String>,
164        /// Maximum hierarchy depth below `path`.
165        #[serde(default)]
166        depth: Option<usize>,
167    },
168    /// Fuzzy-match repository paths and basenames.
169    Find {
170        /// Non-empty fuzzy filename or path query.
171        #[schemars(length(min = 1, max = 65536))]
172        query: String,
173    },
174    /// Match indexed repository paths with a glob.
175    Glob {
176        /// Non-empty glob pattern such as `src/**/*.rs`.
177        #[schemars(length(min = 1, max = 4096))]
178        pattern: String,
179    },
180}
181
182impl FilesMcpRequest {
183    fn into_parts(self) -> (FilesRequest, IndexConsistency) {
184        let (operation, path, query, pattern, depth) = match self.operation {
185            FilesMcpOperation::Tree { path, depth } => {
186                (FileOperation::Tree, path, None, None, depth)
187            }
188            FilesMcpOperation::Find { query } => {
189                (FileOperation::Find, None, Some(query), None, None)
190            }
191            FilesMcpOperation::Glob { pattern } => {
192                (FileOperation::Glob, None, None, Some(pattern), None)
193            }
194        };
195        (
196            FilesRequest {
197                operation,
198                path,
199                query,
200                pattern,
201                max_results: Some(self.max_results),
202                cursor: self.cursor,
203                depth,
204            },
205            self.consistency,
206        )
207    }
208}

388/// Wire representation used for successful MCP tool results.
389#[derive(Debug, Clone, Copy, Default, clap::ValueEnum, PartialEq, Eq)]
390pub enum McpResultMode {
391    /// Send JSON as both text and structured content for broad host compatibility.
392    #[default]
393    Dual,

400impl LeanTokenMcp {
401    #[must_use]
402    pub fn new(services: Arc<Services>) -> Self {
403        Self {
404            services: McpServices::ready(services),
405            result_mode: McpResultMode::Dual,
406        }
407    }
408
409    /// Construct a protocol-ready server before storage and indexing start.
410    #[must_use]
411    pub fn pending() -> (Self, McpServices) {
412        let services = McpServices::starting();
413        (
414            Self {
415                services: services.clone(),
416                result_mode: McpResultMode::Dual,
417            },
418            services,
419        )
420    }
421
422    /// Select the successful-result representation for this server instance.
423    #[must_use]
424    pub fn with_result_mode(mut self, result_mode: McpResultMode) -> Self {
425        self.result_mode = result_mode;
426        self
427    }
428
429    fn result<T: Serialize>(&self, value: T) -> Result<CallToolResult, ErrorData> {
430        tool_result(value, self.result_mode)
431    }
432
433    fn services(&self) -> std::result::Result<Arc<Services>, CallToolResult> {
434        match self.services.get() {
435            McpServiceState::Ready(services) => Ok(services),
436            McpServiceState::Starting => Err(self.retryable_result(RetryableToolResponse::new(
437                "index_starting",
438                "repository index is starting; retry the same call shortly",
439                500,
440            ))),
441            McpServiceState::Failed => Err(tool_unavailable(
442                "repository index is unavailable; check server logs and retry",
443            )),
444        }
445    }
446
447    fn retryable_result(&self, response: RetryableToolResponse) -> CallToolResult {
448        self.result(response).unwrap_or_else(|error| {
449            tracing::error!(%error, "MCP retry response serialization failed");
450            tool_unavailable("repository retrieval is temporarily unavailable; retry shortly")
451        })
452    }
453
454    fn service_result<T: Serialize>(
455        &self,
456        result: crate::Result<T>,
457    ) -> Result<CallToolResult, ErrorData> {
458        match result {
459            Ok(value) => self.result(value),
460            Err(crate::Error::IndexNotReady) => {
461                Ok(self.retryable_result(RetryableToolResponse::new(
462                    "index_building",
463                    "repository index is being built; retry the same call shortly",
464                    500,
465                )))
466            }
467            Err(crate::Error::RetryableConflict(_)) => {
468                Ok(self.retryable_result(RetryableToolResponse::new(
469                    "repository_changed",
470                    "repository index changed during retrieval; retry the same call",
471                    100,
472                )))
473            }
474            Err(error) => Err(into_mcp_error(error)),
475        }
476    }
477}

479impl McpServices {
480    fn starting() -> Self {
481        Self {
482            state: Arc::new(RwLock::new(McpServiceState::Starting)),
483            protocol_initialized: Arc::new(AtomicBool::new(false)),
484            initialized: Arc::new(tokio::sync::Notify::new()),
485        }
486    }
487
488    fn ready(services: Arc<Services>) -> Self {
489        Self {
490            state: Arc::new(RwLock::new(McpServiceState::Ready(services))),
491            protocol_initialized: Arc::new(AtomicBool::new(false)),
492            initialized: Arc::new(tokio::sync::Notify::new()),
493        }
494    }
495
496    fn get(&self) -> McpServiceState {
497        self.state
498            .read()
499            .unwrap_or_else(|poisoned| poisoned.into_inner())
500            .clone()
501    }
502
503    /// Make initialized retrieval services visible to MCP tool handlers.
504    pub fn set_ready(&self, services: Arc<Services>) {
505        *self
506            .state
507            .write()
508            .unwrap_or_else(|poisoned| poisoned.into_inner()) = McpServiceState::Ready(services);
509    }
510
511    /// Mark startup as failed without exposing internal diagnostics to clients.
512    pub fn set_failed(&self) {
513        *self
514            .state
515            .write()
516            .unwrap_or_else(|poisoned| poisoned.into_inner()) = McpServiceState::Failed;
517    }
518
519    fn mark_protocol_initialized(&self) {
520        self.protocol_initialized.store(true, Ordering::Release);
521        self.initialized.notify_waiters();
522    }
523
524    /// Wait until the client completes the MCP initialization phase.
525    pub async fn wait_initialized(&self) {
526        loop {
527            let notified = self.initialized.notified();
528            if self.protocol_initialized.load(Ordering::Acquire) {
529                return;
530            }
531            notified.await;
532        }
533    }
534}

655    mode: McpResultMode,
656) -> Result<CallToolResult, ErrorData> {
657    serde_json::to_value(value)
658        .map(|value| match mode {
659            McpResultMode::Dual => CallToolResult::structured(value),
660            McpResultMode::Text => {
661                CallToolResult::success(vec![ContentBlock::text(value.to_string())])
662            }
663            McpResultMode::Structured => {
664                let mut result = CallToolResult::default();
665                result.structured_content = Some(value);
666                result.is_error = Some(false);
667                result
668            }
669        })
670        .map_err(|error| {
671            tracing::error!(%error, "MCP response serialization failed");
672            ErrorData::internal_error("repository retrieval failed", None)
673        })
674}
675
676fn into_mcp_error(error: crate::Error) -> ErrorData {
677    match &error {
678        crate::Error::Cancelled => ErrorData::invalid_request("request cancelled", None),
679        crate::Error::RootNotFound(_)
680        | crate::Error::PathOutsideRoot(_)
681        | crate::Error::NotIndexed(_)
682        | crate::Error::LimitExceeded
683        | crate::Error::UnsupportedLanguage(_)
684        | crate::Error::InvalidRequest(_)
685        | crate::Error::StaleCursor
686        | crate::Error::Regex(_)
687        | crate::Error::Glob(_) => ErrorData::invalid_params(error.to_string(), None),
688        crate::Error::InvalidConfiguration(_) => {
689            tracing::error!(%error, "repository configuration is invalid");
690            ErrorData::internal_error("repository configuration is invalid", None)
691        }
692        crate::Error::RuntimeCapabilityUnavailable { .. } => {
693            tracing::error!(%error, "repository runtime is unavailable");
694            ErrorData::internal_error("repository runtime is unavailable", None)
695        }
696        crate::Error::IndexNotReady => {
697            ErrorData::internal_error("repository index is not ready", None)
698        }
699        crate::Error::RetryableConflict(_) => {
700            ErrorData::internal_error("repository operation should be retried", None)
701        }
702        _ => {
703            tracing::error!(%error, "MCP tool failed");
704            ErrorData::internal_error("repository retrieval failed", None)
705        }
706    }
707}
708
709fn tool_unavailable(message: &'static str) -> CallToolResult {
710    CallToolResult::error(vec![ContentBlock::text(message)])
711}

719/// Run the MCP server over stdio until the transport closes or SIGINT is received.
720pub async fn serve_stdio(services: Arc<Services>, result_mode: McpResultMode) -> crate::Result<()> {
721    let server = LeanTokenMcp::new(services).with_result_mode(result_mode);
722    serve_stdio_server(server).await
723}
724
725/// Run a prepared MCP server over stdio.
726pub async fn serve_stdio_server(server: LeanTokenMcp) -> crate::Result<()> {
727    let token = CancellationToken::new();
728
729    let signal_task = tokio::spawn({
730        let token = token.clone();
731        async move {
732            let _ = tokio::signal::ctrl_c().await;
733            token.cancel();
734        }
735    });
736
737    let result = async {
738        let service = match server.serve_with_ct(stdio(), token.child_token()).await {
739            Ok(service) => service,
740            Err(
741                rmcp::service::ServerInitializeError::ConnectionClosed(_)
742                | rmcp::service::ServerInitializeError::ExpectedInitializeRequest(None),
743            ) => return Ok(()),
744            Err(error) => return Err(crate::Error::Io(std::io::Error::other(error))),
745        };
746        service.waiting().await?;
747        Ok(())
748    }
749    .await;
750
751    signal_task.abort();
752    result
753}
