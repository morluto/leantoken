24/// Wire representation used for successful MCP tool results.
25#[derive(Debug, Clone, Copy, Default, clap::ValueEnum, PartialEq, Eq)]
26pub enum McpResultMode {
27    /// Send JSON as both text and structured content for broad host compatibility.
28    #[default]
29    Dual,
30    /// Send JSON only as text content for hosts that ignore structured content.
31    Text,
32    /// Send only structured content for hosts verified to support it.
33    Structured,
34}
35
36impl LeanTokenMcp {
37    #[must_use]
38    pub fn new(services: Arc<Services>) -> Self {
39        Self {
40            services,
41            result_mode: McpResultMode::Dual,
42        }
43    }
44
45    /// Select the successful-result representation for this server instance.
46    #[must_use]
47    pub fn with_result_mode(mut self, result_mode: McpResultMode) -> Self {
48        self.result_mode = result_mode;
49        self
50    }
51
52    fn result<T: Serialize>(&self, value: T) -> Result<CallToolResult, ErrorData> {
53        tool_result(value, self.result_mode)
54    }
55}

150/// Serialize a successful tool value using an explicit wire representation.
151pub fn tool_result<T: Serialize>(
152    value: T,
153    mode: McpResultMode,
154) -> Result<CallToolResult, ErrorData> {
155    serde_json::to_value(value)
156        .map(|value| match mode {
157            McpResultMode::Dual => CallToolResult::structured(value),
158            McpResultMode::Text => {
159                CallToolResult::success(vec![ContentBlock::text(value.to_string())])
160            }
161            McpResultMode::Structured => {
162                let mut result = CallToolResult::default();
163                result.structured_content = Some(value);
164                result.is_error = Some(false);
165                result
166            }
167        })
168        .map_err(|error| {
169            tracing::error!(%error, "MCP response serialization failed");
170            ErrorData::internal_error("repository retrieval failed", None)
171        })
172}
173
174fn into_mcp_error(error: crate::Error) -> ErrorData {
175    match &error {
176        crate::Error::Cancelled => ErrorData::invalid_request("request cancelled", None),
177        crate::Error::RootNotFound(_)
178        | crate::Error::PathOutsideRoot(_)
179        | crate::Error::NotIndexed(_)
180        | crate::Error::LimitExceeded
181        | crate::Error::UnsupportedLanguage(_)
182        | crate::Error::InvalidRequest(_)
183        | crate::Error::StaleCursor
184        | crate::Error::Regex(_)
185        | crate::Error::Glob(_) => ErrorData::invalid_params(error.to_string(), None),
186        _ => {
187            tracing::error!(%error, "MCP tool failed");
188            ErrorData::internal_error("repository retrieval failed", None)
189        }
190    }
191}

199/// Run the MCP server over stdio until the transport closes or SIGINT is received.
200pub async fn serve_stdio(services: Arc<Services>, result_mode: McpResultMode) -> crate::Result<()> {
201    let server = LeanTokenMcp::new(services).with_result_mode(result_mode);
202    let token = CancellationToken::new();
203
204    let signal_task = tokio::spawn({
205        let token = token.clone();
206        async move {
207            let _ = tokio::signal::ctrl_c().await;
208            token.cancel();
209        }
210    });
211
212    let result = async {
213        let service = match server.serve_with_ct(stdio(), token.child_token()).await {
214            Ok(service) => service,
215            Err(
216                rmcp::service::ServerInitializeError::ConnectionClosed(_)
217                | rmcp::service::ServerInitializeError::ExpectedInitializeRequest(None),
218            ) => return Ok(()),
219            Err(error) => return Err(crate::Error::Io(std::io::Error::other(error))),
220        };
221        service.waiting().await?;
222        Ok(())
223    }
224    .await;
225
226    signal_task.abort();
227    result
228}

271    fn result_modes_emit_only_the_selected_representations() {
272        let value = serde_json::json!({"answer": 42});
273        let dual = tool_result(value.clone(), McpResultMode::Dual).expect("dual");
274        let text = tool_result(value.clone(), McpResultMode::Text).expect("text");
275        let structured = tool_result(value, McpResultMode::Structured).expect("structured");
276
277        assert!(!dual.content.is_empty());
278        assert!(dual.structured_content.is_some());
279        assert!(!text.content.is_empty());
280        assert!(text.structured_content.is_none());
281        assert!(structured.content.is_empty());
282        assert!(structured.structured_content.is_some());
283    }
