34/// Shared readiness handle used by handshake-first MCP startup.
35#[derive(Debug, Clone)]
36pub struct McpServices {
37    state: Arc<RwLock<McpServiceState>>,
38    protocol_initialized: Arc<AtomicBool>,
39    initialized: Arc<tokio::sync::Notify>,

42/// Wire representation used for successful MCP tool results.
43#[derive(Debug, Clone, Copy, Default, clap::ValueEnum, PartialEq, Eq)]
44pub enum McpResultMode {
45    /// Send JSON as both text and structured content for broad host compatibility.
46    #[default]
47    Dual,
48    /// Send JSON only as text content for hosts that ignore structured content.
49    Text,
50    /// Send only structured content for hosts verified to support it.
51    Structured,
52}

54impl LeanTokenMcp {
55    #[must_use]
56    pub fn new(services: Arc<Services>) -> Self {
57        Self {
58            services: McpServices::ready(services),
59            result_mode: McpResultMode::Dual,
60        }
61    }
62
63    /// Construct a protocol-ready server before storage and indexing start.
64    #[must_use]
65    pub fn pending() -> (Self, McpServices) {
66        let services = McpServices::starting();
67        (
68            Self {
69                services: services.clone(),
70                result_mode: McpResultMode::Dual,
71            },
72            services,
73        )
74    }
75
76    /// Select the successful-result representation for this server instance.
77    #[must_use]
78    pub fn with_result_mode(mut self, result_mode: McpResultMode) -> Self {
79        self.result_mode = result_mode;
80        self
81    }
82
83    fn result<T: Serialize>(&self, value: T) -> Result<CallToolResult, ErrorData> {
84        tool_result(value, self.result_mode)
85    }
86
87    fn services(&self) -> std::result::Result<Arc<Services>, CallToolResult> {
88        match self.services.get() {
89            McpServiceState::Ready(services) => Ok(services),
90            McpServiceState::Starting => Err(tool_unavailable(
91                "repository index is starting; retry shortly",
92            )),
93            McpServiceState::Failed => Err(tool_unavailable(
94                "repository index is unavailable; check server logs and retry",
95            )),
96        }
97    }
98
99    fn service_result<T: Serialize>(
100        &self,
101        result: crate::Result<T>,
102    ) -> Result<CallToolResult, ErrorData> {
103        match result {
104            Ok(value) => self.result(value),
105            Err(crate::Error::IndexNotReady) => Ok(tool_unavailable(
106                "repository index is being built; retry shortly",
107            )),
108            Err(crate::Error::RetryableConflict(_)) => Ok(tool_unavailable(
109                "repository index changed during retrieval; retry shortly",
110            )),
111            Err(error) => Err(into_mcp_error(error)),
112        }
113    }
114}

116impl McpServices {
117    fn starting() -> Self {
118        Self {
119            state: Arc::new(RwLock::new(McpServiceState::Starting)),
120            protocol_initialized: Arc::new(AtomicBool::new(false)),
121            initialized: Arc::new(tokio::sync::Notify::new()),
122        }
123    }
124
125    fn ready(services: Arc<Services>) -> Self {
126        Self {
127            state: Arc::new(RwLock::new(McpServiceState::Ready(services))),
128            protocol_initialized: Arc::new(AtomicBool::new(false)),
129            initialized: Arc::new(tokio::sync::Notify::new()),
130        }
131    }
132
133    fn get(&self) -> McpServiceState {
134        self.state
135            .read()
136            .unwrap_or_else(|poisoned| poisoned.into_inner())
137            .clone()
138    }
139
140    /// Make initialized retrieval services visible to MCP tool handlers.
141    pub fn set_ready(&self, services: Arc<Services>) {
142        *self
143            .state
144            .write()
145            .unwrap_or_else(|poisoned| poisoned.into_inner()) = McpServiceState::Ready(services);
146    }
147
148    /// Mark startup as failed without exposing internal diagnostics to clients.
149    pub fn set_failed(&self) {
150        *self
151            .state
152            .write()
153            .unwrap_or_else(|poisoned| poisoned.into_inner()) = McpServiceState::Failed;
154    }
155
156    fn mark_protocol_initialized(&self) {
157        self.protocol_initialized.store(true, Ordering::Release);
158        self.initialized.notify_waiters();
159    }
160
161    /// Wait until the client completes the MCP initialization phase.
162    pub async fn wait_initialized(&self) {
163        loop {
164            let notified = self.initialized.notified();
165            if self.protocol_initialized.load(Ordering::Acquire) {
166                return;
167            }
168            notified.await;
169        }
170    }
171}

274/// Serialize a successful tool value using an explicit wire representation.
275pub fn tool_result<T: Serialize>(
276    value: T,
277    mode: McpResultMode,
278) -> Result<CallToolResult, ErrorData> {
279    serde_json::to_value(value)
280        .map(|value| match mode {
281            McpResultMode::Dual => CallToolResult::structured(value),
282            McpResultMode::Text => {
283                CallToolResult::success(vec![ContentBlock::text(value.to_string())])
284            }
285            McpResultMode::Structured => {
286                let mut result = CallToolResult::default();
287                result.structured_content = Some(value);
288                result.is_error = Some(false);
289                result
290            }
291        })
292        .map_err(|error| {
293            tracing::error!(%error, "MCP response serialization failed");
294            ErrorData::internal_error("repository retrieval failed", None)
295        })
296}

298fn into_mcp_error(error: crate::Error) -> ErrorData {
299    match &error {
300        crate::Error::Cancelled => ErrorData::invalid_request("request cancelled", None),
301        crate::Error::RootNotFound(_)
302        | crate::Error::PathOutsideRoot(_)
303        | crate::Error::NotIndexed(_)
304        | crate::Error::LimitExceeded
305        | crate::Error::UnsupportedLanguage(_)
306        | crate::Error::InvalidRequest(_)
307        | crate::Error::StaleCursor
308        | crate::Error::Regex(_)
309        | crate::Error::Glob(_) => ErrorData::invalid_params(error.to_string(), None),
310        crate::Error::IndexNotReady => {
311            ErrorData::internal_error("repository index is not ready", None)
312        }
313        crate::Error::RetryableConflict(_) => {
314            ErrorData::internal_error("repository operation should be retried", None)
315        }
316        _ => {
317            tracing::error!(%error, "MCP tool failed");
318            ErrorData::internal_error("repository retrieval failed", None)
319        }
320    }
321}

333/// Run the MCP server over stdio until the transport closes or SIGINT is received.
334pub async fn serve_stdio(services: Arc<Services>, result_mode: McpResultMode) -> crate::Result<()> {
335    let server = LeanTokenMcp::new(services).with_result_mode(result_mode);
336    serve_stdio_server(server).await
337}

339/// Run a prepared MCP server over stdio.
340pub async fn serve_stdio_server(server: LeanTokenMcp) -> crate::Result<()> {
341    let token = CancellationToken::new();
342
343    let signal_task = tokio::spawn({
344        let token = token.clone();
345        async move {
346            let _ = tokio::signal::ctrl_c().await;
347            token.cancel();
348        }
349    });
350
351    let result = async {
352        let service = match server.serve_with_ct(stdio(), token.child_token()).await {
353            Ok(service) => service,
354            Err(
355                rmcp::service::ServerInitializeError::ConnectionClosed(_)
356                | rmcp::service::ServerInitializeError::ExpectedInitializeRequest(None),
357            ) => return Ok(()),
358            Err(error) => return Err(crate::Error::Io(std::io::Error::other(error))),
359        };
360        service.waiting().await?;
361        Ok(())
362    }
363    .await;
364
365    signal_task.abort();
366    result
367}
