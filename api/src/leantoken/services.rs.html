52        let config = Arc::new(config);
53        let storage = match Storage::open(&config.database_path) {
54            Ok(storage) => storage,

84    pub async fn index(&self, rebuild: bool) -> Result<IndexResponse> {
85        let this = self.clone();
86        let active_reconciliations = Arc::clone(&self.active_reconciliations);
87        active_reconciliations.fetch_add(1, Ordering::AcqRel);
88        tokio::task::spawn_blocking(move || {
89            let _active = ActiveReconciliation(active_reconciliations);
90            let _guard = this
91                .index_lock
92                .lock()
93                .unwrap_or_else(|poisoned| poisoned.into_inner());
94            this.indexer.reconcile(rebuild)
95        })
96        .await?
97    }

219        &self,
220        request: FilesRequest,
221        cancellation: &CancellationToken,
222    ) -> Result<FilesResponse> {
223        check_cancelled(cancellation)?;
224        validate_optional_input(request.path.as_deref(), "path", MAX_PATH_BYTES)?;
225        validate_optional_input(request.query.as_deref(), "query", MAX_QUERY_BYTES)?;
226        validate_optional_input(request.pattern.as_deref(), "pattern", MAX_PATTERN_BYTES)?;
227        self.consistent(|generation| {

229            let offset = parse_cursor(request.cursor.as_deref(), generation)?;

231            let mut entries = match request.operation {
232                FileOperation::Tree => {
233                    tree_entries(&files, request.path.as_deref(), request.depth)?
234                }
235                FileOperation::Find => fuzzy_entries(&files, request.query.as_deref())?,
236                FileOperation::Glob => glob_entries(&files, request.pattern.as_deref())?,
237            };
238            let has_more = offset.saturating_add(limit) < entries.len();
239            entries = entries.into_iter().skip(offset).take(limit).collect();
240            Ok(FilesResponse {
241                entries,
242                meta: self.meta(
243                    generation,
244                    0,
245                    has_more.then(|| make_cursor(generation, offset + limit)),
246                ),
247            })
248        })

251    fn search_sync(
252        &self,
253        request: SearchRequest,
254        cancellation: &CancellationToken,
255    ) -> Result<SearchResponse> {
256        check_cancelled(cancellation)?;
257        if request.query.trim().is_empty() {
258            return Err(Error::InvalidRequest(
259                "search query must not be empty".into(),
260            ));
261        }
262        validate_input(&request.query, "search query", MAX_QUERY_BYTES)?;
263        validate_patterns(&request.include_paths)?;
264        validate_patterns(&request.exclude_paths)?;
265        validate_patterns(&request.focus_paths)?;
266        self.consistent(|generation| {
267            let limit = self.result_limit(request.max_results);
268            let token_limit = self.token_limit(request.max_tokens, self.config.default_read_tokens);
269            let offset = parse_cursor(request.cursor.as_deref(), generation)?;
270            let context_lines = request
271                .context_lines
272                .unwrap_or(self.config.context_lines)
273                .min(20);
274            let mut hits = Vec::new();
275
276            if matches!(
277                request.mode,
278                SearchMode::Auto | SearchMode::Identifier | SearchMode::Symbol
279            ) {
280                for hit in self.storage.search_symbols(
281                    &request.query,
282                    request.case_sensitive,
283                    limit * 4,
284                )? {
285                    check_cancelled(cancellation)?;
286                    if path_allowed(&hit.path, &request.include_paths, &request.exclude_paths)?
287                        && let Some(search_hit) =
288                            self.symbol_search_hit(hit, &request.query, context_lines)?
289                    {
290                        hits.push(search_hit);
291                    }
292                }
293            }
294            if matches!(
295                request.mode,
296                SearchMode::Auto | SearchMode::Identifier | SearchMode::Reference
297            ) {
298                for hit in self.storage.search_references(
299                    &request.query,
300                    request.case_sensitive,
301                    limit * 4,
302                )? {
303                    check_cancelled(cancellation)?;
304                    if path_allowed(&hit.path, &request.include_paths, &request.exclude_paths)?
305                        && let Some(search_hit) =
306                            self.reference_search_hit(hit, &request.query, context_lines)?
307                    {
308                        hits.push(search_hit);
309                    }
310                }
311            }
312
313            let lexical = match request.mode {
314                SearchMode::Regex => self.regex_hits(&request, limit * 20, cancellation)?,
315                SearchMode::Text | SearchMode::Auto => {
316                    if request.query.chars().count() >= 3 {
317                        self.storage.search_trigram(&request.query, limit * 8)?
318                    } else {
319                        self.storage
320                            .search_word(&fts_quote(&request.query), limit * 8)?
321                    }
322                }
323                SearchMode::Identifier => self
324                    .storage
325                    .search_word(&fts_quote(&request.query), limit * 8)?,
326                SearchMode::Symbol | SearchMode::Reference => Vec::new(),
327            };
328            for hit in lexical {
329                check_cancelled(cancellation)?;
330                if path_allowed(&hit.path, &request.include_paths, &request.exclude_paths)?
331                    && let Some(search_hit) = chunk_search_hit(
332                        hit,
333                        &request.query,
334                        request.case_sensitive,
335                        context_lines,
336                        matches!(request.mode, SearchMode::Regex),
337                    )?
338                {
339                    hits.push(search_hit);
340                }
341            }
342
343            apply_focus(&mut hits, &request.focus_paths)?;
344            hits.sort_by(|left, right| {
345                right
346                    .score
347                    .total_cmp(&left.score)
348                    .then_with(|| left.path.cmp(&right.path))
349                    .then_with(|| left.start_line.cmp(&right.start_line))
350            });
351            let mut seen = HashSet::new();
352            hits.retain(|hit| {
353                seen.insert((
354                    hit.path.clone(),
355                    hit.start_line,
356                    hit.end_line,
357                    hit.content_hash.clone(),
358                ))
359            });
360
361            let mut emitted_tokens = 0usize;
362            let mut selected = Vec::new();
363            let remaining = hits.len().saturating_sub(offset);
364            let mut consumed = 0usize;
365            for hit in hits.into_iter().skip(offset) {
366                check_cancelled(cancellation)?;
367                if selected.len() >= limit {
368                    break;
369                }
370                consumed += 1;
371                let count = self.config.tokenizer.count(&hit.excerpt);
372                if emitted_tokens.saturating_add(count) > token_limit {
373                    continue;
374                }
375                emitted_tokens += count;
376                selected.push(hit);
377            }
378            let has_more = consumed < remaining;
379            Ok(SearchResponse {
380                hits: selected,
381                meta: self.meta(
382                    generation,
383                    emitted_tokens,
384                    has_more.then(|| make_cursor(generation, offset + consumed)),
385                ),
386            })
387        })
388    }

390    fn outline_sync(
391        &self,
392        request: OutlineRequest,
393        cancellation: &CancellationToken,
394    ) -> Result<OutlineResponse> {
395        check_cancelled(cancellation)?;
396        if request.paths.is_empty() {
397            return Err(Error::InvalidRequest(
398                "outline requires at least one path".into(),
399            ));
400        }
401        if request.paths.len() > MAX_INPUT_ITEMS {
402            return Err(Error::LimitExceeded);
403        }
404        for path in &request.paths {
405            validate_input(path, "path", MAX_PATH_BYTES)?;
406        }
407        validate_optional_input(
408            request.symbol_name.as_deref(),
409            "symbol name",
410            MAX_PATTERN_BYTES,
411        )?;
412        validate_optional_input(
413            request.symbol_kind.as_deref(),
414            "symbol kind",
415            MAX_PATTERN_BYTES,
416        )?;
417        self.consistent(|generation| {
418            let limit = self.result_limit(request.max_results);
419            let token_limit = self.token_limit(request.max_tokens, self.config.default_read_tokens);
420            let mut remaining = limit;
421            let mut emitted_tokens = 0usize;
422            let mut files = Vec::new();
423            for path in &request.paths {
424                check_cancelled(cancellation)?;
425                validate_relative(path)?;
426                let file = self
427                    .storage
428                    .find_file(path)?
429                    .ok_or_else(|| Error::NotIndexed(path.clone()))?;
430                let mut symbols = self
431                    .storage
432                    .get_symbols_for_file_filtered(
433                        file.id,
434                        request.symbol_name.as_deref(),
435                        request.symbol_kind.as_deref(),
436                        remaining.max(1),
437                    )?
438                    .into_iter()
439                    .map(storage_symbol)
440                    .collect::<Vec<_>>();
441                symbols.retain(|symbol| {
442                    let cost = symbol
443                        .signature
444                        .as_deref()
445                        .map_or(1, |value| self.config.tokenizer.count(value));
446                    if remaining == 0 || emitted_tokens.saturating_add(cost) > token_limit {
447                        false
448                    } else {
449                        remaining -= 1;
450                        emitted_tokens += cost;
451                        true
452                    }
453                });
454                let mut imports = self
455                    .storage
456                    .get_imports_for_file(file.id, limit)?
457                    .into_iter()
458                    .map(|import| Import {
459                        raw_target: import.raw_target,
460                        resolved_path: import.resolved_path,
461                        line: import.line,
462                    })
463                    .collect::<Vec<_>>();
464                imports.retain(|import| {
465                    let cost = self.config.tokenizer.count(&import.raw_target)
466                        + import
467                            .resolved_path
468                            .as_deref()
469                            .map_or(0, |value| self.config.tokenizer.count(value));
470                    if remaining == 0 || emitted_tokens.saturating_add(cost) > token_limit {
471                        false
472                    } else {
473                        remaining -= 1;
474                        emitted_tokens += cost;
475                        true
476                    }
477                });
478                files.push(OutlineFile {
479                    path: file.path,
480                    language: file.language.clone(),
481                    structurally_complete: file.structurally_complete,
482                    symbols,
483                    imports,
484                });
485                if remaining == 0 {
486                    break;
487                }
488            }
489            Ok(OutlineResponse {
490                files,
491                meta: self.meta(generation, emitted_tokens, None),
492            })
493        })
494    }

518    fn read_at_generation(&self, request: &ReadRequest, generation: u64) -> Result<ReadResponse> {
519        let indexed = self
520            .storage
521            .find_file(&request.path)?
522            .ok_or_else(|| Error::NotIndexed(request.path.clone()))?;
523        let path = resolve_existing(&self.config.root, &request.path)?;
524        let bytes = fs::read(path)?;
525        let source = std::str::from_utf8(&bytes)
526            .map_err(|_| Error::InvalidRequest("requested file is not UTF-8 text".into()))?;
527        let line_count = source.lines().count().max(1);
528
529        let (start_line, requested_end) = if let Some(symbol_name) = &request.symbol {
530            let symbol = self
531                .storage
532                .find_symbol(indexed.id, symbol_name)?
533                .ok_or_else(|| Error::NotIndexed(format!("{}::{symbol_name}", request.path)))?;
534            (symbol.start_line, symbol.end_line)
535        } else {
536            (
537                request.start_line.unwrap_or(1),
538                request.end_line.unwrap_or(line_count),
539            )
540        };
541        if start_line == 0 || requested_end < start_line || start_line > line_count {
542            return Err(Error::InvalidRequest(
543                "invalid or out-of-range line range".into(),
544            ));
545        }
546        let requested_end = requested_end.min(line_count);
547        let content = crate::text::excerpt(source, start_line, requested_end);
548        let max_tokens = self.token_limit(request.max_tokens, self.config.default_read_tokens);
549        let (content, emitted_tokens) = self.config.tokenizer.truncate(&content, max_tokens);
550        let returned_lines = content
551            .lines()
552            .count()
553            .max(usize::from(!content.is_empty()));
554        let end_line = if returned_lines == 0 {
555            start_line
556        } else {
557            start_line + returned_lines - 1
558        };
559        let content_hash = hash(content);
560        let full_hash = crate::text::hash_bytes(&bytes);
561        let index_stale = indexed.content_hash != full_hash;
562        let indexed_hash = Some(indexed.content_hash);
563        let not_modified = request.expected_hash.as_deref() == Some(content_hash.as_str());
564
565        Ok(ReadResponse {
566            path: request.path.clone(),
567            status: if not_modified {
568                ReadStatus::NotModified
569            } else {
570                ReadStatus::Content
571            },
572            start_line,
573            end_line,
574            content: (!not_modified).then(|| content.to_string()),
575            content_hash,
576            indexed_hash,
577            index_stale,
578            meta: self.meta(
579                generation,
580                if not_modified { 0 } else { emitted_tokens },
581                None,
582            ),
583        })
584    }

607    fn context_sync(
608        &self,
609        request: ContextRequest,
610        cancellation: &CancellationToken,
611    ) -> Result<ContextResponse> {
612        check_cancelled(cancellation)?;
613        if request.task.trim().is_empty() || request.token_budget == 0 {
614            return Err(Error::InvalidRequest(
615                "context requires a task and positive token budget".into(),
616            ));
617        }
618        validate_input(&request.task, "task", MAX_QUERY_BYTES)?;
619        validate_patterns(&request.focus_paths)?;
620        validate_patterns(&request.exclude_paths)?;
621        if request.focus_symbols.len() > MAX_INPUT_ITEMS {
622            return Err(Error::LimitExceeded);
623        }
624        for symbol in &request.focus_symbols {
625            validate_input(symbol, "focus symbol", MAX_PATTERN_BYTES)?;
626        }
627        if request.token_budget > self.config.max_output_tokens {
628            return Err(Error::LimitExceeded);
629        }
630        if request.known_hashes.len() > MAX_INPUT_ITEMS {
631            return Err(Error::LimitExceeded);
632        }
633        for hash in &request.known_hashes {
634            validate_input(hash, "known hash", 128)?;
635        }
636        let changed_paths = git_changed_paths(&self.config.root, GIT_CHANGED_PATHS_MAX)
637            .unwrap_or_else(|error| {
638                tracing::debug!(%error, "working-tree signal unavailable");
639                HashSet::new()
640            });
641        self.consistent(|generation| {
642            let queries = context_queries(&request.task, 12);
643            let terms = queries
644                .iter()
645                .map(|query| query.value.clone())
646                .collect::<Vec<_>>();
647            let files = self.all_files(cancellation)?;
648            let file_map: HashMap<String, FileRecord> = files
649                .into_iter()
650                .map(|file| (file.path.clone(), file))
651                .collect();
652            let mut candidates = Vec::new();
653            let mut query_fusion = HashMap::<String, HashMap<String, f64>>::new();
654
655            // Workflow words such as `test` are useful path priors but terrible
656            // retrieval queries: nearly every test function becomes a high-
657            // scoring symbol candidate. Keep them out of candidate generation.
658            for query in queries.iter().filter(|query| query.value != "test") {
659                let term = &query.value;
660                let concept = query.fusion_key.as_deref().unwrap_or(term);
661                check_cancelled(cancellation)?;
662                for (rank, hit) in self
663                    .storage
664                    .search_symbols(term, false, 20)?
665                    .into_iter()
666                    .enumerate()
667                {
668                    check_cancelled(cancellation)?;
669                    if !path_allowed(&hit.path, &[], &request.exclude_paths)? {
670                        continue;

883                };

917                            .match_kind("import")
918                            .concept(seed_path, 0.2)
919                            .representation("import_neighbor")
920                            .path_score(context_path_score(&target_path, &terms, &request.task))
921                            .import_boost(1.0)
922                            .change_boost(change_boost),
923                    );
924                    neighbor_count += 1;
925                    if neighbor_count >= 24 {
926                        break;
927                    }
928                }
929                if neighbor_count >= 24 {
930                    break;
931                }
932            }
933
934            let mut response = ranking::select_with_tokenizer(
935                candidates,
936                &request,
937                generation,
938                self.config.tokenizer,
939            );
940            response.meta.freshness = self.freshness();
941            if response.fragments.is_empty() {
942                response
943                    .warnings
944                    .push("no relevant indexed evidence found".into());
945            }
946            Ok(response)
947        })

1022    fn stored_excerpt(
1023        &self,
1024        file_id: i64,
1025        start_line: usize,
1026        end_line: usize,
1027        context: usize,
1028        max_lines: usize,
1029    ) -> Result<Option<StoredExcerpt>> {
1030        let first = start_line.saturating_sub(context).max(1);
1031        let mut last = end_line.saturating_add(context);
1032        if max_lines > 0 && last.saturating_sub(first).saturating_add(1) > max_lines {
1033            last = first + max_lines - 1;
1034        }
1035        let selected = self.storage.get_chunks_overlapping(file_id, first, last)?;
1036        let (Some(first_chunk), Some(last_chunk)) = (selected.first(), selected.last()) else {
1037            return Ok(None);
1038        };
1039        last = last.min(last_chunk.end_line);
1040        let base_line = first_chunk.start_line;
1041        let mut combined = String::new();
1042        for chunk in &selected {
1043            combined.push_str(&chunk.content);
1044        }
1045        let local_start = first.saturating_sub(base_line) + 1;
1046        let local_end = last.saturating_sub(base_line) + 1;
1047        Ok(Some(StoredExcerpt {
1048            content: crate::text::excerpt(&combined, local_start, local_end),
1049            start_line: first,
1050            end_line: last,
1051        }))
1052    }

1054    fn adaptive_context_excerpt(
1055        &self,
1056        file_id: i64,
1057        declaration_start: usize,
1058        declaration_end: usize,
1059        matched_line: usize,
1060        token_budget: usize,
1061    ) -> Result<Option<StoredExcerpt>> {
1062        let Some(full) = self.stored_excerpt(file_id, declaration_start, declaration_end, 0, 0)?
1063        else {
1064            return Ok(None);
1065        };
1066        let full_tokens = self.config.tokenizer.count(&full.content).max(1);
1067        if full_tokens <= token_budget {
1068            return Ok(Some(full));
1069        }
1070
1071        let declaration_lines = declaration_end
1072            .saturating_sub(declaration_start)
1073            .saturating_add(1);
1074        let proportional_lines = declaration_lines
1075            .saturating_mul(token_budget)
1076            .saturating_div(full_tokens)
1077            .clamp(MIN_CONTEXT_RANGE_LINES, MAX_CONTEXT_RANGE_LINES)
1078            .min(declaration_lines);
1079        let before = proportional_lines / 3;
1080        let mut start = matched_line.saturating_sub(before).max(declaration_start);
1081        let mut end = start
1082            .saturating_add(proportional_lines.saturating_sub(1))
1083            .min(declaration_end);
1084        if end.saturating_sub(start).saturating_add(1) < proportional_lines {
1085            start = end
1086                .saturating_add(1)
1087                .saturating_sub(proportional_lines)
1088                .max(declaration_start);
1089        }
1090        end = start
1091            .saturating_add(proportional_lines.saturating_sub(1))
1092            .min(declaration_end);
1093        self.stored_excerpt(file_id, start, end, 0, 0)
1094    }

1261fn tree_entries(
1262    files: &[FileRecord],
1263    root: Option<&str>,
1264    depth: Option<usize>,
1265) -> Result<Vec<FileEntry>> {
1266    let root = root.unwrap_or("");
1267    if !root.is_empty() {
1268        validate_relative(root)?;
1269    }
1270    let root = root.trim_matches('/');
1271    let max_depth = depth.unwrap_or(usize::MAX);
1272    let root_depth = root.split('/').filter(|part| !part.is_empty()).count();
1273    let mut entries = BTreeMap::new();
1274    for file in files {
1275        if !root.is_empty() && file.path != root && !file.path.starts_with(&format!("{root}/")) {
1276            continue;
1277        }
1278        let parts = file.path.split('/').collect::<Vec<_>>();
1279        for index in 1..parts.len() {
1280            let path = parts[..index].join("/");
1281            let relative_depth = index.saturating_sub(root_depth);
1282            if relative_depth <= max_depth
1283                && (root.is_empty() || path == root || path.starts_with(&format!("{root}/")))
1284            {
1285                entries.entry(path.clone()).or_insert(FileEntry {
1286                    path,
1287                    kind: FileEntryKind::Directory,
1288                    language: None,
1289                    size_bytes: None,
1290                    score: None,
1291                });
1292            }
1293        }
1294        let file_depth = parts.len().saturating_sub(root_depth);
1295        if file_depth <= max_depth {
1296            entries.insert(
1297                file.path.clone(),
1298                FileEntry {
1299                    path: file.path.clone(),
1300                    kind: FileEntryKind::File,
1301                    language: file.language.clone(),
1302                    size_bytes: Some(file.size_bytes),
1303                    score: None,
1304                },
1305            );
1306        }
1307    }
1308    Ok(entries.into_values().collect())
1309}

1363fn chunk_search_hit(
1364    hit: ChunkHit,
1365    query: &str,
1366    case_sensitive: bool,
1367    context: usize,
1368    is_regex: bool,
1369) -> Result<Option<SearchHit>> {
1370    let byte_range = if is_regex {
1371        regex::RegexBuilder::new(query)
1372            .case_insensitive(!case_sensitive)
1373            .build()?
1374            .find(&hit.content)
1375            .map(|matched| (matched.start(), matched.end()))
1376    } else if case_sensitive {
1377        hit.content
1378            .find(query)
1379            .map(|start| (start, start + query.len()))
1380    } else {
1381        regex::RegexBuilder::new(&regex::escape(query))
1382            .case_insensitive(true)
1383            .build()?
1384            .find(&hit.content)
1385            .map(|matched| (matched.start(), matched.end()))
1386    };
1387    let Some((start, end)) = byte_range else {
1388        return Ok(None);
1389    };
1390    let (local_start, local_end) = byte_range_to_line_range(&hit.content, start, end);
1391    let excerpt_start = local_start.saturating_sub(context).max(1);
1392    let available_lines = hit.content.lines().count().max(1);
1393    let mut excerpt_end = local_end.saturating_add(context).min(available_lines);
1394    if excerpt_end.saturating_sub(excerpt_start).saturating_add(1) > 20 {
1395        excerpt_end = excerpt_start + 19;
1396    }
1397    let excerpt = excerpt(&hit.content, excerpt_start, excerpt_end);
1398    Ok(Some(SearchHit {
1399        path: hit.path,
1400        start_line: hit.start_line + excerpt_start - 1,
1401        end_line: hit.start_line + excerpt_end - 1,
1402        content_hash: hash(&excerpt),
1403        excerpt,
1404        match_kind: if is_regex {
1405            "regex".into()
1406        } else {
1407            "text".into()
1408        },
1409        role: None,
1410        symbol: None,
1411        enclosing_symbol: None,
1412        score: 3.0 + (-hit.score).max(0.0) * 1_000_000.0,
1413        score_reasons: vec![if is_regex {
1414            "regex match".into()
1415        } else {
1416            "text match".into()
1417        }],
1418    }))
1419}

1469fn context_path_score(path: &str, terms: &[String], task: &str) -> f64 {
1470    let path = path.to_lowercase();
1471    let mut score = terms
1472        .iter()
1473        .filter(|term| path.contains(term.to_ascii_lowercase().as_str()))
1474        .count() as f64;
1475    for code_token in code_tokens(task).into_iter().filter(|token| {
1476        token.contains("::")
1477            || token
1478                .split('.')
1479                .any(|part| part.chars().next().is_some_and(char::is_uppercase))
1480    }) {
1481        let matched_parts = expand_terms(&code_token)
1482            .into_iter()
1483            .map(|part| part.to_ascii_lowercase())
1484            .filter(|part| part.chars().count() >= 2 && path.contains(part))
1485            .collect::<HashSet<_>>()
1486            .len();
1487        if matched_parts >= 2 {
1488            #[allow(clippy::cast_precision_loss)]
1489            {
1490                score += (matched_parts * matched_parts) as f64;
1491            }
1492        }
1493    }
1494    for (language, component) in [
1495        ("javascript", "/js/"),
1496        ("typescript", "/ts/"),
1497        ("python", "/python/"),
1498        ("rust", "/rust/"),
1499        ("go", "/go/"),
1500    ] {
1501        if task_mentions_language(task, language) && format!("/{path}/").contains(component) {
1502            // An explicit language name in the task is strong repository-scope
1503            // evidence. Keep this above an exact-name match in another
1504            // language so common names such as `Point` do not dominate.
1505            score += 12.0;
1506        }
1507    }
1508    score
1509}

1511fn qualified_symbol_match(
1512    concept: &str,
1513    name: &str,
1514    parent: Option<&str>,
1515    signature: Option<&str>,
1516) -> f64 {
1517    if !concept.contains(['.', ':']) {
1518        return 0.0;
1519    }
1520    let parts = concept
1521        .split(|character: char| !character.is_alphanumeric() && character != '_')
1522        .flat_map(identifier_words)
1523        .map(|part| part.to_ascii_lowercase())
1524        .filter(|part| part.chars().count() >= 2)
1525        .collect::<HashSet<_>>();
1526    if parts.len() < 2 {
1527        return 0.0;
1528    }
1529    let haystack = format!(
1530        "{} {} {}",
1531        name,
1532        parent.unwrap_or_default(),
1533        signature.unwrap_or_default()
1534    )
1535    .to_ascii_lowercase();
1536    f64::from(parts.iter().all(|part| haystack.contains(part)))
1537}

1546fn context_queries(task: &str, limit: usize) -> Vec<ContextQuery> {
1547    if limit == 0 {
1548        return Vec::new();
1549    }
1550    let wants_tests = task_terms(task).iter().any(|term| is_test_term(term));
1551    let available = limit.saturating_sub(usize::from(wants_tests));
1552    let code_terms = code_tokens(task);
1553    let code_parts = code_terms
1554        .iter()
1555        .flat_map(|term| std::iter::once(term.clone()).chain(expand_terms(term)))
1556        .map(|term| term.to_ascii_lowercase())
1557        .collect::<HashSet<_>>();
1558    let mut prose = task_terms(task)
1559        .into_iter()
1560        .filter(|value| {
1561            !is_test_term(value)
1562                && !is_context_stop_word(value)
1563                && !code_parts.contains(&value.to_ascii_lowercase())
1564        })
1565        .enumerate()
1566        .collect::<Vec<_>>();
1567    prose.sort_by(|(left_index, left), (right_index, right)| {
1568        context_query_weight(right, false)
1569            .total_cmp(&context_query_weight(left, false))
1570            .then_with(|| left_index.cmp(right_index))
1571    });
1572
1573    let prose_reserve = prose.len().min(4).min(available);
1574    let exact_limit = available.saturating_sub(prose_reserve);
1575    let mut seen = HashSet::new();
1576    let mut terms = Vec::new();
1577    for code_term in code_terms.iter().take(exact_limit) {
1578        push_context_query(
1579            &mut terms,
1580            &mut seen,
1581            code_term.clone(),
1582            true,
1583            Some(code_term.to_ascii_lowercase()),
1584        );
1585    }
1586    for (_, value) in prose.iter().take(prose_reserve) {
1587        push_context_query(&mut terms, &mut seen, value.clone(), false, None);
1588    }
1589
1590    let mut expansion_round = 0usize;
1591    while terms.len() < available {
1592        let before = terms.len();
1593        for code_term in &code_terms {
1594            let expansions = expand_terms(code_term);
1595            if let Some(value) = expansions.get(expansion_round) {
1596                push_context_query(
1597                    &mut terms,
1598                    &mut seen,
1599                    value.clone(),
1600                    true,
1601                    Some(code_term.to_ascii_lowercase()),
1602                );
1603                if terms.len() == available {
1604                    break;
1605                }
1606            }
1607        }
1608        if terms.len() == before {
1609            break;
1610        }
1611        expansion_round += 1;
1612    }
1613    if wants_tests {
1614        terms.push(ContextQuery {
1615            value: "test".into(),
1616            weight: 0.2,
1617            fusion_key: None,
1618        });
1619    }
1620    terms
1621}

1657fn context_query_weight(term: &str, explicit_code_token: bool) -> f64 {
1658    if explicit_code_token {
1659        return if term.contains(['_', ':', '.', '-']) {
1660            1.0
1661        } else {
1662            0.95
1663        };
1664    }
1665    if term.contains(['_', ':', '.', '-']) {
1666        return 0.9;
1667    }
1668    match term.chars().count() {
1669        10.. => 0.8,
1670        7..=9 => 0.65,
1671        4..=6 => 0.45,
1672        _ => 0.25,
1673    }

1676fn record_query_hit(
1677    fusion: &mut HashMap<String, HashMap<String, f64>>,
1678    path: &str,
1679    fusion_key: &str,
1680    weight: f64,
1681    rank: usize,
1682) {
1683    if weight < 0.65 {
1684        return;
1685    }
1686    const RRF_K: f64 = 60.0;
1687    #[allow(clippy::cast_precision_loss)]
1688    let score = weight * RRF_K / (RRF_K + rank as f64 + 1.0);
1689    fusion
1690        .entry(path.to_owned())
1691        .or_default()
1692        .entry(fusion_key.to_owned())
1693        .and_modify(|current| *current = current.max(score))
1694        .or_insert(score);
1695}

1698    candidates: &mut [Candidate],

1712                .any(|kind| kind == "multi-query")
1713            {
1714                candidate.match_kinds.push("multi-query".into());
1715            }
1716        }
1717    }
1718}
